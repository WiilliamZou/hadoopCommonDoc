019package org.apache.hadoop.util;
020
021import javax.xml.transform.*;
022import javax.xml.transform.stream.*;
023
024import org.apache.hadoop.classification.InterfaceAudience;
025import org.apache.hadoop.classification.InterfaceStability;
026
027import java.io.*;
028
029/**
030 * General xml utilities.
031 *   
032 */
033@InterfaceAudience.Private
034@InterfaceStability.Unstable
035public class XMLUtils {
036  /**
037   * Transform input xml given a stylesheet.
038   * 
039   * @param styleSheet the style-sheet
040   * @param xml input xml data
041   * @param out output
042   * @throws TransformerConfigurationException
043   * @throws TransformerException
044   */
045  public static void transform(
046                               InputStream styleSheet, InputStream xml, Writer out
047                               ) 
048    throws TransformerConfigurationException, TransformerException {
049    // Instantiate a TransformerFactory
050    TransformerFactory tFactory = TransformerFactory.newInstance();
051
052    // Use the TransformerFactory to process the  
053    // stylesheet and generate a Transformer
054    Transformer transformer = tFactory.newTransformer(
055                                                      new StreamSource(styleSheet)
056                                                      );
057
058    // Use the Transformer to transform an XML Source 
059    // and send the output to a Result object.
060    transformer.transform(new StreamSource(xml), new StreamResult(out));
061  }
062}
