048/**
049 * <code>GenericOptionsParser</code> is a utility to parse command line
050 * arguments generic to the Hadoop framework. 
051 * 
052 * <code>GenericOptionsParser</code> recognizes several standard command
053 * line arguments, enabling applications to easily specify a namenode, a 
054 * ResourceManager, additional configuration resources etc.
055 * 
056 * <h4 id="GenericOptions">Generic Options</h4>
057 * 
058 * <p>The supported generic options are:</p>
059 * <p><blockquote><pre>
060 *     -conf &lt;configuration file&gt;     specify a configuration file
061 *     -D &lt;property=value&gt;            use value for given property
062 *     -fs &lt;local|namenode:port&gt;      specify a namenode
063 *     -jt &lt;local|resourcemanager:port&gt;    specify a ResourceManager
064 *     -files &lt;comma separated list of files&gt;    specify comma separated
065 *                            files to be copied to the map reduce cluster
066 *     -libjars &lt;comma separated list of jars&gt;   specify comma separated
067 *                            jar files to include in the classpath.
068 *     -archives &lt;comma separated list of archives&gt;    specify comma
069 *             separated archives to be unarchived on the compute machines.
070
071 * </pre></blockquote></p>
072 * 
073 * <p>The general command line syntax is:</p>
074 * <p><tt><pre>
075 * bin/hadoop command [genericOptions] [commandOptions]
076 * </pre></tt></p>
077 * 
078 * <p>Generic command line arguments <strong>might</strong> modify 
079 * <code>Configuration </code> objects, given to constructors.</p>
080 * 
081 * <p>The functionality is implemented using Commons CLI.</p>
082 *
083 * <p>Examples:</p>
084 * <p><blockquote><pre>
085 * $ bin/hadoop dfs -fs darwin:8020 -ls /data
086 * list /data directory in dfs with namenode darwin:8020
087 * 
088 * $ bin/hadoop dfs -D fs.default.name=darwin:8020 -ls /data
089 * list /data directory in dfs with namenode darwin:8020
090 *     
091 * $ bin/hadoop dfs -conf core-site.xml -conf hdfs-site.xml -ls /data
092 * list /data directory in dfs with multiple conf files specified.
093 *
094 * $ bin/hadoop job -D yarn.resourcemanager.address=darwin:8032 -submit job.xml
095 * submit a job to ResourceManager darwin:8032
096 *
097 * $ bin/hadoop job -jt darwin:8032 -submit job.xml
098 * submit a job to ResourceManager darwin:8032
099 *
100 * $ bin/hadoop job -jt local -submit job.xml
101 * submit a job to local runner
102 * 
103 * $ bin/hadoop jar -libjars testlib.jar 
104 * -archives test.tgz -files file.txt inputjar args
105 * job submission with libjars, files and archives
106 * </pre></blockquote></p>
107 *
108 * @see Tool
109 * @see ToolRunner
110 */
111@InterfaceAudience.Private
112@InterfaceStability.Evolving
113public class GenericOptionsParser {

140  /** 
141   * Create a <code>GenericOptionsParser<code> to parse only the generic Hadoop  
142   * arguments. 
143   * 
144   * The array of string arguments other than the generic arguments can be 
145   * obtained by {@link #getRemainingArgs()}.
146   * 
147   * @param conf the <code>Configuration</code> to modify.
148   * @param args command-line arguments.

152      throws IOException {
153    this(conf, new Options(), args);

156  /** 
157   * Create a <code>GenericOptionsParser</code> to parse given options as well 
158   * as generic Hadoop options. 
159   * 
160   * The resulting <code>CommandLine</code> object can be obtained by 
161   * {@link #getCommandLine()}.
162   * 
163   * @param conf the configuration to modify  
164   * @param options options built by the caller 
165   * @param args User-specified arguments
166   * @throws IOException 
167   */
168  public GenericOptionsParser(Configuration conf,
169      Options options, String[] args) throws IOException {
170    parseGeneralOptions(options, conf, args);
171    this.conf = conf;
172  }

174  /**
175   * Returns an array of Strings containing only application-specific arguments.
176   * 
177   * @return array of <code>String</code>s containing the un-parsed arguments
178   * or <strong>empty array</strong> if commandLine was not defined.
179   */
180  public String[] getRemainingArgs() {
181    return (commandLine == null) ? new String[]{} : commandLine.getArgs();
182  }

207  /**
208   * Specify properties of each generic option
209   */
210  @SuppressWarnings("static-access")
211  private static Options buildGeneralOptions(Options opts) {
212    Option fs = OptionBuilder.withArgName("local|namenode:port")
213    .hasArg()
214    .withDescription("specify a namenode")
215    .create("fs");
216    Option jt = OptionBuilder.withArgName("local|resourcemanager:port")
217    .hasArg()
218    .withDescription("specify a ResourceManager")
219    .create("jt");
220    Option oconf = OptionBuilder.withArgName("configuration file")
221    .hasArg()
222    .withDescription("specify an application configuration file")
223    .create("conf");
224    Option property = OptionBuilder.withArgName("property=value")
225    .hasArg()
226    .withDescription("use value for given property")
227    .create('D');
228    Option libjars = OptionBuilder.withArgName("paths")
229    .hasArg()
230    .withDescription("comma separated jar files to include in the classpath.")
231    .create("libjars");
232    Option files = OptionBuilder.withArgName("paths")
233    .hasArg()
234    .withDescription("comma separated files to be copied to the " +
235           "map reduce cluster")
236    .create("files");
237    Option archives = OptionBuilder.withArgName("paths")
238    .hasArg()
239    .withDescription("comma separated archives to be unarchived" +
240                     " on the compute machines.")
241    .create("archives");
242    
243    // file with security tokens
244    Option tokensFile = OptionBuilder.withArgName("tokensFile")
245    .hasArg()
246    .withDescription("name of the file with the tokens")
247    .create("tokenCacheFile");
248
249    opts.addOption(fs);
250    opts.addOption(jt);
251    opts.addOption(oconf);
252    opts.addOption(property);
253    opts.addOption(libjars);
254    opts.addOption(files);
255    opts.addOption(archives);
256    opts.addOption(tokensFile);
257
258    return opts;
259  }

261  /**
262   * Modify configuration according user-specified generic options
263   * @param conf Configuration to be modified
264   * @param line User-specified generic options
265   */
266  private void processGeneralOptions(Configuration conf,
267      CommandLine line) throws IOException {
268    if (line.hasOption("fs")) {
269      FileSystem.setDefaultUri(conf, line.getOptionValue("fs"));
270    }
271
272    if (line.hasOption("jt")) {
273      String optionValue = line.getOptionValue("jt");
274      if (optionValue.equalsIgnoreCase("local")) {
275        conf.set("mapreduce.framework.name", optionValue);
276      }
277
278      conf.set("yarn.resourcemanager.address", optionValue, 
279          "from -jt command line option");
280    }
281    if (line.hasOption("conf")) {
282      String[] values = line.getOptionValues("conf");
283      for(String value : values) {
284        conf.addResource(new Path(value));
285      }
286    }
287
288    if (line.hasOption('D')) {
289      String[] property = line.getOptionValues('D');
290      for(String prop : property) {
291        String[] keyval = prop.split("=", 2);
292        if (keyval.length == 2) {
293          conf.set(keyval[0], keyval[1], "from command line");
294        }
295      }
296    }
297
298    if (line.hasOption("libjars")) {
299      conf.set("tmpjars", 
300               validateFiles(line.getOptionValue("libjars"), conf),
301               "from -libjars command line option");
302      //setting libjars in client classpath
303      URL[] libjars = getLibJars(conf);
304      if(libjars!=null && libjars.length>0) {
305        conf.setClassLoader(new URLClassLoader(libjars, conf.getClassLoader()));
306        Thread.currentThread().setContextClassLoader(
307            new URLClassLoader(libjars, 
308                Thread.currentThread().getContextClassLoader()));
309      }
310    }
311    if (line.hasOption("files")) {
312      conf.set("tmpfiles", 
313               validateFiles(line.getOptionValue("files"), conf),
314               "from -files command line option");
315    }
316    if (line.hasOption("archives")) {
317      conf.set("tmparchives", 
318                validateFiles(line.getOptionValue("archives"), conf),
319                "from -archives command line option");
320    }
321    conf.setBoolean("mapreduce.client.genericoptionsparser.used", true);
322    
323    // tokensFile
324    if(line.hasOption("tokenCacheFile")) {
325      String fileName = line.getOptionValue("tokenCacheFile");
326      // check if the local file exists
327      FileSystem localFs = FileSystem.getLocal(conf);
328      Path p = localFs.makeQualified(new Path(fileName));
329      if (!localFs.exists(p)) {
330          throw new FileNotFoundException("File "+fileName+" does not exist.");
331      }
332      if(LOG.isDebugEnabled()) {
333        LOG.debug("setting conf tokensFile: " + fileName);
334      }
335      UserGroupInformation.getCurrentUser().addCredentials(
336          Credentials.readTokenStorageFile(p, conf));
337      conf.set("mapreduce.job.credentials.binary", p.toString(),
338               "from -tokenCacheFile command line option");
339
340    }
341  }

343  /**
344   * If libjars are set in the conf, parse the libjars.
345   * @param conf
346   * @return libjar urls
347   * @throws IOException
348   */
349  public static URL[] getLibJars(Configuration conf) throws IOException {
350    String jars = conf.get("tmpjars");
351    if(jars==null) {
352      return null;
353    }
354    String[] files = jars.split(",");
355    List<URL> cp = new ArrayList<URL>();
356    for (String file : files) {
357      Path tmp = new Path(file);
358      if (tmp.getFileSystem(conf).equals(FileSystem.getLocal(conf))) {
359        cp.add(FileSystem.getLocal(conf).pathToFile(tmp).toURI().toURL());
360      } else {
361        LOG.warn("The libjars file " + tmp + " is not on the local " +
362          "filesystem. Ignoring.");
363      }
364    }
365    return cp.toArray(new URL[0]);
366  }

368  /**
369   * takes input as a comma separated list of files
370   * and verifies if they exist. It defaults for file:///
371   * if the files specified do not have a scheme.
372   * it returns the paths uri converted defaulting to file:///.
373   * So an input of  /home/user/file1,/home/user/file2 would return
374   * file:///home/user/file1,file:///home/user/file2
375   * @param files
376   * @return
377   */
378  private String validateFiles(String files, Configuration conf) 
379      throws IOException  {
380    if (files == null) 
381      return null;
382    String[] fileArr = files.split(",");
383    if (fileArr.length == 0) {
384      throw new IllegalArgumentException("File name can't be empty string");
385    }
386    String[] finalArr = new String[fileArr.length];
387    for (int i =0; i < fileArr.length; i++) {
388      String tmp = fileArr[i];
389      if (tmp.isEmpty()) {
390        throw new IllegalArgumentException("File name can't be empty string");
391      }
392      String finalPath;
393      URI pathURI;
394      try {
395        pathURI = new URI(tmp);
396      } catch (URISyntaxException e) {
397        throw new IllegalArgumentException(e);
398      }
399      Path path = new Path(pathURI);
400      FileSystem localFs = FileSystem.getLocal(conf);
401      if (pathURI.getScheme() == null) {
402        //default to the local file system
403        //check if the file exists or not first
404        if (!localFs.exists(path)) {
405          throw new FileNotFoundException("File " + tmp + " does not exist.");
406        }
407        finalPath = path.makeQualified(localFs.getUri(),
408            localFs.getWorkingDirectory()).toString();
409      }
410      else {
411        // check if the file exists in this file system
412        // we need to recreate this filesystem object to copy
413        // these files to the file system ResourceManager is running
414        // on.
415        FileSystem fs = path.getFileSystem(conf);
416        if (!fs.exists(path)) {
417          throw new FileNotFoundException("File " + tmp + " does not exist.");
418        }
419        finalPath = path.makeQualified(fs.getUri(),
420            fs.getWorkingDirectory()).toString();
421      }
422      finalArr[i] = finalPath;
423    }
424    return StringUtils.arrayToString(finalArr);
425  }

427  /**
428   * Windows powershell and cmd can parse key=value themselves, because
429   * /pkey=value is same as /pkey value under windows. However this is not
430   * compatible with how we get arbitrary key values in -Dkey=value format.
431   * Under windows -D key=value or -Dkey=value might be passed as
432   * [-Dkey, value] or [-D key, value]. This method does undo these and
433   * return a modified args list by manually changing [-D, key, value]
434   * into [-D, key=value]
435   *
436   * @param args command line arguments
437   * @return fixed command line arguments that GnuParser can parse
438   */
439  private String[] preProcessForWindows(String[] args) {
440    if (!Shell.WINDOWS) {
441      return args;
442    }
443    if (args == null) {
444      return null;
445    }
446    List<String> newArgs = new ArrayList<String>(args.length);
447    for (int i=0; i < args.length; i++) {
448      String prop = null;
449      if (args[i].equals("-D")) {
450        newArgs.add(args[i]);
451        if (i < args.length - 1) {
452          prop = args[++i];
453        }
454      } else if (args[i].startsWith("-D")) {
455        prop = args[i];
456      } else {
457        newArgs.add(args[i]);
458      }
459      if (prop != null) {
460        if (prop.contains("=")) {
461          // everything good
462        } else {
463          if (i < args.length - 1) {
464            prop += "=" + args[++i];
465          }
466        }
467        newArgs.add(prop);
468      }
469    }
470
471    return newArgs.toArray(new String[newArgs.size()]);
472  }

474  /**
475   * Parse the user-specified options, get the generic options, and modify
476   * configuration accordingly
477   * @param opts Options to use for parsing args.
478   * @param conf Configuration to be modified
479   * @param args User-specified arguments
480   */
481  private void parseGeneralOptions(Options opts, Configuration conf, 
482      String[] args) throws IOException {
483    opts = buildGeneralOptions(opts);
484    CommandLineParser parser = new GnuParser();
485    try {
486      commandLine = parser.parse(opts, preProcessForWindows(args), true);
487      processGeneralOptions(conf, commandLine);
488    } catch(ParseException e) {
489      LOG.warn("options parsing failed: "+e.getMessage());
490
491      HelpFormatter formatter = new HelpFormatter();
492      formatter.printHelp("general options are: ", opts);
493    }
494  }

517  }
