040/**
041 * InodeTree implements a mount-table as a tree of inodes.
042 * It is used to implement ViewFs and ViewFileSystem.
043 * In order to use it the caller must subclass it and implement
044 * the abstract methods {@link #getTargetFileSystem(INodeDir)}, etc.
045 * 
046 * The mountable is initialized from the config variables as 
047 * specified in {@link ViewFs}
048 *
049 * @param <T> is AbstractFileSystem or FileSystem
050 * 
051 * The three main methods are
052 * {@link #InodeTreel(Configuration)} // constructor
053 * {@link #InodeTree(Configuration, String)} // constructor
054 * {@link #resolve(String, boolean)} 
055 */
056
057@InterfaceAudience.Private
058@InterfaceStability.Unstable 
059abstract class InodeTree<T> {
060  static enum ResultKind {isInternalDir, isExternalDir;};
061  static final Path SlashPath = new Path("/");
062  
063  final INodeDir<T> root; // the root of the mount table
064  
065  final String homedirPrefix; // the homedir config value for this mount table
066  
067  List<MountPoint<T>> mountPoints = new ArrayList<MountPoint<T>>();
068  
069  
070  static class MountPoint<T> {
071    String src;
072    INodeLink<T> target;
073    MountPoint(String srcPath, INodeLink<T> mountLink) {
074      src = srcPath;
075      target = mountLink;
076    }
077
078  }

080  /**
081   * Breaks file path into component names.
082   * @param path
083   * @return array of names component names
084   */
085  static String[] breakIntoPathComponents(final String path) {
086    return path == null ? null : path.split(Path.SEPARATOR);
087  } 
088  
089  /**
090   * Internal class for inode tree
091   * @param <T>
092   */
093  abstract static class INode<T> {
094    final String fullPath; // the full path to the root
095    public INode(String pathToNode, UserGroupInformation aUgi) {
096      fullPath = pathToNode;
097    }
098  };
099
100  /**
101   * Internal class to represent an internal dir of the mount table
102   * @param <T>
103   */
104  static class INodeDir<T> extends INode<T> {
105    final Map<String,INode<T>> children = new HashMap<String,INode<T>>();
106    T InodeDirFs =  null; // file system of this internal directory of mountT
107    boolean isRoot = false;
108    
109    INodeDir(final String pathToNode, final UserGroupInformation aUgi) {
110      super(pathToNode, aUgi);
111    }
112
113    INode<T> resolve(final String pathComponent) throws FileNotFoundException {
114      final INode<T> result = resolveInternal(pathComponent);
115      if (result == null) {
116        throw new FileNotFoundException();
117      }
118      return result;
119    }
120    
121    INode<T> resolveInternal(final String pathComponent)
122        throws FileNotFoundException {
123      return children.get(pathComponent);
124    }
125    
126    INodeDir<T> addDir(final String pathComponent,
127        final UserGroupInformation aUgi)
128      throws FileAlreadyExistsException {
129      if (children.containsKey(pathComponent)) {
130        throw new FileAlreadyExistsException();
131      }
132      final INodeDir<T> newDir = new INodeDir<T>(fullPath+ (isRoot ? "" : "/") + 
133          pathComponent, aUgi);
134      children.put(pathComponent, newDir);
135      return newDir;
136    }
137    
138    void addLink(final String pathComponent, final INodeLink<T> link)
139      throws FileAlreadyExistsException {
140      if (children.containsKey(pathComponent)) {
141        throw new FileAlreadyExistsException();
142      }
143      children.put(pathComponent, link);
144    }
145  }

147  /**
148   * In internal class to represent a mount link
149   * A mount link can be single dir link or a merge dir link.
150
151   * A merge dir link is  a merge (junction) of links to dirs:
152   * example : <merge of 2 dirs
153   *     /users -> hdfs:nn1//users
154   *     /users -> hdfs:nn2//users
155   * 
156   * For a merge, each target is checked to be dir when created but if target
157   * is changed later it is then ignored (a dir with null entries)
158   */
159  static class INodeLink<T> extends INode<T> {
160    final boolean isMergeLink; // true if MergeLink
161    final URI[] targetDirLinkList;
162    final T targetFileSystem;   // file system object created from the link.
163    
164    /**
165     * Construct a mergeLink
166     */
167    INodeLink(final String pathToNode, final UserGroupInformation aUgi,
168        final T targetMergeFs, final URI[] aTargetDirLinkList) {
169      super(pathToNode, aUgi);
170      targetFileSystem = targetMergeFs;
171      targetDirLinkList = aTargetDirLinkList;
172      isMergeLink = true;
173    }
174    
175    /**
176     * Construct a simple link (i.e. not a mergeLink)
177     */
178    INodeLink(final String pathToNode, final UserGroupInformation aUgi,
179        final T targetFs, final URI aTargetDirLink) {
180      super(pathToNode, aUgi);
181      targetFileSystem = targetFs;
182      targetDirLinkList = new URI[1];
183      targetDirLinkList[0] = aTargetDirLink;
184      isMergeLink = false;
185    }
186    
187    /**
188     * Get the target of the link
189     * If a merge link then it returned as "," separated URI list.
190     */
191    Path getTargetLink() {
192      // is merge link - use "," as separator between the merged URIs
193      //String result = targetDirLinkList[0].toString();
194      StringBuilder result = new StringBuilder(targetDirLinkList[0].toString());
195      for (int i=1; i < targetDirLinkList.length; ++i) { 
196        result.append(',').append(targetDirLinkList[i].toString());
197      }
198      return new Path(result.toString());
199    }
200  }

203  private void createLink(final String src, final String target,
204      final boolean isLinkMerge, final UserGroupInformation aUgi)
205      throws URISyntaxException, IOException,
206    FileAlreadyExistsException, UnsupportedFileSystemException {
207    // Validate that src is valid absolute path
208    final Path srcPath = new Path(src); 
209    if (!srcPath.isAbsoluteAndSchemeAuthorityNull()) {
210      throw new IOException("ViewFs:Non absolute mount name in config:" + src);
211    }
212 
213    final String[] srcPaths = breakIntoPathComponents(src);
214    INodeDir<T> curInode = root;
215    int i;
216    // Ignore first initial slash, process all except last component
217    for (i = 1; i < srcPaths.length-1; i++) {
218      final String iPath = srcPaths[i];
219      INode<T> nextInode = curInode.resolveInternal(iPath);
220      if (nextInode == null) {
221        INodeDir<T> newDir = curInode.addDir(iPath, aUgi);
222        newDir.InodeDirFs = getTargetFileSystem(newDir);
223        nextInode = newDir;
224      }
225      if (nextInode instanceof INodeLink) {
226        // Error - expected a dir but got a link
227        throw new FileAlreadyExistsException("Path " + nextInode.fullPath +
228            " already exists as link");
229      } else {
230        assert(nextInode instanceof INodeDir);
231        curInode = (INodeDir<T>) nextInode;
232      }
233    }
234    
235    // Now process the last component
236    // Add the link in 2 cases: does not exist or a link exists
237    String iPath = srcPaths[i];// last component
238    if (curInode.resolveInternal(iPath) != null) {
239      //  directory/link already exists
240      StringBuilder strB = new StringBuilder(srcPaths[0]);
241      for (int j = 1; j <= i; ++j) {
242        strB.append('/').append(srcPaths[j]);
243      }
244      throw new FileAlreadyExistsException("Path " + strB +
245            " already exists as dir; cannot create link here");
246    }
247    
248    final INodeLink<T> newLink;
249    final String fullPath = curInode.fullPath + (curInode == root ? "" : "/")
250        + iPath;
251    if (isLinkMerge) { // Target is list of URIs
252      String[] targetsList = StringUtils.getStrings(target);
253      URI[] targetsListURI = new URI[targetsList.length];
254      int k = 0;
255      for (String itarget : targetsList) {
256        targetsListURI[k++] = new URI(itarget);
257      }
258      newLink = new INodeLink<T>(fullPath, aUgi,
259          getTargetFileSystem(targetsListURI), targetsListURI);
260    } else {
261      newLink = new INodeLink<T>(fullPath, aUgi,
262          getTargetFileSystem(new URI(target)), new URI(target));
263    }
264    curInode.addLink(iPath, newLink);
265    mountPoints.add(new MountPoint<T>(src, newLink));
266  }

286  /**
287   * Create Inode Tree from the specified mount-table specified in Config
288   * @param config - the mount table keys are prefixed with 
289   *       FsConstants.CONFIG_VIEWFS_PREFIX
290   * @param viewName - the name of the mount table - if null use defaultMT name
291   * @throws UnsupportedFileSystemException
292   * @throws URISyntaxException
293   * @throws FileAlreadyExistsException
294   * @throws IOException
295   */
296  protected InodeTree(final Configuration config, final String viewName)
297      throws UnsupportedFileSystemException, URISyntaxException,
298    FileAlreadyExistsException, IOException { 
299    String vName = viewName;
300    if (vName == null) {
301      vName = Constants.CONFIG_VIEWFS_DEFAULT_MOUNT_TABLE;
302    }
303    homedirPrefix = ConfigUtil.getHomeDirValue(config, vName);
304    root = new INodeDir<T>("/", UserGroupInformation.getCurrentUser());
305    root.InodeDirFs = getTargetFileSystem(root);
306    root.isRoot = true;
307    
308    final String mtPrefix = Constants.CONFIG_VIEWFS_PREFIX + "." + 
309                            vName + ".";
310    final String linkPrefix = Constants.CONFIG_VIEWFS_LINK + ".";
311    final String linkMergePrefix = Constants.CONFIG_VIEWFS_LINK_MERGE + ".";
312    boolean gotMountTableEntry = false;
313    final UserGroupInformation ugi = UserGroupInformation.getCurrentUser();
314    for (Entry<String, String> si : config) {
315      final String key = si.getKey();
316      if (key.startsWith(mtPrefix)) {
317        gotMountTableEntry = true;
318        boolean isMergeLink = false;
319        String src = key.substring(mtPrefix.length());
320        if (src.startsWith(linkPrefix)) {
321          src = src.substring(linkPrefix.length());
322        } else if (src.startsWith(linkMergePrefix)) { // A merge link
323          isMergeLink = true;
324          src = src.substring(linkMergePrefix.length());
325        } else if (src.startsWith(Constants.CONFIG_VIEWFS_HOMEDIR)) {
326          // ignore - we set home dir from config
327          continue;
328        } else {
329          throw new IOException(
330          "ViewFs: Cannot initialize: Invalid entry in Mount table in config: "+ 
331          src);
332        }
333        final String target = si.getValue(); // link or merge link
334        createLink(src, target, isMergeLink, ugi); 
335      }
336    }
337    if (!gotMountTableEntry) {
338      throw new IOException(
339          "ViewFs: Cannot initialize: Empty Mount table in config for " + 
340             vName == null ? "viewfs:///" : ("viewfs://" + vName + "/"));
341    }
342  }

344  /**
345   * Resolve returns ResolveResult.
346   * The caller can continue the resolution of the remainingPath
347   * in the targetFileSystem.
348   * 
349   * If the input pathname leads to link to another file system then
350   * the targetFileSystem is the one denoted by the link (except it is
351   * file system chrooted to link target.
352   * If the input pathname leads to an internal mount-table entry then
353   * the target file system is one that represents the internal inode.
354   */
355  static class ResolveResult<T> {
356    final ResultKind kind;
357    final T targetFileSystem;
358    final String resolvedPath;
359    final Path remainingPath;   // to resolve in the target FileSystem
360    
361    ResolveResult(final ResultKind k, final T targetFs, final String resolveP,
362        final Path remainingP) {
363      kind = k;
364      targetFileSystem = targetFs;
365      resolvedPath = resolveP;
366      remainingPath = remainingP;  
367    }
368    
369    // isInternalDir of path resolution completed within the mount table 
370    boolean isInternalDir() {
371      return (kind == ResultKind.isInternalDir);
372    }
373  }
374  
375  /**
376   * Resolve the pathname p relative to root InodeDir
377   * @param p - inout path
378   * @param resolveLastComponent 
379   * @return ResolveResult which allows further resolution of the remaining path
380   * @throws FileNotFoundException
381   */
382  ResolveResult<T> resolve(final String p, final boolean resolveLastComponent)
383    throws FileNotFoundException {
384    // TO DO: - more efficient to not split the path, but simply compare
385    String[] path = breakIntoPathComponents(p); 
386    if (path.length <= 1) { // special case for when path is "/"
387      ResolveResult<T> res = 
388        new ResolveResult<T>(ResultKind.isInternalDir, 
389              root.InodeDirFs, root.fullPath, SlashPath);
390      return res;
391    }
392    
393    INodeDir<T> curInode = root;
394    int i;
395    // ignore first slash
396    for (i = 1; i < path.length - (resolveLastComponent ? 0 : 1); i++) {
397      INode<T> nextInode = curInode.resolveInternal(path[i]);
398      if (nextInode == null) {
399        StringBuilder failedAt = new StringBuilder(path[0]);
400        for ( int j = 1; j <=i; ++j) {
401          failedAt.append('/').append(path[j]);
402        }
403        throw (new FileNotFoundException(failedAt.toString()));      
404      }
405
406      if (nextInode instanceof INodeLink) {
407        final INodeLink<T> link = (INodeLink<T>) nextInode;
408        final Path remainingPath;
409        if (i >= path.length-1) {
410          remainingPath = SlashPath;
411        } else {
412          StringBuilder remainingPathStr = new StringBuilder("/" + path[i+1]);
413          for (int j = i+2; j< path.length; ++j) {
414            remainingPathStr.append('/').append(path[j]);
415          }
416          remainingPath = new Path(remainingPathStr.toString());
417        }
418        final ResolveResult<T> res = 
419          new ResolveResult<T>(ResultKind.isExternalDir,
420              link.targetFileSystem, nextInode.fullPath, remainingPath);
421        return res;
422      } else if (nextInode instanceof INodeDir) {
423        curInode = (INodeDir<T>) nextInode;
424      }
425    }
426
427    // We have resolved to an internal dir in mount table.
428    Path remainingPath;
429    if (resolveLastComponent) {
430      remainingPath = SlashPath;
431    } else {
432      // note we have taken care of when path is "/" above
433      // for internal dirs rem-path does not start with / since the lookup
434      // that follows will do a children.get(remaningPath) and will have to
435      // strip-out the initial /
436      StringBuilder remainingPathStr = new StringBuilder("/" + path[i]);
437      for (int j = i+1; j< path.length; ++j) {
438        remainingPathStr.append('/').append(path[j]);
439      }
440      remainingPath = new Path(remainingPathStr.toString());
441    }
442    final ResolveResult<T> res = 
443       new ResolveResult<T>(ResultKind.isInternalDir,
444           curInode.InodeDirFs, curInode.fullPath, remainingPath); 
445    return res;
446  }
