034/** An implementation of a round-robin scheme for disk allocation for creating
035 * files. The way it works is that it is kept track what disk was last
036 * allocated for a file write. For the current request, the next disk from
037 * the set of disks would be allocated if the free space on the disk is 
038 * sufficient enough to accommodate the file that is being considered for
039 * creation. If the space requirements cannot be met, the next disk in order
040 * would be tried and so on till a disk is found with sufficient capacity.
041 * Once a disk with sufficient space is identified, a check is done to make
042 * sure that the disk is writable. Also, there is an API provided that doesn't
043 * take the space requirements into consideration but just checks whether the
044 * disk under consideration is writable (this should be used for cases where
045 * the file size is not known apriori). An API is provided to read a path that
046 * was created earlier. That API works by doing a scan of all the disks for the
047 * input pathname.
048 * This implementation also provides the functionality of having multiple 
049 * allocators per JVM (one for each unique functionality or context, like 
050 * mapred, dfs-client, etc.). It ensures that there is only one instance of
051 * an allocator per context per JVM.
052 * Note:
053 * 1. The contexts referred above are actually the configuration items defined
054 * in the Configuration class like "mapred.local.dir" (for which we want to 
055 * control the dir allocations). The context-strings are exactly those 
056 * configuration items.
057 * 2. This implementation does not take into consideration cases where
058 * a disk becomes read-only or goes out of space while a file is being written
059 * to (disks are shared between multiple processes, and so the latter situation
060 * is probable).
061 * 3. In the class implementation, "Disk" is referred to as "Dir", which
062 * actually points to the configured directory on the Disk which will be the
063 * parent for all file write/read allocations.
064 */
065@InterfaceAudience.LimitedPrivate({"HDFS", "MapReduce"})
066@InterfaceStability.Unstable
067public class LocalDirAllocator {
068  
069  //A Map from the config item names like "mapred.local.dir"
070  //to the instance of the AllocatorPerContext. This
071  //is a static object to make sure there exists exactly one instance per JVM
072  private static Map <String, AllocatorPerContext> contexts = 
073                 new TreeMap<String, AllocatorPerContext>();
074  private String contextCfgItemName;
075
076  /** Used when size of file to be allocated is unknown. */
077  public static final int SIZE_UNKNOWN = -1;
078
079  /**Create an allocator object
080   * @param contextCfgItemName
081   */
082  public LocalDirAllocator(String contextCfgItemName) {
083    this.contextCfgItemName = contextCfgItemName;
084  }
085  
086  /** This method must be used to obtain the dir allocation context for a 
087   * particular value of the context name. The context name must be an item
088   * defined in the Configuration object for which we want to control the 
089   * dir allocations (e.g., <code>mapred.local.dir</code>). The method will
090   * create a context for that name if it doesn't already exist.
091   */
092  private AllocatorPerContext obtainContext(String contextCfgItemName) {
093    synchronized (contexts) {
094      AllocatorPerContext l = contexts.get(contextCfgItemName);
095      if (l == null) {
096        contexts.put(contextCfgItemName, 
097                    (l = new AllocatorPerContext(contextCfgItemName)));
098      }
099      return l;
100    }
101  }

245  private static class AllocatorPerContext {
246
247    private final Log LOG =
248      LogFactory.getLog(AllocatorPerContext.class);
249
250    private int dirNumLastAccessed;
251    private Random dirIndexRandomizer = new Random();
252    private FileSystem localFS;
253    private DF[] dirDF;
254    private String contextCfgItemName;
255    private String[] localDirs;
256    private String savedLocalDirs = "";
257
258    public AllocatorPerContext(String contextCfgItemName) {
259      this.contextCfgItemName = contextCfgItemName;
260    }
261
262    /** This method gets called everytime before any read/write to make sure
263     * that any change to localDirs is reflected immediately.
264     */
265    private synchronized void confChanged(Configuration conf) 
266        throws IOException {
267      String newLocalDirs = conf.get(contextCfgItemName);
268      if (!newLocalDirs.equals(savedLocalDirs)) {
269        localDirs = StringUtils.getTrimmedStrings(newLocalDirs);
270        localFS = FileSystem.getLocal(conf);
271        int numDirs = localDirs.length;
272        ArrayList<String> dirs = new ArrayList<String>(numDirs);
273        ArrayList<DF> dfList = new ArrayList<DF>(numDirs);
274        for (int i = 0; i < numDirs; i++) {
275          try {
276            // filter problematic directories
277            Path tmpDir = new Path(localDirs[i]);
278            if(localFS.mkdirs(tmpDir)|| localFS.exists(tmpDir)) {
279              try {
280
281                File tmpFile = tmpDir.isAbsolute()
282                  ? new File(localFS.makeQualified(tmpDir).toUri())
283                  : new File(localDirs[i]);
284
285                DiskChecker.checkDir(tmpFile);
286                dirs.add(tmpFile.getPath());
287                dfList.add(new DF(tmpFile, 30000));
288
289              } catch (DiskErrorException de) {
290                LOG.warn( localDirs[i] + " is not writable\n", de);
291              }
292            } else {
293              LOG.warn( "Failed to create " + localDirs[i]);
294            }
295          } catch (IOException ie) { 
296            LOG.warn( "Failed to create " + localDirs[i] + ": " +
297                ie.getMessage() + "\n", ie);
298          } //ignore
299        }
300        localDirs = dirs.toArray(new String[dirs.size()]);
301        dirDF = dfList.toArray(new DF[dirs.size()]);
302        savedLocalDirs = newLocalDirs;
303        
304        // randomize the first disk picked in the round-robin selection 
305        dirNumLastAccessed = dirIndexRandomizer.nextInt(dirs.size());
306      }
307    }
308
309    private Path createPath(String path, 
310        boolean checkWrite) throws IOException {
311      Path file = new Path(new Path(localDirs[dirNumLastAccessed]),
312                                    path);
313      if (checkWrite) {
314        //check whether we are able to create a directory here. If the disk
315        //happens to be RDONLY we will fail
316        try {
317          DiskChecker.checkDir(new File(file.getParent().toUri().getPath()));
318          return file;
319        } catch (DiskErrorException d) {
320          LOG.warn("Disk Error Exception: ", d);
321          return null;
322        }
323      }
324      return file;
325    }
326
327    /**
328     * Get the current directory index.
329     * @return the current directory index.
330     */
331    int getCurrentDirectoryIndex() {
332      return dirNumLastAccessed;
333    }

335    /** Get a path from the local FS. If size is known, we go
336     *  round-robin over the set of disks (via the configured dirs) and return
337     *  the first complete path which has enough space.
338     *  
339     *  If size is not known, use roulette selection -- pick directories
340     *  with probability proportional to their available space.
341     */
342    public synchronized Path getLocalPathForWrite(String pathStr, long size, 
343        Configuration conf, boolean checkWrite) throws IOException {
344      confChanged(conf);
345      int numDirs = localDirs.length;
346      int numDirsSearched = 0;
347      //remove the leading slash from the path (to make sure that the uri
348      //resolution results in a valid path on the dir being checked)
349      if (pathStr.startsWith("/")) {
350        pathStr = pathStr.substring(1);
351      }
352      Path returnPath = null;
353      
354      if(size == SIZE_UNKNOWN) {  //do roulette selection: pick dir with probability 
355                    //proportional to available size
356        long[] availableOnDisk = new long[dirDF.length];
357        long totalAvailable = 0;
358        
359            //build the "roulette wheel"
360        for(int i =0; i < dirDF.length; ++i) {
361          availableOnDisk[i] = dirDF[i].getAvailable();
362          totalAvailable += availableOnDisk[i];
363        }
364
365        if (totalAvailable == 0){
366          throw new DiskErrorException("No space available in any of the local directories.");
367        }
368
369        // Keep rolling the wheel till we get a valid path
370        Random r = new java.util.Random();
371        while (numDirsSearched < numDirs && returnPath == null) {
372          long randomPosition = (r.nextLong() >>> 1) % totalAvailable;
373          int dir = 0;
374          while (randomPosition > availableOnDisk[dir]) {
375            randomPosition -= availableOnDisk[dir];
376            dir++;
377          }
378          dirNumLastAccessed = dir;
379          returnPath = createPath(pathStr, checkWrite);
380          if (returnPath == null) {
381            totalAvailable -= availableOnDisk[dir];
382            availableOnDisk[dir] = 0; // skip this disk
383            numDirsSearched++;
384          }
385        }
386      } else {
387        while (numDirsSearched < numDirs && returnPath == null) {
388          long capacity = dirDF[dirNumLastAccessed].getAvailable();
389          if (capacity > size) {
390            returnPath = createPath(pathStr, checkWrite);
391          }
392          dirNumLastAccessed++;
393          dirNumLastAccessed = dirNumLastAccessed % numDirs; 
394          numDirsSearched++;
395        } 
396      }
397      if (returnPath != null) {
398        return returnPath;
399      }
400      
401      //no path found
402      throw new DiskErrorException("Could not find any valid local " +
403          "directory for " + pathStr);
404    }
405
406    /** Creates a file on the local FS. Pass size as 
407     * {@link LocalDirAllocator.SIZE_UNKNOWN} if not known apriori. We
408     *  round-robin over the set of disks (via the configured dirs) and return
409     *  a file on the first path which has enough space. The file is guaranteed
410     *  to go away when the JVM exits.
411     */
412    public File createTmpFileForWrite(String pathStr, long size, 
413        Configuration conf) throws IOException {
414
415      // find an appropriate directory
416      Path path = getLocalPathForWrite(pathStr, size, conf, true);
417      File dir = new File(path.getParent().toUri().getPath());
418      String prefix = path.getName();
419
420      // create a temp file on this directory
421      File result = File.createTempFile(prefix, null, dir);
422      result.deleteOnExit();
423      return result;
424    }

426    /** Get a path from the local FS for reading. We search through all the
427     *  configured dirs for the file's existence and return the complete
428     *  path to the file when we find one 
429     */
430    public synchronized Path getLocalPathToRead(String pathStr, 
431        Configuration conf) throws IOException {
432      confChanged(conf);
433      int numDirs = localDirs.length;
434      int numDirsSearched = 0;
435      //remove the leading slash from the path (to make sure that the uri
436      //resolution results in a valid path on the dir being checked)
437      if (pathStr.startsWith("/")) {
438        pathStr = pathStr.substring(1);
439      }
440      while (numDirsSearched < numDirs) {
441        Path file = new Path(localDirs[numDirsSearched], pathStr);
442        if (localFS.exists(file)) {
443          return file;
444        }
445        numDirsSearched++;
446      }
447
448      //no path found
449      throw new DiskErrorException ("Could not find " + pathStr +" in any of" +
450      " the configured local directories");
451    }

453    private static class PathIterator implements Iterator<Path>, Iterable<Path> {
454      private final FileSystem fs;
455      private final String pathStr;
456      private int i = 0;
457      private final String[] rootDirs;
458      private Path next = null;
459
460      private PathIterator(FileSystem fs, String pathStr, String[] rootDirs)
461          throws IOException {
462        this.fs = fs;
463        this.pathStr = pathStr;
464        this.rootDirs = rootDirs;
465        advance();
466      }
467
468      @Override
469      public boolean hasNext() {
470        return next != null;
471      }
472
473      private void advance() throws IOException {
474        while (i < rootDirs.length) {
475          next = new Path(rootDirs[i++], pathStr);
476          if (fs.exists(next)) {
477            return;
478          }
479        }
480        next = null;
481      }
482
483      @Override
484      public Path next() {
485        final Path result = next;
486        try {
487          advance();
488        } catch (IOException ie) {
489          throw new RuntimeException("Can't check existance of " + next, ie);
490        }
491        if (result == null) {
492          throw new NoSuchElementException();
493        }
494        return result;
495      }
496
497      @Override
498      public void remove() {
499        throw new UnsupportedOperationException("read only iterator");
500      }
501
502      @Override
503      public Iterator<Path> iterator() {
504        return this;
505      }
506    }

508    /**
509     * Get all of the paths that currently exist in the working directories.
510     * @param pathStr the path underneath the roots
511     * @param conf the configuration to look up the roots in
512     * @return all of the paths that exist under any of the roots
513     * @throws IOException
514     */
515    synchronized Iterable<Path> getAllLocalPathsToRead(String pathStr,
516        Configuration conf) throws IOException {
517      confChanged(conf);
518      if (pathStr.startsWith("/")) {
519        pathStr = pathStr.substring(1);
520      }
521      return new PathIterator(localFS, pathStr, localDirs);
522    }
523
524    /** We search through all the configured dirs for the file's existence
525     *  and return true when we find one 
526     */
527    public synchronized boolean ifExists(String pathStr,Configuration conf) {
528      try {
529        int numDirs = localDirs.length;
530        int numDirsSearched = 0;
531        //remove the leading slash from the path (to make sure that the uri
532        //resolution results in a valid path on the dir being checked)
533        if (pathStr.startsWith("/")) {
534          pathStr = pathStr.substring(1);
535        }
536        while (numDirsSearched < numDirs) {
537          Path file = new Path(localDirs[numDirsSearched], pathStr);
538          if (localFS.exists(file)) {
539            return true;
540          }
541          numDirsSearched++;
542        }
543      } catch (IOException e) {
544        // IGNORE and try again
545      }
546      return false;
547    }
