033/**
034 * This is a generic input stream for verifying checksums for
035 * data before it is read by a user.
036 */
037@InterfaceAudience.LimitedPrivate({"HDFS"})
038@InterfaceStability.Unstable
039abstract public class FSInputChecker extends FSInputStream {
040  public static final Log LOG 
041  = LogFactory.getLog(FSInputChecker.class);
042  
043  /** The file name from which data is read from */
044  protected Path file;
045  private Checksum sum;
046  private boolean verifyChecksum = true;
047  private int maxChunkSize; // data bytes for checksum (eg 512)
048  private byte[] buf; // buffer for non-chunk-aligned reading
049  private byte[] checksum;
050  private IntBuffer checksumInts; // wrapper on checksum buffer
051  private int pos; // the position of the reader inside buf
052  private int count; // the number of bytes currently in buf
053  
054  private int numOfRetries;
055  
056  // cached file position
057  // this should always be a multiple of maxChunkSize
058  private long chunkPos = 0;
059
060  // Number of checksum chunks that can be read at once into a user
061  // buffer. Chosen by benchmarks - higher values do not reduce
062  // CPU usage. The size of the data reads made to the underlying stream
063  // will be CHUNKS_PER_READ * maxChunkSize.
064  private static final int CHUNKS_PER_READ = 32;
065  protected static final int CHECKSUM_SIZE = 4; // 32-bit checksum
066
067  /** Constructor
068   * 
069   * @param file The name of the file to be read
070   * @param numOfRetries Number of read retries when ChecksumError occurs
071   */
072  protected FSInputChecker( Path file, int numOfRetries) {
073    this.file = file;
074    this.numOfRetries = numOfRetries;
075  }
076  
077  /** Constructor
078   * 
079   * @param file The name of the file to be read
080   * @param numOfRetries Number of read retries when ChecksumError occurs
081   * @param sum the type of Checksum engine
082   * @param chunkSize maximun chunk size
083   * @param checksumSize the number byte of each checksum
084   */
085  protected FSInputChecker( Path file, int numOfRetries, 
086      boolean verifyChecksum, Checksum sum, int chunkSize, int checksumSize ) {
087    this(file, numOfRetries);
088    set(verifyChecksum, sum, chunkSize, checksumSize);
089  }
090  
091  /**
092   * Reads in checksum chunks into <code>buf</code> at <code>offset</code>
093   * and checksum into <code>checksum</code>.
094   * Since checksums can be disabled, there are two cases implementors need
095   * to worry about:
096   *
097   *  (a) needChecksum() will return false:
098   *     - len can be any positive value
099   *     - checksum will be null
100   *     Implementors should simply pass through to the underlying data stream.
101   * or
102   *  (b) needChecksum() will return true:
103   *    - len >= maxChunkSize
104   *    - checksum.length is a multiple of CHECKSUM_SIZE
105   *    Implementors should read an integer number of data chunks into
106   *    buf. The amount read should be bounded by len or by 
107   *    checksum.length / CHECKSUM_SIZE * maxChunkSize. Note that len may
108   *    be a value that is not a multiple of maxChunkSize, in which case
109   *    the implementation may return less than len.
110   *
111   * The method is used for implementing read, therefore, it should be optimized
112   * for sequential reading.
113   *
114   * @param pos chunkPos
115   * @param buf desitination buffer
116   * @param offset offset in buf at which to store data
117   * @param len maximum number of bytes to read
118   * @param checksum the data buffer into which to write checksums
119   * @return number of bytes read
120   */
121  abstract protected int readChunk(long pos, byte[] buf, int offset, int len,
122      byte[] checksum) throws IOException;
123
124  /** Return position of beginning of chunk containing pos. 
125   *
126   * @param pos a postion in the file
127   * @return the starting position of the chunk which contains the byte
128   */
129  abstract protected long getChunkPosition(long pos);
130
131  /** Return true if there is a need for checksum verification */
132  protected synchronized boolean needChecksum() {
133    return verifyChecksum && sum != null;
134  }
135
136  /**
137   * Read one checksum-verified byte
138   * 
139   * @return     the next byte of data, or <code>-1</code> if the end of the
140   *             stream is reached.
141   * @exception  IOException  if an I/O error occurs.
142   */
143
144  @Override
145  public synchronized int read() throws IOException {
146    if (pos >= count) {
147      fill();
148      if (pos >= count) {
149        return -1;
150      }
151    }
152    return buf[pos++] & 0xff;
153  }
154  
155  /**
156   * Read checksum verified bytes from this byte-input stream into 
157   * the specified byte array, starting at the given offset.
158   *
159   * <p> This method implements the general contract of the corresponding
160   * <code>{@link InputStream#read(byte[], int, int) read}</code> method of
161   * the <code>{@link InputStream}</code> class.  As an additional
162   * convenience, it attempts to read as many bytes as possible by repeatedly
163   * invoking the <code>read</code> method of the underlying stream.  This
164   * iterated <code>read</code> continues until one of the following
165   * conditions becomes true: <ul>
166   *
167   *   <li> The specified number of bytes have been read,
168   *
169   *   <li> The <code>read</code> method of the underlying stream returns
170   *   <code>-1</code>, indicating end-of-file.
171   *
172   * </ul> If the first <code>read</code> on the underlying stream returns
173   * <code>-1</code> to indicate end-of-file then this method returns
174   * <code>-1</code>.  Otherwise this method returns the number of bytes
175   * actually read.
176   *
177   * @param      b     destination buffer.
178   * @param      off   offset at which to start storing bytes.
179   * @param      len   maximum number of bytes to read.
180   * @return     the number of bytes read, or <code>-1</code> if the end of
181   *             the stream has been reached.
182   * @exception  IOException  if an I/O error occurs.
183   *             ChecksumException if any checksum error occurs
184   */
185  @Override
186  public synchronized int read(byte[] b, int off, int len) throws IOException {
187    // parameter check
188    if ((off | len | (off + len) | (b.length - (off + len))) < 0) {
189      throw new IndexOutOfBoundsException();
190    } else if (len == 0) {
191      return 0;
192    }
193
194    int n = 0;
195    for (;;) {
196      int nread = read1(b, off + n, len - n);
197      if (nread <= 0) 
198        return (n == 0) ? nread : n;
199      n += nread;
200      if (n >= len)
201        return n;
202    }
203  }
204  
205  /**
206   * Fills the buffer with a chunk data. 
207   * No mark is supported.
208   * This method assumes that all data in the buffer has already been read in,
209   * hence pos > count.
210   */
211  private void fill(  ) throws IOException {
212    assert(pos>=count);
213    // fill internal buffer
214    count = readChecksumChunk(buf, 0, maxChunkSize);
215    if (count < 0) count = 0;
216  }
217  
218  /*
219   * Read characters into a portion of an array, reading from the underlying
220   * stream at most once if necessary.
221   */
222  private int read1(byte b[], int off, int len)
223  throws IOException {
224    int avail = count-pos;
225    if( avail <= 0 ) {
226      if(len >= maxChunkSize) {
227        // read a chunk to user buffer directly; avoid one copy
228        int nread = readChecksumChunk(b, off, len);
229        return nread;
230      } else {
231        // read a chunk into the local buffer
232         fill();
233        if( count <= 0 ) {
234          return -1;
235        } else {
236          avail = count;
237        }
238      }
239    }
240    
241    // copy content of the local buffer to the user buffer
242    int cnt = (avail < len) ? avail : len;
243    System.arraycopy(buf, pos, b, off, cnt);
244    pos += cnt;
245    return cnt;    
246  }
247  
248  /* Read up one or more checksum chunk to array <i>b</i> at pos <i>off</i>
249   * It requires at least one checksum chunk boundary
250   * in between <cur_pos, cur_pos+len> 
251   * and it stops reading at the last boundary or at the end of the stream;
252   * Otherwise an IllegalArgumentException is thrown.
253   * This makes sure that all data read are checksum verified.
254   * 
255   * @param b   the buffer into which the data is read.
256   * @param off the start offset in array <code>b</code>
257   *            at which the data is written.
258   * @param len the maximum number of bytes to read.
259   * @return    the total number of bytes read into the buffer, or
260   *            <code>-1</code> if there is no more data because the end of
261   *            the stream has been reached.
262   * @throws IOException if an I/O error occurs.
263   */ 
264  private int readChecksumChunk(byte b[], final int off, final int len)
265  throws IOException {
266    // invalidate buffer
267    count = pos = 0;
268          
269    int read = 0;
270    boolean retry = true;
271    int retriesLeft = numOfRetries; 
272    do {
273      retriesLeft--;
274
275      try {
276        read = readChunk(chunkPos, b, off, len, checksum);
277        if( read > 0) {
278          if( needChecksum() ) {
279            verifySums(b, off, read);
280          }
281          chunkPos += read;
282        }
283        retry = false;
284      } catch (ChecksumException ce) {
285          LOG.info("Found checksum error: b[" + off + ", " + (off+read) + "]="
286              + StringUtils.byteToHexString(b, off, off + read), ce);
287          if (retriesLeft == 0) {
288            throw ce;
289          }
290          
291          // try a new replica
292          if (seekToNewSource(chunkPos)) {
293            // Since at least one of the sources is different, 
294            // the read might succeed, so we'll retry.
295            seek(chunkPos);
296          } else {
297            // Neither the data stream nor the checksum stream are being read
298            // from different sources, meaning we'll still get a checksum error 
299            // if we try to do the read again.  We throw an exception instead.
300            throw ce;
301          }
302        }
303    } while (retry);
304    return read;
305  }
306
307  private void verifySums(final byte b[], final int off, int read)
308    throws ChecksumException
309  {
310    int leftToVerify = read;
311    int verifyOff = 0;
312    checksumInts.rewind();
313    checksumInts.limit((read - 1)/maxChunkSize + 1);
314
315    while (leftToVerify > 0) {
316      sum.update(b, off + verifyOff, Math.min(leftToVerify, maxChunkSize));
317      int expected = checksumInts.get();
318      int calculated = (int)sum.getValue();
319      sum.reset();
320
321      if (expected != calculated) {
322        long errPos = chunkPos + verifyOff;
323        throw new ChecksumException(
324          "Checksum error: "+file+" at "+ errPos +
325          " exp: " + expected + " got: " + calculated, errPos);
326      }
327      leftToVerify -= maxChunkSize;
328      verifyOff += maxChunkSize;
329    }
330  }

337  @Deprecated

346  @Override
347  public synchronized long getPos() throws IOException {
348    return chunkPos-Math.max(0L, count - pos);
349  }
350
351  @Override
352  public synchronized int available() throws IOException {
353    return Math.max(0, count - pos);
354  }

356  /**
357   * Skips over and discards <code>n</code> bytes of data from the
358   * input stream.
359   *
360   * <p>This method may skip more bytes than are remaining in the backing
361   * file. This produces no exception and the number of bytes skipped
362   * may include some number of bytes that were beyond the EOF of the
363   * backing file. Attempting to read from the stream after skipping past
364   * the end will result in -1 indicating the end of the file.
365   *
366   *<p>If <code>n</code> is negative, no bytes are skipped.
367   *
368   * @param      n   the number of bytes to be skipped.
369   * @return     the actual number of bytes skipped.
370   * @exception  IOException  if an I/O error occurs.
371   *             ChecksumException if the chunk to skip to is corrupted
372   */
373  @Override
374  public synchronized long skip(long n) throws IOException {
375    if (n <= 0) {
376      return 0;
377    }
378
379    seek(getPos()+n);
380    return n;
381  }
382
383  /**
384   * Seek to the given position in the stream.
385   * The next read() will be from that position.
386   * 
387   * <p>This method may seek past the end of the file.
388   * This produces no exception and an attempt to read from
389   * the stream will result in -1 indicating the end of the file.
390   *
391   * @param      pos   the postion to seek to.
392   * @exception  IOException  if an I/O error occurs.
393   *             ChecksumException if the chunk to seek to is corrupted
394   */
395
396  @Override
397  public synchronized void seek(long pos) throws IOException {
398    if( pos < 0 ) {
399      throw new EOFException(FSExceptionMessages.NEGATIVE_SEEK);
400    }
401    // optimize: check if the pos is in the buffer
402    long start = chunkPos - this.count;
403    if( pos>=start && pos<chunkPos) {
404      this.pos = (int)(pos-start);
405      return;
406    }
407    
408    // reset the current state
409    resetState();
410    
411    // seek to a checksum boundary
412    chunkPos = getChunkPosition(pos);
413    
414    // scan to the desired position
415    int delta = (int)(pos - chunkPos);
416    if( delta > 0) {
417      readFully(this, new byte[delta], 0, delta);
418    }
419  }

421  /**
422   * A utility function that tries to read up to <code>len</code> bytes from
423   * <code>stm</code>
424   * 
425   * @param stm    an input stream
426   * @param buf    destiniation buffer
427   * @param offset offset at which to store data
428   * @param len    number of bytes to read
429   * @return actual number of bytes read
430   * @throws IOException if there is any IO error
431   */
432  protected static int readFully(InputStream stm, 
433      byte[] buf, int offset, int len) throws IOException {
434    int n = 0;
435    for (;;) {
436      int nread = stm.read(buf, offset + n, len - n);
437      if (nread <= 0) 
438        return (n == 0) ? nread : n;
439      n += nread;
440      if (n >= len)
441        return n;
442    }
443  }

445  /**
446   * Set the checksum related parameters
447   * @param verifyChecksum whether to verify checksum
448   * @param sum which type of checksum to use
449   * @param maxChunkSize maximun chunk size
450   * @param checksumSize checksum size
451   */
452  final protected synchronized void set(boolean verifyChecksum,
453      Checksum sum, int maxChunkSize, int checksumSize) {
454
455    // The code makes assumptions that checksums are always 32-bit.
456    assert !verifyChecksum || sum == null || checksumSize == CHECKSUM_SIZE;
457
458    this.maxChunkSize = maxChunkSize;
459    this.verifyChecksum = verifyChecksum;
460    this.sum = sum;
461    this.buf = new byte[maxChunkSize];
462    // The size of the checksum array here determines how much we can
463    // read in a single call to readChunk
464    this.checksum = new byte[CHUNKS_PER_READ * checksumSize];
465    this.checksumInts = ByteBuffer.wrap(checksum).asIntBuffer();
466    this.count = 0;
467    this.pos = 0;
468  }

477  }

481    throw new IOException("mark/reset not supported");

485  /* reset this FSInputChecker's state */
486  private void resetState() {
487    // invalidate buffer
488    count = 0;
489    pos = 0;
490    // reset Checksum
491    if (sum != null) {
492      sum.reset();
493    }
494  }
