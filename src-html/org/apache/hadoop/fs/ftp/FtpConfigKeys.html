019package org.apache.hadoop.fs.ftp;
020
021import java.io.IOException;
022import org.apache.hadoop.classification.InterfaceAudience;
023import org.apache.hadoop.classification.InterfaceStability;
024import org.apache.hadoop.fs.CommonConfigurationKeys;
025import org.apache.hadoop.fs.FsServerDefaults;
026import org.apache.hadoop.util.DataChecksum;
027
028/** 
029 * This class contains constants for configuration keys used
030 * in the ftp file system.
031 *
032 * Note that the settings for unimplemented features are ignored. 
033 * E.g. checksum related settings are just place holders. Even when
034 * wrapped with {@link ChecksumFileSystem}, these settings are not
035 * used. 
036 */
037@InterfaceAudience.Private
038@InterfaceStability.Unstable
039public class FtpConfigKeys extends CommonConfigurationKeys {
040  public static final String  BLOCK_SIZE_KEY = "ftp.blocksize";
041  public static final long    BLOCK_SIZE_DEFAULT = 4*1024;
042  public static final String  REPLICATION_KEY = "ftp.replication";
043  public static final short   REPLICATION_DEFAULT = 1;
044  public static final String  STREAM_BUFFER_SIZE_KEY = 
045                                                    "ftp.stream-buffer-size";
046  public static final int     STREAM_BUFFER_SIZE_DEFAULT = 1024*1024;
047  public static final String  BYTES_PER_CHECKSUM_KEY = 
048                                                    "ftp.bytes-per-checksum";
049  public static final int     BYTES_PER_CHECKSUM_DEFAULT = 512;
050  public static final String  CLIENT_WRITE_PACKET_SIZE_KEY =
051                                                "ftp.client-write-packet-size";
052  public static final int     CLIENT_WRITE_PACKET_SIZE_DEFAULT = 64*1024;
053  public static final boolean ENCRYPT_DATA_TRANSFER_DEFAULT = false;
054  public static final long    FS_TRASH_INTERVAL_DEFAULT = 0;
055  public static final DataChecksum.Type CHECKSUM_TYPE_DEFAULT =
056      DataChecksum.Type.CRC32;
057  
058  protected static FsServerDefaults getServerDefaults() throws IOException {
059    return new FsServerDefaults(
060        BLOCK_SIZE_DEFAULT,
061        BYTES_PER_CHECKSUM_DEFAULT,
062        CLIENT_WRITE_PACKET_SIZE_DEFAULT,
063        REPLICATION_DEFAULT,
064        STREAM_BUFFER_SIZE_DEFAULT,
065        ENCRYPT_DATA_TRANSFER_DEFAULT,
066        FS_TRASH_INTERVAL_DEFAULT,
067        CHECKSUM_TYPE_DEFAULT);
068  }
069}
