108  /** Build the description used by the help command. */
109  private static String buildDescription(ExpressionFactory factory) {
110    ArrayList<Expression> operators = new ArrayList<Expression>();
111    ArrayList<Expression> primaries = new ArrayList<Expression>();
112    for (Class<? extends Expression> exprClass : EXPRESSIONS) {
113      Expression expr = factory.createExpression(exprClass, null);
114      if (expr.isOperator()) {
115        operators.add(expr);
116      } else {
117        primaries.add(expr);
118      }
119    }
120    Collections.sort(operators, new Comparator<Expression>() {
121      @Override
122      public int compare(Expression arg0, Expression arg1) {
123        return arg0.getClass().getName().compareTo(arg1.getClass().getName());
124      }
125    });
126    Collections.sort(primaries, new Comparator<Expression>() {
127      @Override
128      public int compare(Expression arg0, Expression arg1) {
129        return arg0.getClass().getName().compareTo(arg1.getClass().getName());
130      }
131    });
132
133    StringBuilder sb = new StringBuilder();
134    for (String line : HELP) {
135      sb.append(line).append("\n");
136    }
137    sb.append("\n");
138    sb.append("The following primary expressions are recognised:\n");
139    for (Expression expr : primaries) {
140      for (String line : expr.getUsage()) {
141        sb.append("  ").append(line).append("\n");
142      }
143      for (String line : expr.getHelp()) {
144        sb.append("    ").append(line).append("\n");
145      }
146      sb.append("\n");
147    }
148    sb.append("The following operators are recognised:\n");
149    for (Expression expr : operators) {
150      for (String line : expr.getUsage()) {
151        sb.append("  ").append(line).append("\n");
152      }
153      for (String line : expr.getHelp()) {
154        sb.append("    ").append(line).append("\n");
155      }
156      sb.append("\n");
157    }
158    return sb.toString();
159  }

161  /** Default constructor for the Find command. */

166  @Override
167  protected void processOptions(LinkedList<String> args) throws IOException {
168    CommandFormat cf =
169        new CommandFormat(1, Integer.MAX_VALUE, OPTION_FOLLOW_LINK,
170            OPTION_FOLLOW_ARG_LINK, null);
171    cf.parse(args);
172
173    if (cf.getOpt(OPTION_FOLLOW_LINK)) {
174      getOptions().setFollowLink(true);
175    } else if (cf.getOpt(OPTION_FOLLOW_ARG_LINK)) {
176      getOptions().setFollowArgLink(true);
177    }
178
179    // search for first non-path argument (ie starts with a "-") and capture and
180    // remove the remaining arguments as expressions
181    LinkedList<String> expressionArgs = new LinkedList<String>();
182    Iterator<String> it = args.iterator();
183    boolean isPath = true;
184    while (it.hasNext()) {
185      String arg = it.next();
186      if (isPath) {
187        if (arg.startsWith("-")) {
188          isPath = false;
189        }
190      }
191      if (!isPath) {
192        expressionArgs.add(arg);
193        it.remove();
194      }
195    }
196
197    if (args.isEmpty()) {
198      args.add(Path.CUR_DIR);
199    }
200
201    Expression expression = parseExpression(expressionArgs);
202    if (!expression.isAction()) {
203      Expression and = getExpression(And.class);
204      Deque<Expression> children = new LinkedList<Expression>();
205      children.add(getExpression(Print.class));
206      children.add(expression);
207      and.addChildren(children);
208      expression = and;
209    }
210
211    setRootExpression(expression);
212  }

243  /** Create a new set of find options. */
244  private FindOptions createOptions() {
245    FindOptions options = new FindOptions();
246    options.setOut(out);
247    options.setErr(err);
248    options.setIn(System.in);
249    options.setCommandFactory(getCommandFactory());
250    options.setConfiguration(getConf());
251    return options;

254  /** Add the {@link PathData} item to the stop set. */
255  private void addStop(PathData item) {
256    stopPaths.add(item.path);
257  }
258
259  /** Returns true if the {@link PathData} item is in the stop set. */
260  private boolean isStop(PathData item) {
261    return stopPaths.contains(item.path);
262  }
263
264  /**
265   * Parse a list of arguments to to extract the {@link Expression} elements.
266   * The input Deque will be modified to remove the used elements.
267   * 
268   * @param args arguments to be parsed
269   * @return list of {@link Expression} elements applicable to this command
270   * @throws IOException if list can not be parsed
271   */
272  private Expression parseExpression(Deque<String> args) throws IOException {
273    Deque<Expression> primaries = new LinkedList<Expression>();
274    Deque<Expression> operators = new LinkedList<Expression>();
275    Expression prevExpr = getExpression(And.class);
276    while (!args.isEmpty()) {
277      String arg = args.pop();
278      if ("(".equals(arg)) {
279        Expression expr = parseExpression(args);
280        primaries.add(expr);
281        prevExpr = new BaseExpression() {
282          @Override
283          public Result apply(PathData item, int depth) throws IOException {
284            return Result.PASS;
285          }
286        }; // stub the previous expression to be a non-op
287      } else if (")".equals(arg)) {
288        break;
289      } else if (isExpression(arg)) {
290        Expression expr = getExpression(arg);
291        expr.addArguments(args);
292        if (expr.isOperator()) {
293          while (!operators.isEmpty()) {
294            if (operators.peek().getPrecedence() >= expr.getPrecedence()) {
295              Expression op = operators.pop();
296              op.addChildren(primaries);
297              primaries.push(op);
298            } else {
299              break;
300            }
301          }
302          operators.push(expr);
303        } else {
304          if (!prevExpr.isOperator()) {
305            Expression and = getExpression(And.class);
306            while (!operators.isEmpty()) {
307              if (operators.peek().getPrecedence() >= and.getPrecedence()) {
308                Expression op = operators.pop();
309                op.addChildren(primaries);
310                primaries.push(op);
311              } else {
312                break;
313              }
314            }
315            operators.push(and);
316          }
317          primaries.push(expr);
318        }
319        prevExpr = expr;
320      } else {
321        throw new IOException("Unexpected argument: " + arg);
322      }
323    }
324
325    while (!operators.isEmpty()) {
326      Expression operator = operators.pop();
327      operator.addChildren(primaries);
328      primaries.push(operator);
329    }
330
331    return primaries.isEmpty() ? getExpression(Print.class) : primaries.pop();
332  }

334  /** Returns true if the target is an ancestor of the source. */
335  private boolean isAncestor(PathData source, PathData target) {
336    for (Path parent = source.path; (parent != null) && !parent.isRoot();
337        parent = parent.getParent()) {
338      if (parent.equals(target.path)) {
339        return true;
340      }
341    }
342    return false;
343  }

345  @Override
346  protected void recursePath(PathData item) throws IOException {
347    if (isStop(item)) {
348      // this item returned a stop result so don't recurse any further
349      return;
350    }
351    if (getDepth() >= getOptions().getMaxDepth()) {
352      // reached the maximum depth so don't got any further.
353      return;
354    }
355    if (item.stat.isSymlink() && getOptions().isFollowLink()) {
356      PathData linkedItem =
357          new PathData(item.stat.getSymlink().toString(), getConf());
358      if (isAncestor(item, linkedItem)) {
359        getOptions().getErr().println(
360            "Infinite loop ignored: " + item.toString() + " -> "
361                + linkedItem.toString());
362        return;
363      }
364      if (linkedItem.exists) {
365        item = linkedItem;
366      }
367    }
368    if (item.stat.isDirectory()) {
369      super.recursePath(item);
370    }
371  }
372
373  @Override
374  protected boolean isPathRecursable(PathData item) throws IOException {
375    if (item.stat.isDirectory()) {
376      return true;
377    }
378    if (item.stat.isSymlink()) {
379      PathData linkedItem =
380          new PathData(item.fs.resolvePath(item.stat.getSymlink()).toString(),
381              getConf());
382      if (linkedItem.stat.isDirectory()) {
383        if (getOptions().isFollowLink()) {
384          return true;
385        }
386        if (getOptions().isFollowArgLink() && (getDepth() == 0)) {
387          return true;
388        }
389      }
390    }
391    return false;
392  }
393
394  @Override
395  protected void processPath(PathData item) throws IOException {
396    if (getOptions().isDepthFirst()) {
397      // depth first so leave until post processing
398      return;
399    }
400    applyItem(item);
401  }
402
403  @Override
404  protected void postProcessPath(PathData item) throws IOException {
405    if (!getOptions().isDepthFirst()) {
406      // not depth first so already processed
407      return;
408    }
409    applyItem(item);
410  }
411
412  private void applyItem(PathData item) throws IOException {
413    if (getDepth() >= getOptions().getMinDepth()) {
414      Result result = getRootExpression().apply(item, getDepth());
415      if (Result.STOP.equals(result)) {
416        addStop(item);
417      }
418    }
419  }
420
421  @Override
422  protected void processArguments(LinkedList<PathData> args)
423      throws IOException {
424    Expression expr = getRootExpression();
425    expr.setOptions(getOptions());
426    expr.prepare();
427    super.processArguments(args);
428    expr.finish();
429  }
