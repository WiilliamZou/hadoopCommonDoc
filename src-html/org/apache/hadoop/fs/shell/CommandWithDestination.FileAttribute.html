052/**
053 * Provides: argument processing to ensure the destination is valid
054 * for the number of source arguments.  A processPaths that accepts both
055 * a source and resolved target.  Sources are resolved as children of
056 * a destination directory.
057 */
058abstract class CommandWithDestination extends FsCommand {  
059  protected PathData dst;
060  private boolean overwrite = false;
061  private boolean verifyChecksum = true;
062  private boolean writeChecksum = true;
063  private boolean lazyPersist = false;
064  
065  /**
066   * The name of the raw xattr namespace. It would be nice to use
067   * XAttr.RAW.name() but we can't reference the hadoop-hdfs project.
068   */
069  private static final String RAW = "raw.";
070
071  /**
072   * The name of the reserved raw directory.
073   */
074  private static final String RESERVED_RAW = "/.reserved/raw";

098  /**
099   * If true, the last modified time, last access time,
100   * owner, group and permission information of the source
101   * file will be preserved as far as target {@link FileSystem}
102   * implementation allows.
103   */
104  protected void setPreserve(boolean preserve) {
105    if (preserve) {
106      preserve(FileAttribute.TIMESTAMPS);
107      preserve(FileAttribute.OWNERSHIP);
108      preserve(FileAttribute.PERMISSION);
109    } else {
110      preserveStatus.clear();
111    }
112  }
113  
114  protected static enum FileAttribute {
115    TIMESTAMPS, OWNERSHIP, PERMISSION, ACL, XATTR;
116
117    public static FileAttribute getAttribute(char symbol) {
118      for (FileAttribute attribute : values()) {
119        if (attribute.name().charAt(0) == Character.toUpperCase(symbol)) {
120          return attribute;
121        }
122      }
123      throw new NoSuchElementException("No attribute for " + symbol);
124    }
125  }
126  
127  private EnumSet<FileAttribute> preserveStatus = 
128      EnumSet.noneOf(FileAttribute.class);
129  
130  /**
131   * Checks if the input attribute should be preserved or not
132   *
133   * @param attribute - Attribute to check
134   * @return boolean true if attribute should be preserved, false otherwise
135   */
136  private boolean shouldPreserve(FileAttribute attribute) {
137    return preserveStatus.contains(attribute);
138  }
139  
140  /**
141   * Add file attributes that need to be preserved. This method may be
142   * called multiple times to add attributes.
143   *
144   * @param fileAttribute - Attribute to add, one at a time
145   */
146  protected void preserve(FileAttribute fileAttribute) {
147    for (FileAttribute attribute : preserveStatus) {
148      if (attribute.equals(fileAttribute)) {
149        return;
150      }
151    }
152    preserveStatus.add(fileAttribute);
153  }

155  /**
156   *  The last arg is expected to be a local path, if only one argument is
157   *  given then the destination will be the current directory 
158   *  @param args is the list of arguments
159   */
160  protected void getLocalDestination(LinkedList<String> args)
161  throws IOException {
162    String pathString = (args.size() < 2) ? Path.CUR_DIR : args.removeLast();
163    try {
164      dst = new PathData(new URI(pathString), getConf());
165    } catch (URISyntaxException e) {
166      if (Path.WINDOWS) {
167        // Unlike URI, PathData knows how to parse Windows drive-letter paths.
168        dst = new PathData(pathString, getConf());
169      } else {
170        throw new IOException("unexpected URISyntaxException", e);
171      }
172    }
173  }
174
175  /**
176   *  The last arg is expected to be a remote path, if only one argument is
177   *  given then the destination will be the remote user's directory 
178   *  @param args is the list of arguments
179   *  @throws PathIOException if path doesn't exist or matches too many times 
180   */
181  protected void getRemoteDestination(LinkedList<String> args)
182  throws IOException {
183    if (args.size() < 2) {
184      dst = new PathData(Path.CUR_DIR, getConf());
185    } else {
186      String pathString = args.removeLast();
187      // if the path is a glob, then it must match one and only one path
188      PathData[] items = PathData.expandAsGlob(pathString, getConf());
189      switch (items.length) {
190        case 0:
191          throw new PathNotFoundException(pathString);
192        case 1:
193          dst = items[0];
194          break;
195        default:
196          throw new PathIOException(pathString, "Too many matches");
197      }
198    }
199  }

201  @Override
202  protected void processArguments(LinkedList<PathData> args)
203  throws IOException {
204    // if more than one arg, the destination must be a directory
205    // if one arg, the dst must not exist or must be a directory
206    if (args.size() > 1) {
207      if (!dst.exists) {
208        throw new PathNotFoundException(dst.toString());
209      }
210      if (!dst.stat.isDirectory()) {
211        throw new PathIsNotDirectoryException(dst.toString());
212      }
213    } else if (dst.exists) {
214      if (!dst.stat.isDirectory() && !overwrite) {
215        throw new PathExistsException(dst.toString());
216      }
217    } else if (!dst.parentExists()) {
218      throw new PathNotFoundException(dst.toString());
219    }
220    super.processArguments(args);
221  }
222
223  @Override
224  protected void processPathArgument(PathData src)
225  throws IOException {
226    if (src.stat.isDirectory() && src.fs.equals(dst.fs)) {
227      PathData target = getTargetPath(src);
228      String srcPath = src.fs.makeQualified(src.path).toString();
229      String dstPath = dst.fs.makeQualified(target.path).toString();
230      if (dstPath.equals(srcPath)) {
231        PathIOException e = new PathIOException(src.toString(),
232            "are identical");
233        e.setTargetPath(dstPath.toString());
234        throw e;
235      }
236      if (dstPath.startsWith(srcPath+Path.SEPARATOR)) {
237        PathIOException e = new PathIOException(src.toString(),
238            "is a subdirectory of itself");
239        e.setTargetPath(target.toString());
240        throw e;
241      }
242    }
243    super.processPathArgument(src);
244  }
245
246  @Override
247  protected void processPath(PathData src) throws IOException {
248    processPath(src, getTargetPath(src));
249  }
250  
251  /**
252   * Called with a source and target destination pair
253   * @param src for the operation
254   * @param dst for the operation
255   * @throws IOException if anything goes wrong
256   */
257  protected void processPath(PathData src, PathData dst) throws IOException {
258    if (src.stat.isSymlink()) {
259      // TODO: remove when FileContext is supported, this needs to either
260      // copy the symlink or deref the symlink
261      throw new PathOperationException(src.toString());        
262    } else if (src.stat.isFile()) {
263      copyFileToTarget(src, dst);
264    } else if (src.stat.isDirectory() && !isRecursive()) {
265      throw new PathIsDirectoryException(src.toString());
266    }
267  }
268
269  @Override
270  protected void recursePath(PathData src) throws IOException {
271    PathData savedDst = dst;
272    try {
273      // modify dst as we descend to append the basename of the
274      // current directory being processed
275      dst = getTargetPath(src);
276      final boolean preserveRawXattrs =
277          checkPathsForReservedRaw(src.path, dst.path);
278      if (dst.exists) {
279        if (!dst.stat.isDirectory()) {
280          throw new PathIsNotDirectoryException(dst.toString());
281        }
282      } else {
283        if (!dst.fs.mkdirs(dst.path)) {
284          // too bad we have no clue what failed
285          PathIOException e = new PathIOException(dst.toString());
286          e.setOperation("mkdir");
287          throw e;
288        }    
289        dst.refreshStatus(); // need to update stat to know it exists now
290      }      
291      super.recursePath(src);
292      if (dst.stat.isDirectory()) {
293        preserveAttributes(src, dst, preserveRawXattrs);
294      }
295    } finally {
296      dst = savedDst;
297    }
298  }
299  
300  protected PathData getTargetPath(PathData src) throws IOException {
301    PathData target;
302    // on the first loop, the dst may be directory or a file, so only create
303    // a child path if dst is a dir; after recursion, it's always a dir
304    if ((getDepth() > 0) || (dst.exists && dst.stat.isDirectory())) {
305      target = dst.getPathDataForChild(src);
306    } else if (dst.representsDirectory()) { // see if path looks like a dir
307      target = dst.getPathDataForChild(src);
308    } else {
309      target = dst;
310    }
311    return target;
312  }

314  /**
315   * Copies the source file to the target.
316   * @param src item to copy
317   * @param target where to copy the item
318   * @throws IOException if copy fails
319   */ 
320  protected void copyFileToTarget(PathData src, PathData target)
321      throws IOException {
322    final boolean preserveRawXattrs =
323        checkPathsForReservedRaw(src.path, target.path);
324    src.fs.setVerifyChecksum(verifyChecksum);
325    InputStream in = null;
326    try {
327      in = src.fs.open(src.path);
328      copyStreamToTarget(in, target);
329      preserveAttributes(src, target, preserveRawXattrs);
330    } finally {
331      IOUtils.closeStream(in);
332    }
333  }

335  /**
336   * Check the source and target paths to ensure that they are either both in
337   * /.reserved/raw or neither in /.reserved/raw. If neither src nor target are
338   * in /.reserved/raw, then return false, indicating not to preserve raw.*
339   * xattrs. If both src/target are in /.reserved/raw, then return true,
340   * indicating raw.* xattrs should be preserved. If only one of src/target is
341   * in /.reserved/raw then throw an exception.
342   *
343   * @param src The source path to check. This should be a fully-qualified
344   *            path, not relative.
345   * @param target The target path to check. This should be a fully-qualified
346   *               path, not relative.
347   * @return true if raw.* xattrs should be preserved.
348   * @throws PathOperationException is only one of src/target are in
349   * /.reserved/raw.
350   */
351  private boolean checkPathsForReservedRaw(Path src, Path target)
352      throws PathOperationException {
353    final boolean srcIsRR = Path.getPathWithoutSchemeAndAuthority(src).
354        toString().startsWith(RESERVED_RAW);
355    final boolean dstIsRR = Path.getPathWithoutSchemeAndAuthority(target).
356        toString().startsWith(RESERVED_RAW);
357    boolean preserveRawXattrs = false;
358    if (srcIsRR && !dstIsRR) {
359      final String s = "' copy from '" + RESERVED_RAW + "' to non '" +
360          RESERVED_RAW + "'. Either both source and target must be in '" +
361          RESERVED_RAW + "' or neither.";
362      throw new PathOperationException("'" + src.toString() + s);
363    } else if (!srcIsRR && dstIsRR) {
364      final String s = "' copy from non '" + RESERVED_RAW +"' to '" +
365          RESERVED_RAW + "'. Either both source and target must be in '" +
366          RESERVED_RAW + "' or neither.";
367      throw new PathOperationException("'" + dst.toString() + s);
368    } else if (srcIsRR && dstIsRR) {
369      preserveRawXattrs = true;
370    }
371    return preserveRawXattrs;
372  }

374  /**
375   * Copies the stream contents to a temporary file.  If the copy is
376   * successful, the temporary file will be renamed to the real path,
377   * else the temporary file will be deleted.
378   * @param in the input stream for the copy
379   * @param target where to store the contents of the stream
380   * @throws IOException if copy fails
381   */ 
382  protected void copyStreamToTarget(InputStream in, PathData target)
383  throws IOException {
384    if (target.exists && (target.stat.isDirectory() || !overwrite)) {
385      throw new PathExistsException(target.toString());
386    }
387    TargetFileSystem targetFs = new TargetFileSystem(target.fs);
388    try {
389      PathData tempTarget = target.suffix("._COPYING_");
390      targetFs.setWriteChecksum(writeChecksum);
391      targetFs.writeStreamToFile(in, tempTarget, lazyPersist);
392      targetFs.rename(tempTarget, target);
393    } finally {
394      targetFs.close(); // last ditch effort to ensure temp file is removed
395    }
396  }

398  /**
399   * Preserve the attributes of the source to the target.
400   * The method calls {@link #shouldPreserve(FileAttribute)} to check what
401   * attribute to preserve.
402   * @param src source to preserve
403   * @param target where to preserve attributes
404   * @param preserveRawXAttrs true if raw.* xattrs should be preserved
405   * @throws IOException if fails to preserve attributes
406   */
407  protected void preserveAttributes(PathData src, PathData target,
408      boolean preserveRawXAttrs)
409      throws IOException {
410    if (shouldPreserve(FileAttribute.TIMESTAMPS)) {
411      target.fs.setTimes(
412          target.path,
413          src.stat.getModificationTime(),
414          src.stat.getAccessTime());
415    }
416    if (shouldPreserve(FileAttribute.OWNERSHIP)) {
417      target.fs.setOwner(
418          target.path,
419          src.stat.getOwner(),
420          src.stat.getGroup());
421    }
422    if (shouldPreserve(FileAttribute.PERMISSION) ||
423        shouldPreserve(FileAttribute.ACL)) {
424      target.fs.setPermission(
425          target.path,
426          src.stat.getPermission());
427    }
428    if (shouldPreserve(FileAttribute.ACL)) {
429      FsPermission perm = src.stat.getPermission();
430      if (perm.getAclBit()) {
431        List<AclEntry> srcEntries =
432            src.fs.getAclStatus(src.path).getEntries();
433        List<AclEntry> srcFullEntries =
434            AclUtil.getAclFromPermAndEntries(perm, srcEntries);
435        target.fs.setAcl(target.path, srcFullEntries);
436      }
437    }
438    final boolean preserveXAttrs = shouldPreserve(FileAttribute.XATTR);
439    if (preserveXAttrs || preserveRawXAttrs) {
440      Map<String, byte[]> srcXAttrs = src.fs.getXAttrs(src.path);
441      if (srcXAttrs != null) {
442        Iterator<Entry<String, byte[]>> iter = srcXAttrs.entrySet().iterator();
443        while (iter.hasNext()) {
444          Entry<String, byte[]> entry = iter.next();
445          final String xattrName = entry.getKey();
446          if (xattrName.startsWith(RAW) || preserveXAttrs) {
447            target.fs.setXAttr(target.path, entry.getKey(), entry.getValue());
448          }
449        }
450      }
451    }
452  }

454  // Helper filter filesystem that registers created files as temp files to
455  // be deleted on exit unless successfully renamed
456  private static class TargetFileSystem extends FilterFileSystem {
457    TargetFileSystem(FileSystem fs) {
458      super(fs);
459    }
460
461    void writeStreamToFile(InputStream in, PathData target,
462                           boolean lazyPersist) throws IOException {
463      FSDataOutputStream out = null;
464      try {
465        out = create(target, lazyPersist);
466        IOUtils.copyBytes(in, out, getConf(), true);
467      } finally {
468        IOUtils.closeStream(out); // just in case copyBytes didn't
469      }
470    }
471    
472    // tag created files as temp files
473    FSDataOutputStream create(PathData item, boolean lazyPersist)
474        throws IOException {
475      try {
476        if (lazyPersist) {
477          EnumSet<CreateFlag> createFlags = EnumSet.of(CREATE, LAZY_PERSIST);
478          return create(item.path,
479                        FsPermission.getFileDefault().applyUMask(
480                            FsPermission.getUMask(getConf())),
481                        createFlags,
482                        getConf().getInt("io.file.buffer.size", 4096),
483                        lazyPersist ? 1 : getDefaultReplication(item.path),
484                        getDefaultBlockSize(),
485                        null,
486                        null);
487        } else {
488          return create(item.path, true);
489        }
490      } finally { // might have been created but stream was interrupted
491        deleteOnExit(item.path);
492      }
493    }
494
495    void rename(PathData src, PathData target) throws IOException {
496      // the rename method with an option to delete the target is deprecated
497      if (target.exists && !delete(target.path, false)) {
498        // too bad we don't know why it failed
499        PathIOException e = new PathIOException(target.toString());
500        e.setOperation("delete");
501        throw e;
502      }
503      if (!rename(src.path, target.path)) {
504        // too bad we don't know why it failed
505        PathIOException e = new PathIOException(src.toString());
506        e.setOperation("rename");
507        e.setTargetPath(target.toString());
508        throw e;
509      }
510      // cancel delete on exit if rename is successful
511      cancelDeleteOnExit(src.path);
512    }
513    @Override
514    public void close() {
515      // purge any remaining temp files, but don't close underlying fs
516      processDeleteOnExit();
517    }
518  }
