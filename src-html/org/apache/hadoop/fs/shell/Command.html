021import java.io.IOException;
022import java.io.PrintStream;
023import java.lang.reflect.Field;
024import java.util.ArrayList;
025import java.util.Arrays;
026import java.util.LinkedList;
027import java.util.List;

029import org.apache.commons.logging.Log;
030import org.apache.commons.logging.LogFactory;
031import org.apache.hadoop.classification.InterfaceAudience;
032import org.apache.hadoop.classification.InterfaceStability;
033import org.apache.hadoop.conf.Configuration;

039/**
040 * An abstract class for the execution of a file system command
041 */
042@InterfaceAudience.Private
043@InterfaceStability.Evolving
044
045abstract public class Command extends Configured {
046  /** default name of the command */
047  public static String NAME;
048  /** the command's usage switches and arguments format */
049  public static String USAGE;
050  /** the command's long description */
051  public static String DESCRIPTION;
052    
053  protected String[] args;
054  protected String name;
055  protected int exitCode = 0;
056  protected int numErrors = 0;
057  protected boolean recursive = false;
058  private int depth = 0;
059  protected ArrayList<Exception> exceptions = new ArrayList<Exception>();
060
061  private static final Log LOG = LogFactory.getLog(Command.class);
062
063  /** allows stdout to be captured if necessary */
064  public PrintStream out = System.out;
065  /** allows stderr to be captured if necessary */
066  public PrintStream err = System.err;
067  /** allows the command factory to be used if necessary */
068  private CommandFactory commandFactory = null;
069
070  /** Constructor */
071  protected Command() {
072    out = System.out;
073    err = System.err;
074  }
075  
076  /** Constructor */
077  protected Command(Configuration conf) {
078    super(conf);
079  }
080  
081  /** @return the command's name excluding the leading character - */
082  abstract public String getCommandName();
083  
084  protected void setRecursive(boolean flag) {
085    recursive = flag;
086  }
087  
088  protected boolean isRecursive() {
089    return recursive;
090  }
091
092  protected int getDepth() {
093    return depth;
094  }
095  
096  /** 
097   * Execute the command on the input path
098   * 
099   * @param path the input path
100   * @throws IOException if any error occurs
101   */
102  abstract protected void run(Path path) throws IOException;
103  
104  /** 
105   * For each source path, execute the command
106   * 
107   * @return 0 if it runs successfully; -1 if it fails
108   */
109  public int runAll() {
110    int exitCode = 0;
111    for (String src : args) {
112      try {
113        PathData[] srcs = PathData.expandAsGlob(src, getConf());
114        for (PathData s : srcs) {
115          run(s.path);
116        }
117      } catch (IOException e) {
118        exitCode = -1;
119        displayError(e);
120      }
121    }
122    return exitCode;
123  }
124
125  /** sets the command factory for later use */
126  public void setCommandFactory(CommandFactory factory) {
127    this.commandFactory = factory;
128  }
129  /** retrieves the command factory */
130  protected CommandFactory getCommandFactory() {
131    return this.commandFactory;
132  }
133
134  /**
135   * Invokes the command handler.  The default behavior is to process options,
136   * expand arguments, and then process each argument.
137   * <pre>
138   * run
139   * |-> {@link #processOptions(LinkedList)}
140   * \-> {@link #processRawArguments(LinkedList)}
141   *      |-> {@link #expandArguments(LinkedList)}
142   *      |   \-> {@link #expandArgument(String)}*
143   *      \-> {@link #processArguments(LinkedList)}
144   *          |-> {@link #processArgument(PathData)}*
145   *          |   |-> {@link #processPathArgument(PathData)}
146   *          |   \-> {@link #processPaths(PathData, PathData...)}
147   *          |        \-> {@link #processPath(PathData)}*
148   *          \-> {@link #processNonexistentPath(PathData)}
149   * </pre>
150   * Most commands will chose to implement just
151   * {@link #processOptions(LinkedList)} and {@link #processPath(PathData)}
152   * 
153   * @param argv the list of command line arguments
154   * @return the exit code for the command
155   * @throws IllegalArgumentException if called with invalid arguments
156   */
157  public int run(String...argv) {
158    LinkedList<String> args = new LinkedList<String>(Arrays.asList(argv));
159    try {
160      if (isDeprecated()) {
161        displayWarning(
162            "DEPRECATED: Please use '"+ getReplacementCommand() + "' instead.");
163      }
164      processOptions(args);
165      processRawArguments(args);
166    } catch (IOException e) {
167      displayError(e);
168    }
169    
170    return (numErrors == 0) ? exitCode : exitCodeForError();
171  }
172
173  /**
174   * The exit code to be returned if any errors occur during execution.
175   * This method is needed to account for the inconsistency in the exit
176   * codes returned by various commands.
177   * @return a non-zero exit code
178   */
179  protected int exitCodeForError() { return 1; }
180  
181  /**
182   * Must be implemented by commands to process the command line flags and
183   * check the bounds of the remaining arguments.  If an
184   * IllegalArgumentException is thrown, the FsShell object will print the
185   * short usage of the command.
186   * @param args the command line arguments
187   * @throws IOException
188   */
189  protected void processOptions(LinkedList<String> args) throws IOException {}
190
191  /**
192   * Allows commands that don't use paths to handle the raw arguments.
193   * Default behavior is to expand the arguments via
194   * {@link #expandArguments(LinkedList)} and pass the resulting list to
195   * {@link #processArguments(LinkedList)} 
196   * @param args the list of argument strings
197   * @throws IOException
198   */
199  protected void processRawArguments(LinkedList<String> args)
200  throws IOException {
201    processArguments(expandArguments(args));
202  }
203
204  /**
205   *  Expands a list of arguments into {@link PathData} objects.  The default
206   *  behavior is to call {@link #expandArgument(String)} on each element
207   *  which by default globs the argument.  The loop catches IOExceptions,
208   *  increments the error count, and displays the exception.
209   * @param args strings to expand into {@link PathData} objects
210   * @return list of all {@link PathData} objects the arguments
211   * @throws IOException if anything goes wrong...
212   */
213  protected LinkedList<PathData> expandArguments(LinkedList<String> args)
214  throws IOException {
215    LinkedList<PathData> expandedArgs = new LinkedList<PathData>();
216    for (String arg : args) {
217      try {
218        expandedArgs.addAll(expandArgument(arg));
219      } catch (IOException e) { // other exceptions are probably nasty
220        displayError(e);
221      }
222    }
223    return expandedArgs;
224  }
225
226  /**
227   * Expand the given argument into a list of {@link PathData} objects.
228   * The default behavior is to expand globs.  Commands may override to
229   * perform other expansions on an argument.
230   * @param arg string pattern to expand
231   * @return list of {@link PathData} objects
232   * @throws IOException if anything goes wrong...
233   */
234  protected List<PathData> expandArgument(String arg) throws IOException {
235    PathData[] items = PathData.expandAsGlob(arg, getConf());
236    if (items.length == 0) {
237      // it's a glob that failed to match
238      throw new PathNotFoundException(arg);
239    }
240    return Arrays.asList(items);
241  }
242
243  /**
244   *  Processes the command's list of expanded arguments.
245   *  {@link #processArgument(PathData)} will be invoked with each item
246   *  in the list.  The loop catches IOExceptions, increments the error
247   *  count, and displays the exception.
248   *  @param args a list of {@link PathData} to process
249   *  @throws IOException if anything goes wrong... 
250   */
251  protected void processArguments(LinkedList<PathData> args)
252  throws IOException {
253    for (PathData arg : args) {
254      try {
255        processArgument(arg);
256      } catch (IOException e) {
257        displayError(e);
258      }
259    }
260  }
261
262  /**
263   * Processes a {@link PathData} item, calling
264   * {@link #processPathArgument(PathData)} or
265   * {@link #processNonexistentPath(PathData)} on each item.
266   * @param item {@link PathData} item to process
267   * @throws IOException if anything goes wrong...
268   */
269  protected void processArgument(PathData item) throws IOException {
270    if (item.exists) {
271      processPathArgument(item);
272    } else {
273      processNonexistentPath(item);
274    }
275  }
276
277  /**
278   *  This is the last chance to modify an argument before going into the
279   *  (possibly) recursive {@link #processPaths(PathData, PathData...)}
280   *  -> {@link #processPath(PathData)} loop.  Ex.  ls and du use this to
281   *  expand out directories.
282   *  @param item a {@link PathData} representing a path which exists
283   *  @throws IOException if anything goes wrong... 
284   */
285  protected void processPathArgument(PathData item) throws IOException {
286    // null indicates that the call is not via recursion, ie. there is
287    // no parent directory that was expanded
288    depth = 0;
289    processPaths(null, item);
290  }
291  
292  /**
293   *  Provides a hook for handling paths that don't exist.  By default it
294   *  will throw an exception.  Primarily overriden by commands that create
295   *  paths such as mkdir or touch.
296   *  @param item the {@link PathData} that doesn't exist
297   *  @throws FileNotFoundException if arg is a path and it doesn't exist
298   *  @throws IOException if anything else goes wrong... 
299   */
300  protected void processNonexistentPath(PathData item) throws IOException {
301    throw new PathNotFoundException(item.toString());
302  }
303
304  /**
305   *  Iterates over the given expanded paths and invokes
306   *  {@link #processPath(PathData)} on each element.  If "recursive" is true,
307   *  will do a post-visit DFS on directories.
308   *  @param parent if called via a recurse, will be the parent dir, else null
309   *  @param items a list of {@link PathData} objects to process
310   *  @throws IOException if anything goes wrong...
311   */
312  protected void processPaths(PathData parent, PathData ... items)
313  throws IOException {
314    // TODO: this really should be iterative
315    for (PathData item : items) {
316      try {
317        processPath(item);
318        if (recursive && isPathRecursable(item)) {
319          recursePath(item);
320        }
321        postProcessPath(item);
322      } catch (IOException e) {
323        displayError(e);
324      }
325    }
326  }
327
328  /**
329   * Determines whether a {@link PathData} item is recursable. Default
330   * implementation is to recurse directories but can be overridden to recurse
331   * through symbolic links.
332   *
333   * @param item
334   *          a {@link PathData} object
335   * @return true if the item is recursable, false otherwise
336   * @throws IOException
337   *           if anything goes wrong in the user-implementation
338   */
339  protected boolean isPathRecursable(PathData item) throws IOException {
340    return item.stat.isDirectory();
341  }
342
343  /**
344   * Hook for commands to implement an operation to be applied on each
345   * path for the command.  Note implementation of this method is optional
346   * if earlier methods in the chain handle the operation.
347   * @param item a {@link PathData} object
348   * @throws RuntimeException if invoked but not implemented
349   * @throws IOException if anything else goes wrong in the user-implementation
350   */  
351  protected void processPath(PathData item) throws IOException {
352    throw new RuntimeException("processPath() is not implemented");    
353  }
354
355  /**
356   * Hook for commands to implement an operation to be applied on each
357   * path for the command after being processed successfully
358   * @param item a {@link PathData} object
359   * @throws IOException if anything goes wrong...
360   */
361  protected void postProcessPath(PathData item) throws IOException {    
362  }
363
364  /**
365   *  Gets the directory listing for a path and invokes
366   *  {@link #processPaths(PathData, PathData...)}
367   *  @param item {@link PathData} for directory to recurse into
368   *  @throws IOException if anything goes wrong...
369   */
370  protected void recursePath(PathData item) throws IOException {
371    try {
372      depth++;
373      processPaths(item, item.getDirectoryContents());
374    } finally {
375      depth--;
376    }
377  }
378
379  /**
380   * Display an exception prefaced with the command name.  Also increments
381   * the error count for the command which will result in a non-zero exit
382   * code.
383   * @param e exception to display
384   */
385  public void displayError(Exception e) {
386    // build up a list of exceptions that occurred
387    exceptions.add(e);
388    
389    String errorMessage = e.getLocalizedMessage();
390    if (errorMessage == null) {
391      // this is an unexpected condition, so dump the whole exception since
392      // it's probably a nasty internal error where the backtrace would be
393      // useful
394      errorMessage = StringUtils.stringifyException(e);
395      LOG.debug(errorMessage);
396    } else {
397      errorMessage = errorMessage.split("\n", 2)[0];
398    }
399    displayError(errorMessage);
400  }
401  
402  /**
403   * Display an error string prefaced with the command name.  Also increments
404   * the error count for the command which will result in a non-zero exit
405   * code.
406   * @param message error message to display
407   */
408  public void displayError(String message) {
409    numErrors++;
410    displayWarning(message);
411  }
412  
413  /**
414   * Display an warning string prefaced with the command name.
415   * @param message warning message to display
416   */
417  public void displayWarning(String message) {
418    err.println(getName() + ": " + message);
419  }
420  
421  /**
422   * The name of the command.  Will first try to use the assigned name
423   * else fallback to the command's preferred name
424   * @return name of the command
425   */
426  public String getName() {
427    return (name == null)
428      ? getCommandField("NAME")
429      : name.startsWith("-") ? name.substring(1) : name;
430  }
431
432  /**
433   * Define the name of the command.
434   * @param name as invoked
435   */
436  public void setName(String name) {
437    this.name = name;
438  }
439  
440  /**
441   * The short usage suitable for the synopsis
442   * @return "name options"
443   */
444  public String getUsage() {
445    String cmd = "-" + getName();
446    String usage = isDeprecated() ? "" : getCommandField("USAGE");
447    return usage.isEmpty() ? cmd : cmd + " " + usage; 
448  }
449
450  /**
451   * The long usage suitable for help output
452   * @return text of the usage
453   */
454  public String getDescription() {
455    return isDeprecated()
456      ? "(DEPRECATED) Same as '" + getReplacementCommand() + "'"
457      : getCommandField("DESCRIPTION");
458  }
459
460  /**
461   * Is the command deprecated?
462   * @return boolean
463   */
464  public final boolean isDeprecated() {
465    return (getReplacementCommand() != null);
466  }
467  
468  /**
469   * The replacement for a deprecated command
470   * @return null if not deprecated, else alternative command
471   */
472  public String getReplacementCommand() {
473    return null;
474  }
475
476  /**
477   * Get a public static class field
478   * @param field the field to retrieve
479   * @return String of the field
480   */
481  private String getCommandField(String field) {
482    String value;
483    try {
484      Field f = this.getClass().getDeclaredField(field);
485      f.setAccessible(true);
486      value = f.get(this).toString();
487    } catch (Exception e) {
488      throw new RuntimeException(
489          "failed to get " + this.getClass().getSimpleName()+"."+field, e);
490    }
491    return value;
492  }
