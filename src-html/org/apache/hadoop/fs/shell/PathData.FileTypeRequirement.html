041/**
042 * Encapsulates a Path (path), its FileStatus (stat), and its FileSystem (fs).
043 * PathData ensures that the returned path string will be the same as the
044 * one passed in during initialization (unlike Path objects which can
045 * modify the path string).
046 * The stat field will be null if the path does not exist.
047 */
048@InterfaceAudience.Private
049@InterfaceStability.Unstable
050
051public class PathData implements Comparable<PathData> {
052  protected final URI uri;
053  public final FileSystem fs;
054  public final Path path;
055  public FileStatus stat;
056  public boolean exists;
057
058  /* True if the URI scheme was not present in the pathString but inferred.
059   */
060  private boolean inferredSchemeFromPath = false;
061
062  /**
063   *  Pre-compiled regular expressions to detect path formats.
064   */
065  private static final Pattern potentialUri =
066      Pattern.compile("^[a-zA-Z][a-zA-Z0-9+-.]+:");
067  private static final Pattern windowsNonUriAbsolutePath1 =
068      Pattern.compile("^/?[a-zA-Z]:\\\\");
069  private static final Pattern windowsNonUriAbsolutePath2 =
070      Pattern.compile("^/?[a-zA-Z]:/");
071
072  /**
073   * Creates an object to wrap the given parameters as fields.  The string
074   * used to create the path will be recorded since the Path object does not
075   * return exactly the same string used to initialize it
076   * @param pathString a string for a path
077   * @param conf the configuration file
078   * @throws IOException if anything goes wrong...
079   */
080  public PathData(String pathString, Configuration conf) throws IOException {
081    this(FileSystem.get(stringToUri(pathString), conf), pathString);
082  }

085   * Creates an object to wrap the given parameters as fields.  The string

089   * @param conf the configuration file
090   * @throws IOException if anything goes wrong...

093    this(FileSystem.getLocal(conf), localPath.getPath());

096  /**
097   * Looks up the file status for a path.  If the path
098   * doesn't exist, then the status will be null
099   * @param fs the FileSystem for the path
100   * @param pathString a string for a path 
101   * @throws IOException if anything goes wrong
102   */
103  private PathData(FileSystem fs, String pathString) throws IOException {
104    this(fs, pathString, lookupStat(fs, pathString, true));
105  }

107  /**
108   * Validates the given Windows path.
109   * @param pathString a String of the path suppliued by the user.
110   * @return true if the URI scheme was not present in the pathString but
111   * inferred; false, otherwise.
112   * @throws IOException if anything goes wrong
113   */
114  private static boolean checkIfSchemeInferredFromPath(String pathString)
115  throws IOException
116  {
117    if (windowsNonUriAbsolutePath1.matcher(pathString).find()) {
118      // Forward slashes disallowed in a backslash-separated path.
119      if (pathString.indexOf('/') != -1) {
120        throw new IOException("Invalid path string " + pathString);
121      }
122
123      return true;
124    }
125
126    // Is it a forward slash-separated absolute path?
127    if (windowsNonUriAbsolutePath2.matcher(pathString).find()) {
128      return true;
129    }
130
131    // Does it look like a URI? If so then just leave it alone.
132    if (potentialUri.matcher(pathString).find()) {
133      return false;
134    }
135
136    // Looks like a relative path on Windows.
137    return false;
138  }

140  /**
141   * Creates an object to wrap the given parameters as fields.  The string
142   * used to create the path will be recorded since the Path object does not
143   * return exactly the same string used to initialize it.
144   * @param fs the FileSystem
145   * @param pathString a String of the path
146   * @param stat the FileStatus (may be null if the path doesn't exist)
147   */
148  private PathData(FileSystem fs, String pathString, FileStatus stat)
149  throws IOException {
150    this.fs = fs;
151    this.uri = stringToUri(pathString);
152    this.path = fs.makeQualified(new Path(uri));
153    setStat(stat);
154
155    if (Path.WINDOWS) {
156      inferredSchemeFromPath = checkIfSchemeInferredFromPath(pathString);
157    }
158  }

160  // need a static method for the ctor above
161  /**
162   * Get the FileStatus info
163   * @param ignoreFNF if true, stat will be null if the path doesn't exist
164   * @return FileStatus for the given path
165   * @throws IOException if anything goes wrong
166   */
167  private static
168  FileStatus lookupStat(FileSystem fs, String pathString, boolean ignoreFNF)
169  throws IOException {
170    FileStatus status = null;
171    try {
172      status = fs.getFileStatus(new Path(pathString));
173    } catch (FileNotFoundException e) {
174      if (!ignoreFNF) throw new PathNotFoundException(pathString);
175    }
176    // TODO: should consider wrapping other exceptions into Path*Exceptions
177    return status;
178  }
179  
180  private void setStat(FileStatus stat) {
181    this.stat = stat;
182    exists = (stat != null);
183  }
184
185  /**
186   * Updates the paths's file status
187   * @return the updated FileStatus
188   * @throws IOException if anything goes wrong...
189   */
190  public FileStatus refreshStatus() throws IOException {
191    FileStatus status = null;
192    try {
193      status = lookupStat(fs, toString(), false);
194    } finally {
195      // always set the status.  the caller must get the correct result
196      // if it catches the exception and later interrogates the status
197      setStat(status);
198    }
199    return status;
200  }

202  protected enum FileTypeRequirement {
203    SHOULD_NOT_BE_DIRECTORY, SHOULD_BE_DIRECTORY
204  };
205
206  /**
207   * Ensure that the file exists and if it is or is not a directory
208   * @param typeRequirement Set it to the desired requirement.
209   * @throws PathIOException if file doesn't exist or the type does not match
210   * what was specified in typeRequirement.
211   */
212  private void checkIfExists(FileTypeRequirement typeRequirement) 
213  throws PathIOException {
214    if (!exists) {
215      throw new PathNotFoundException(toString());      
216    }
217
218    if ((typeRequirement == FileTypeRequirement.SHOULD_BE_DIRECTORY)
219       && !stat.isDirectory()) {
220      throw new PathIsNotDirectoryException(toString());
221    } else if ((typeRequirement == FileTypeRequirement.SHOULD_NOT_BE_DIRECTORY)
222              && stat.isDirectory()) {
223      throw new PathIsDirectoryException(toString());
224    }
225  }

237  /**
238   * Test if the parent directory exists
239   * @return boolean indicating parent exists
240   * @throws IOException upon unexpected error
241   */
242  public boolean parentExists() throws IOException {
243    return representsDirectory()
244        ? fs.exists(path) : fs.exists(path.getParent());
245  }
246
247  /**
248   * Check if the path represents a directory as determined by the basename
249   * being "." or "..", or the path ending with a directory separator 
250   * @return boolean if this represents a directory
251   */
252  public boolean representsDirectory() {
253    String uriPath = uri.getPath();
254    String name = uriPath.substring(uriPath.lastIndexOf("/")+1);
255    // Path will munch off the chars that indicate a dir, so there's no way
256    // to perform this test except by examining the raw basename we maintain
257    return (name.isEmpty() || name.equals(".") || name.equals(".."));
258  }

260  /**
261   * Returns a list of PathData objects of the items contained in the given
262   * directory.
263   * @return list of PathData objects for its children
264   * @throws IOException if anything else goes wrong...
265   */
266  public PathData[] getDirectoryContents() throws IOException {
267    checkIfExists(FileTypeRequirement.SHOULD_BE_DIRECTORY);
268    FileStatus[] stats = fs.listStatus(path);
269    PathData[] items = new PathData[stats.length];
270    for (int i=0; i < stats.length; i++) {
271      // preserve relative paths
272      String child = getStringForChildPath(stats[i].getPath());
273      items[i] = new PathData(fs, child, stats[i]);
274    }
275    Arrays.sort(items);
276    return items;
277  }

279  /**
280   * Creates a new object for a child entry in this directory
281   * @param child the basename will be appended to this object's path
282   * @return PathData for the child
283   * @throws IOException if this object does not exist or is not a directory
284   */
285  public PathData getPathDataForChild(PathData child) throws IOException {
286    checkIfExists(FileTypeRequirement.SHOULD_BE_DIRECTORY);
287    return new PathData(fs, getStringForChildPath(child.path));
288  }

290  /**
291   * Given a child of this directory, use the directory's path and the child's
292   * basename to construct the string to the child.  This preserves relative
293   * paths since Path will fully qualify.
294   * @param childPath a path contained within this directory
295   * @return String of the path relative to this directory
296   */
297  private String getStringForChildPath(Path childPath) {
298    String basename = childPath.getName();
299    if (Path.CUR_DIR.equals(toString())) {
300      return basename;
301    }
302    // check getPath() so scheme slashes aren't considered part of the path
303    String separator = uri.getPath().endsWith(Path.SEPARATOR)
304        ? "" : Path.SEPARATOR;
305    return uriToString(uri, inferredSchemeFromPath) + separator + basename;
306  }

308  protected enum PathType { HAS_SCHEME, SCHEMELESS_ABSOLUTE, RELATIVE };
309  
310  /**
311   * Expand the given path as a glob pattern.  Non-existent paths do not
312   * throw an exception because creation commands like touch and mkdir need
313   * to create them.  The "stat" field will be null if the path does not
314   * exist.
315   * @param pattern the pattern to expand as a glob
316   * @param conf the hadoop configuration
317   * @return list of {@link PathData} objects.  if the pattern is not a glob,
318   * and does not exist, the list will contain a single PathData with a null
319   * stat 
320   * @throws IOException anything else goes wrong...
321   */
322  public static PathData[] expandAsGlob(String pattern, Configuration conf)
323  throws IOException {
324    Path globPath = new Path(pattern);
325    FileSystem fs = globPath.getFileSystem(conf);    
326    FileStatus[] stats = fs.globStatus(globPath);
327    PathData[] items = null;
328    
329    if (stats == null) {
330      // remove any quoting in the glob pattern
331      pattern = pattern.replaceAll("\\\\(.)", "$1");
332      // not a glob & file not found, so add the path with a null stat
333      items = new PathData[]{ new PathData(fs, pattern, null) };
334    } else {
335      // figure out what type of glob path was given, will convert globbed
336      // paths to match the type to preserve relativity
337      PathType globType;
338      URI globUri = globPath.toUri();
339      if (globUri.getScheme() != null) {
340        globType = PathType.HAS_SCHEME;
341      } else if (!globUri.getPath().isEmpty() &&
342                 new Path(globUri.getPath()).isAbsolute()) {
343        globType = PathType.SCHEMELESS_ABSOLUTE;
344      } else {
345        globType = PathType.RELATIVE;
346      }
347
348      // convert stats to PathData
349      items = new PathData[stats.length];
350      int i=0;
351      for (FileStatus stat : stats) {
352        URI matchUri = stat.getPath().toUri();
353        String globMatch = null;
354        switch (globType) {
355          case HAS_SCHEME: // use as-is, but remove authority if necessary
356            if (globUri.getAuthority() == null) {
357              matchUri = removeAuthority(matchUri);
358            }
359            globMatch = uriToString(matchUri, false);
360            break;
361          case SCHEMELESS_ABSOLUTE: // take just the uri's path
362            globMatch = matchUri.getPath();
363            break;
364          case RELATIVE: // make it relative to the current working dir
365            URI cwdUri = fs.getWorkingDirectory().toUri();
366            globMatch = relativize(cwdUri, matchUri, stat.isDirectory());
367            break;
368        }
369        items[i++] = new PathData(fs, globMatch, stat);
370      }
371    }
372    Arrays.sort(items);
373    return items;
374  }

376  private static URI removeAuthority(URI uri) {
377    try {
378      uri = new URI(
379          uri.getScheme(), "",
380          uri.getPath(), uri.getQuery(), uri.getFragment()
381      );
382    } catch (URISyntaxException e) {
383      throw new IllegalArgumentException(e.getLocalizedMessage());
384    }
385    return uri;
386  }
387  
388  private static String relativize(URI cwdUri, URI srcUri, boolean isDir) {
389    String uriPath = srcUri.getPath();
390    String cwdPath = cwdUri.getPath();
391    if (cwdPath.equals(uriPath)) {
392      return Path.CUR_DIR;
393    }
394
395    // find common ancestor
396    int lastSep = findLongestDirPrefix(cwdPath, uriPath, isDir);
397    
398    StringBuilder relPath = new StringBuilder();    
399    // take the remaining path fragment after the ancestor
400    if (lastSep < uriPath.length()) {
401      relPath.append(uriPath.substring(lastSep+1));
402    }
403
404    // if cwd has a path fragment after the ancestor, convert them to ".."
405    if (lastSep < cwdPath.length()) {
406      while (lastSep != -1) {
407        if (relPath.length() != 0) relPath.insert(0, Path.SEPARATOR);
408        relPath.insert(0, "..");
409        lastSep = cwdPath.indexOf(Path.SEPARATOR, lastSep+1);
410      }
411    }
412    return relPath.toString();
413  }
414
415  private static int findLongestDirPrefix(String cwd, String path, boolean isDir) {
416    // add the path separator to dirs to simplify finding the longest match
417    if (!cwd.endsWith(Path.SEPARATOR)) {
418      cwd += Path.SEPARATOR;
419    }
420    if (isDir && !path.endsWith(Path.SEPARATOR)) {
421      path += Path.SEPARATOR;
422    }
423
424    // find longest directory prefix 
425    int len = Math.min(cwd.length(), path.length());
426    int lastSep = -1;
427    for (int i=0; i < len; i++) {
428      if (cwd.charAt(i) != path.charAt(i)) break;
429      if (cwd.charAt(i) == Path.SEPARATOR_CHAR) lastSep = i;
430    }
431    return lastSep;
432  }

434  /**
435   * Returns the printable version of the path that is either the path
436   * as given on the commandline, or the full path
437   * @return String of the path
438   */
439  @Override
440  public String toString() {
441    return uriToString(uri, inferredSchemeFromPath);
442  }

444  private static String uriToString(URI uri, boolean inferredSchemeFromPath) {
445    String scheme = uri.getScheme();
446    // No interpretation of symbols. Just decode % escaped chars.
447    String decodedRemainder = uri.getSchemeSpecificPart();
448
449    // Drop the scheme if it was inferred to ensure fidelity between
450    // the input and output path strings.
451    if ((scheme == null) || (inferredSchemeFromPath)) {
452      if (Path.isWindowsAbsolutePath(decodedRemainder, true)) {
453        // Strip the leading '/' added in stringToUri so users see a valid
454        // Windows path.
455        decodedRemainder = decodedRemainder.substring(1);
456      }
457      return decodedRemainder;
458    } else {
459      StringBuilder buffer = new StringBuilder();
460      buffer.append(scheme);
461      buffer.append(":");
462      buffer.append(decodedRemainder);
463      return buffer.toString();
464    }
465  }

479  /** Normalize the given Windows path string. This does the following:
480   *    1. Adds "file:" scheme for absolute paths.
481   *    2. Ensures the scheme-specific part starts with '/' per RFC2396.
482   *    3. Replaces backslash path separators with forward slashes.
483   *    @param pathString Path string supplied by the user.
484   *    @return normalized absolute path string. Returns the input string
485   *            if it is not a Windows absolute path.
486   */
487  private static String normalizeWindowsPath(String pathString)
488  throws IOException
489  {
490    if (!Path.WINDOWS) {
491      return pathString;
492    }
493
494    boolean slashed =
495        ((pathString.length() >= 1) && (pathString.charAt(0) == '/'));
496
497    // Is it a backslash-separated absolute path?
498    if (windowsNonUriAbsolutePath1.matcher(pathString).find()) {
499      // Forward slashes disallowed in a backslash-separated path.
500      if (pathString.indexOf('/') != -1) {
501        throw new IOException("Invalid path string " + pathString);
502      }
503
504      pathString = pathString.replace('\\', '/');
505      return "file:" + (slashed ? "" : "/") + pathString;
506    }
507
508    // Is it a forward slash-separated absolute path?
509    if (windowsNonUriAbsolutePath2.matcher(pathString).find()) {
510      return "file:" + (slashed ? "" : "/") + pathString;
511    }
512
513    // Is it a backslash-separated relative file path (no scheme and
514    // no drive-letter specifier)?
515    if ((pathString.indexOf(':') == -1) && (pathString.indexOf('\\') != -1)) {
516      pathString = pathString.replace('\\', '/');
517    }
518
519    return pathString;
520  }

522  /** Construct a URI from a String with unescaped special characters
523   *  that have non-standard semantics. e.g. /, ?, #. A custom parsing
524   *  is needed to prevent misbehavior.
525   *  @param pathString The input path in string form
526   *  @return URI
527   */
528  private static URI stringToUri(String pathString) throws IOException {
529    // We can't use 'new URI(String)' directly. Since it doesn't do quoting
530    // internally, the internal parser may fail or break the string at wrong
531    // places. Use of multi-argument ctors will quote those chars for us,
532    // but we need to do our own parsing and assembly.
533    
534    // parse uri components
535    String scheme = null;
536    String authority = null;
537    int start = 0;
538
539    pathString = normalizeWindowsPath(pathString);
540
541    // parse uri scheme, if any
542    int colon = pathString.indexOf(':');
543    int slash = pathString.indexOf('/');
544    if (colon > 0 && (slash == colon +1)) {
545      // has a non zero-length scheme
546      scheme = pathString.substring(0, colon);
547      start = colon + 1;
548    }
549
550    // parse uri authority, if any
551    if (pathString.startsWith("//", start) &&
552        (pathString.length()-start > 2)) {
553      start += 2;
554      int nextSlash = pathString.indexOf('/', start);
555      int authEnd = nextSlash > 0 ? nextSlash : pathString.length();
556      authority = pathString.substring(start, authEnd);
557      start = authEnd;
558    }
559    // uri path is the rest of the string. ? or # are not interpreted,
560    // but any occurrence of them will be quoted by the URI ctor.
561    String path = pathString.substring(start, pathString.length());
562
563    // Construct the URI
564    try {
565      return new URI(scheme, authority, path, null, null);
566    } catch (URISyntaxException e) {
567      throw new IllegalArgumentException(e);
568    }
569  }
