019package org.apache.hadoop.metrics2.source;
020
021import java.lang.management.ManagementFactory;
022import java.lang.management.MemoryMXBean;
023import java.lang.management.MemoryUsage;
024import java.lang.management.ThreadInfo;
025import java.lang.management.ThreadMXBean;
026import java.lang.management.GarbageCollectorMXBean;
027import java.util.List;
028import java.util.concurrent.ConcurrentHashMap;
029
030import org.apache.hadoop.classification.InterfaceAudience;
031import org.apache.hadoop.log.metrics.EventCounter;
032import org.apache.hadoop.metrics2.MetricsCollector;
033import org.apache.hadoop.metrics2.MetricsInfo;
034import org.apache.hadoop.metrics2.MetricsRecordBuilder;
035import org.apache.hadoop.metrics2.MetricsSource;
036import org.apache.hadoop.metrics2.MetricsSystem;
037import org.apache.hadoop.metrics2.lib.DefaultMetricsSystem;
038import org.apache.hadoop.metrics2.lib.Interns;
039import static org.apache.hadoop.metrics2.source.JvmMetricsInfo.*;
040import static org.apache.hadoop.metrics2.impl.MsInfo.*;
041import org.apache.hadoop.util.JvmPauseMonitor;
042
043/**
044 * JVM and logging related metrics.
045 * Mostly used by various servers as a part of the metrics they export.
046 */
047@InterfaceAudience.Private
048public class JvmMetrics implements MetricsSource {
049  enum Singleton {
050    INSTANCE;
051
052    JvmMetrics impl;
053
054    synchronized JvmMetrics init(String processName, String sessionId) {
055      if (impl == null) {
056        impl = create(processName, sessionId, DefaultMetricsSystem.instance());
057      }
058      return impl;
059    }
060  }
061
062  static final float M = 1024*1024;
063
064  final MemoryMXBean memoryMXBean = ManagementFactory.getMemoryMXBean();
065  final List<GarbageCollectorMXBean> gcBeans =
066      ManagementFactory.getGarbageCollectorMXBeans();
067  final ThreadMXBean threadMXBean = ManagementFactory.getThreadMXBean();
068  final String processName, sessionId;
069  private JvmPauseMonitor pauseMonitor = null;
070  final ConcurrentHashMap<String, MetricsInfo[]> gcInfoCache =
071      new ConcurrentHashMap<String, MetricsInfo[]>();
072
073  JvmMetrics(String processName, String sessionId) {
074    this.processName = processName;
075    this.sessionId = sessionId;
076  }
077
078  public void setPauseMonitor(final JvmPauseMonitor pauseMonitor) {
079    this.pauseMonitor = pauseMonitor;
080  }
081
082  public static JvmMetrics create(String processName, String sessionId,
083                                  MetricsSystem ms) {
084    return ms.register(JvmMetrics.name(), JvmMetrics.description(),
085                       new JvmMetrics(processName, sessionId));
086  }
087
088  public static JvmMetrics initSingleton(String processName, String sessionId) {
089    return Singleton.INSTANCE.init(processName, sessionId);
090  }
091
092  @Override
093  public void getMetrics(MetricsCollector collector, boolean all) {
094    MetricsRecordBuilder rb = collector.addRecord(JvmMetrics)
095        .setContext("jvm").tag(ProcessName, processName)
096        .tag(SessionId, sessionId);
097    getMemoryUsage(rb);
098    getGcUsage(rb);
099    getThreadUsage(rb);
100    getEventCounters(rb);
101  }
102
103  private void getMemoryUsage(MetricsRecordBuilder rb) {
104    MemoryUsage memNonHeap = memoryMXBean.getNonHeapMemoryUsage();
105    MemoryUsage memHeap = memoryMXBean.getHeapMemoryUsage();
106    Runtime runtime = Runtime.getRuntime();
107    rb.addGauge(MemNonHeapUsedM, memNonHeap.getUsed() / M)
108      .addGauge(MemNonHeapCommittedM, memNonHeap.getCommitted() / M)
109      .addGauge(MemNonHeapMaxM, memNonHeap.getMax() / M)
110      .addGauge(MemHeapUsedM, memHeap.getUsed() / M)
111      .addGauge(MemHeapCommittedM, memHeap.getCommitted() / M)
112      .addGauge(MemHeapMaxM, memHeap.getMax() / M)
113      .addGauge(MemMaxM, runtime.maxMemory() / M);
114  }
115
116  private void getGcUsage(MetricsRecordBuilder rb) {
117    long count = 0;
118    long timeMillis = 0;
119    for (GarbageCollectorMXBean gcBean : gcBeans) {
120      long c = gcBean.getCollectionCount();
121      long t = gcBean.getCollectionTime();
122      MetricsInfo[] gcInfo = getGcInfo(gcBean.getName());
123      rb.addCounter(gcInfo[0], c).addCounter(gcInfo[1], t);
124      count += c;
125      timeMillis += t;
126    }
127    rb.addCounter(GcCount, count)
128      .addCounter(GcTimeMillis, timeMillis);
129    
130    if (pauseMonitor != null) {
131      rb.addCounter(GcNumWarnThresholdExceeded,
132          pauseMonitor.getNumGcWarnThreadholdExceeded());
133      rb.addCounter(GcNumInfoThresholdExceeded,
134          pauseMonitor.getNumGcInfoThresholdExceeded());
135      rb.addCounter(GcTotalExtraSleepTime,
136          pauseMonitor.getTotalGcExtraSleepTime());
137    }
138  }
139
140  private MetricsInfo[] getGcInfo(String gcName) {
141    MetricsInfo[] gcInfo = gcInfoCache.get(gcName);
142    if (gcInfo == null) {
143      gcInfo = new MetricsInfo[2];
144      gcInfo[0] = Interns.info("GcCount" + gcName, "GC Count for " + gcName);
145      gcInfo[1] = Interns
146          .info("GcTimeMillis" + gcName, "GC Time for " + gcName);
147      MetricsInfo[] previousGcInfo = gcInfoCache.putIfAbsent(gcName, gcInfo);
148      if (previousGcInfo != null) {
149        return previousGcInfo;
150      }
151    }
152    return gcInfo;
153  }
154
155  private void getThreadUsage(MetricsRecordBuilder rb) {
156    int threadsNew = 0;
157    int threadsRunnable = 0;
158    int threadsBlocked = 0;
159    int threadsWaiting = 0;
160    int threadsTimedWaiting = 0;
161    int threadsTerminated = 0;
162    long threadIds[] = threadMXBean.getAllThreadIds();
163    for (ThreadInfo threadInfo : threadMXBean.getThreadInfo(threadIds, 0)) {
164      if (threadInfo == null) continue; // race protection
165      switch (threadInfo.getThreadState()) {
166        case NEW:           threadsNew++;           break;
167        case RUNNABLE:      threadsRunnable++;      break;
168        case BLOCKED:       threadsBlocked++;       break;
169        case WAITING:       threadsWaiting++;       break;
170        case TIMED_WAITING: threadsTimedWaiting++;  break;
171        case TERMINATED:    threadsTerminated++;    break;
172      }
173    }
174    rb.addGauge(ThreadsNew, threadsNew)
175      .addGauge(ThreadsRunnable, threadsRunnable)
176      .addGauge(ThreadsBlocked, threadsBlocked)
177      .addGauge(ThreadsWaiting, threadsWaiting)
178      .addGauge(ThreadsTimedWaiting, threadsTimedWaiting)
179      .addGauge(ThreadsTerminated, threadsTerminated);
180  }
181
182  private void getEventCounters(MetricsRecordBuilder rb) {
183    rb.addCounter(LogFatal, EventCounter.getFatal())
184      .addCounter(LogError, EventCounter.getError())
185      .addCounter(LogWarn, EventCounter.getWarn())
186      .addCounter(LogInfo, EventCounter.getInfo());
187  }
188}
