144  /**
145   * Initialized the metrics system with a prefix.
146   * @param prefix  the system will look for configs with the prefix
147   * @return the metrics system object itself
148   */
149  @Override
150  public synchronized MetricsSystem init(String prefix) {
151    if (monitoring && !DefaultMetricsSystem.inMiniClusterMode()) {
152      LOG.warn(this.prefix +" metrics system already initialized!");
153      return this;
154    }
155    this.prefix = checkNotNull(prefix, "prefix");
156    ++refCount;
157    if (monitoring) {
158      // in mini cluster mode
159      LOG.info(this.prefix +" metrics system started (again)");
160      return this;
161    }
162    switch (initMode()) {
163      case NORMAL:
164        try { start(); }
165        catch (MetricsConfigException e) {
166          // Configuration errors (e.g., typos) should not be fatal.
167          // We can always start the metrics system later via JMX.
168          LOG.warn("Metrics system not started: "+ e.getMessage());
169          LOG.debug("Stacktrace: ", e);
170        }
171        break;
172      case STANDBY:
173        LOG.info(prefix +" metrics system started in standby mode");
174    }
175    initSystemMBean();
176    return this;
177  }

179  @Override
180  public synchronized void start() {
181    checkNotNull(prefix, "prefix");
182    if (monitoring) {
183      LOG.warn(prefix +" metrics system already started!",
184               new MetricsException("Illegal start"));
185      return;
186    }
187    for (Callback cb : callbacks) cb.preStart();
188    for (Callback cb : namedCallbacks.values()) cb.preStart();
189    configure(prefix);
190    startTimer();
191    monitoring = true;
192    LOG.info(prefix +" metrics system started");
193    for (Callback cb : callbacks) cb.postStart();
194    for (Callback cb : namedCallbacks.values()) cb.postStart();
195  }
196
197  @Override
198  public synchronized void stop() {
199    if (!monitoring && !DefaultMetricsSystem.inMiniClusterMode()) {
200      LOG.warn(prefix +" metrics system not yet started!",
201               new MetricsException("Illegal stop"));
202      return;
203    }
204    if (!monitoring) {
205      // in mini cluster mode
206      LOG.info(prefix +" metrics system stopped (again)");
207      return;
208    }
209    for (Callback cb : callbacks) cb.preStop();
210    for (Callback cb : namedCallbacks.values()) cb.preStop();
211    LOG.info("Stopping "+ prefix +" metrics system...");
212    stopTimer();
213    stopSources();
214    stopSinks();
215    clearConfigs();
216    monitoring = false;
217    LOG.info(prefix +" metrics system stopped.");
218    for (Callback cb : callbacks) cb.postStop();
219    for (Callback cb : namedCallbacks.values()) cb.postStop();
220  }

222  @Override public synchronized <T>
223  T register(String name, String desc, T source) {
224    MetricsSourceBuilder sb = MetricsAnnotations.newSourceBuilder(source);
225    final MetricsSource s = sb.build();
226    MetricsInfo si = sb.info();
227    String name2 = name == null ? si.name() : name;
228    final String finalDesc = desc == null ? si.description() : desc;
229    final String finalName = // be friendly to non-metrics tests
230        DefaultMetricsSystem.sourceName(name2, !monitoring);
231    allSources.put(finalName, s);
232    LOG.debug(finalName +", "+ finalDesc);
233    if (monitoring) {
234      registerSource(finalName, finalDesc, s);
235    }
236    // We want to re-register the source to pick up new config when the
237    // metrics system restarts.
238    register(finalName, new AbstractCallback() {
239      @Override public void postStart() {
240        registerSource(finalName, finalDesc, s);
241      }
242    });
243    return source;
244  }

246  @Override public synchronized
247  void unregisterSource(String name) {
248    if (sources.containsKey(name)) {
249      sources.get(name).stop();
250      sources.remove(name);
251    }
252    if (allSources.containsKey(name)) {
253      allSources.remove(name);
254    }
255    if (namedCallbacks.containsKey(name)) {
256      namedCallbacks.remove(name);
257    }
258  }

260  synchronized
261  void registerSource(String name, String desc, MetricsSource source) {
262    checkNotNull(config, "config");
263    MetricsConfig conf = sourceConfigs.get(name);
264    MetricsSourceAdapter sa = conf != null
265        ? new MetricsSourceAdapter(prefix, name, desc, source,
266                                   injectedTags, period, conf)
267        : new MetricsSourceAdapter(prefix, name, desc, source,
268          injectedTags, period, config.subset(SOURCE_KEY));
269    sources.put(name, sa);
270    sa.start();
271    LOG.debug("Registered source "+ name);
272  }

274  @Override public synchronized <T extends MetricsSink>
275  T register(final String name, final String description, final T sink) {
276    LOG.debug(name +", "+ description);
277    if (allSinks.containsKey(name)) {
278      LOG.warn("Sink "+ name +" already exists!");
279      return sink;
280    }
281    allSinks.put(name, sink);
282    if (config != null) {
283      registerSink(name, description, sink);
284    }
285    // We want to re-register the sink to pick up new config
286    // when the metrics system restarts.
287    register(name, new AbstractCallback() {
288      @Override public void postStart() {
289        register(name, description, sink);
290      }
291    });
292    return sink;
293  }
294
295  synchronized void registerSink(String name, String desc, MetricsSink sink) {
296    checkNotNull(config, "config");
297    MetricsConfig conf = sinkConfigs.get(name);
298    MetricsSinkAdapter sa = conf != null
299        ? newSink(name, desc, sink, conf)
300        : newSink(name, desc, sink, config.subset(SINK_KEY));
301    sinks.put(name, sa);
302    sa.start();
303    LOG.info("Registered sink "+ name);
304  }

314
315  private Object getProxyForCallback(final Callback callback) {
316    return Proxy.newProxyInstance(callback.getClass().getClassLoader(),
317        new Class<?>[] { Callback.class }, new InvocationHandler() {
318          @Override
319          public Object invoke(Object proxy, Method method, Object[] args)
320              throws Throwable {
321            try {
322              return method.invoke(callback, args);
323            }
324            catch (Exception e) {
325              // These are not considered fatal.
326              LOG.warn("Caught exception in callback " + method.getName(), e);
327            }
328            return null;
329          }
330        });
331  }

359  private synchronized void startTimer() {
360    if (timer != null) {
361      LOG.warn(prefix +" metrics system timer already started!");
362      return;
363    }
364    logicalTime = 0;
365    long millis = period * 1000;
366    timer = new Timer("Timer for '"+ prefix +"' metrics system", true);
367    timer.scheduleAtFixedRate(new TimerTask() {
368          public void run() {
369            try {
370              onTimerEvent();
371            }
372            catch (Exception e) {
373              LOG.warn(e);
374            }
375          }
376        }, millis, millis);
377    LOG.info("Scheduled snapshot period at "+ period +" second(s).");
378  }
379
380  synchronized void onTimerEvent() {
381    logicalTime += period;
382    if (sinks.size() > 0) {
383      publishMetrics(sampleMetrics(), false);
384    }
385  }

397  /**
398   * Sample all the sources for a snapshot of metrics/tags
399   * @return  the metrics buffer containing the snapshot
400   */
401  synchronized MetricsBuffer sampleMetrics() {
402    collector.clear();
403    MetricsBufferBuilder bufferBuilder = new MetricsBufferBuilder();
404
405    for (Entry<String, MetricsSourceAdapter> entry : sources.entrySet()) {
406      if (sourceFilter == null || sourceFilter.accepts(entry.getKey())) {
407        snapshotMetrics(entry.getValue(), bufferBuilder);
408      }
409    }
410    if (publishSelfMetrics) {
411      snapshotMetrics(sysSource, bufferBuilder);
412    }
413    MetricsBuffer buffer = bufferBuilder.get();
414    return buffer;
415  }
416
417  private void snapshotMetrics(MetricsSourceAdapter sa,
418                               MetricsBufferBuilder bufferBuilder) {
419    long startTime = Time.now();
420    bufferBuilder.add(sa.name(), sa.getMetrics(collector, true));
421    collector.clear();
422    snapshotStat.add(Time.now() - startTime);
423    LOG.debug("Snapshotted source "+ sa.name());
424  }
425
426  /**
427   * Publish a metrics snapshot to all the sinks
428   * @param buffer  the metrics snapshot to publish
429   * @param immediate  indicates that we should publish metrics immediately
430   *                   instead of using a separate thread.
431   */
432  synchronized void publishMetrics(MetricsBuffer buffer, boolean immediate) {
433    int dropped = 0;
434    for (MetricsSinkAdapter sa : sinks.values()) {
435      long startTime = Time.now();
436      boolean result;
437      if (immediate) {
438        result = sa.putMetricsImmediate(buffer); 
439      } else {
440        result = sa.putMetrics(buffer, logicalTime);
441      }
442      dropped += result ? 0 : 1;
443      publishStat.add(Time.now() - startTime);
444    }
445    droppedPubAll.incr(dropped);
446  }

448  private synchronized void stopTimer() {
449    if (timer == null) {
450      LOG.warn(prefix +" metrics system timer already stopped!");
451      return;
452    }
453    timer.cancel();
454    timer = null;
455  }

457  private synchronized void stopSources() {
458    for (Entry<String, MetricsSourceAdapter> entry : sources.entrySet()) {
459      MetricsSourceAdapter sa = entry.getValue();
460      LOG.debug("Stopping metrics source "+ entry.getKey() +
461          ": class=" + sa.source().getClass());
462      sa.stop();
463    }
464    sysSource.stop();
465    sources.clear();
466  }
467
468  private synchronized void stopSinks() {
469    for (Entry<String, MetricsSinkAdapter> entry : sinks.entrySet()) {
470      MetricsSinkAdapter sa = entry.getValue();
471      LOG.debug("Stopping metrics sink "+ entry.getKey() +
472          ": class=" + sa.sink().getClass());
473      sa.stop();
474    }
475    sinks.clear();
476  }

485  private synchronized void configureSystem() {
486    injectedTags.add(Interns.tag(MsInfo.Hostname, getHostname()));
487  }
488
489  private synchronized void configureSinks() {
490    sinkConfigs = config.getInstanceConfigs(SINK_KEY);
491    int confPeriod = 0;
492    for (Entry<String, MetricsConfig> entry : sinkConfigs.entrySet()) {
493      MetricsConfig conf = entry.getValue();
494      int sinkPeriod = conf.getInt(PERIOD_KEY, PERIOD_DEFAULT);
495      confPeriod = confPeriod == 0 ? sinkPeriod
496                                   : ArithmeticUtils.gcd(confPeriod, sinkPeriod);
497      String clsName = conf.getClassName("");
498      if (clsName == null) continue;  // sink can be registered later on
499      String sinkName = entry.getKey();
500      try {
501        MetricsSinkAdapter sa = newSink(sinkName,
502            conf.getString(DESC_KEY, sinkName), conf);
503        sa.start();
504        sinks.put(sinkName, sa);
505      }
506      catch (Exception e) {
507        LOG.warn("Error creating sink '"+ sinkName +"'", e);
508      }
509    }
510    period = confPeriod > 0 ? confPeriod
511                            : config.getInt(PERIOD_KEY, PERIOD_DEFAULT);
512  }

514  static MetricsSinkAdapter newSink(String name, String desc, MetricsSink sink,
515                                    MetricsConfig conf) {
516    return new MetricsSinkAdapter(name, desc, sink, conf.getString(CONTEXT_KEY),
517        conf.getFilter(SOURCE_FILTER_KEY),
518        conf.getFilter(RECORD_FILTER_KEY),
519        conf.getFilter(METRIC_FILTER_KEY),
520        conf.getInt(PERIOD_KEY, PERIOD_DEFAULT),
521        conf.getInt(QUEUE_CAPACITY_KEY, QUEUE_CAPACITY_DEFAULT),
522        conf.getInt(RETRY_DELAY_KEY, RETRY_DELAY_DEFAULT),
523        conf.getFloat(RETRY_BACKOFF_KEY, RETRY_BACKOFF_DEFAULT),
524        conf.getInt(RETRY_COUNT_KEY, RETRY_COUNT_DEFAULT));
525  }

528                                    MetricsConfig conf) {

545  static String getHostname() {
546    try {
547      return InetAddress.getLocalHost().getHostName();
548    }

555  private void registerSystemSource() {
556    MetricsConfig sysConf = sourceConfigs.get(MS_NAME);
557    sysSource = new MetricsSourceAdapter(prefix, MS_STATS_NAME, MS_STATS_DESC,
558        MetricsAnnotations.makeSource(this), injectedTags, period,
559        sysConf == null ? config.subset(SOURCE_KEY) : sysConf);

563  @Override
564  public synchronized void getMetrics(MetricsCollector builder, boolean all) {
565    MetricsRecordBuilder rb = builder.addRecord(MS_NAME)
566        .addGauge(MsInfo.NumActiveSources, sources.size())
567        .addGauge(MsInfo.NumAllSources, allSources.size())
568        .addGauge(MsInfo.NumActiveSinks, sinks.size())
569        .addGauge(MsInfo.NumAllSinks, allSinks.size());
570
571    for (MetricsSinkAdapter sa : sinks.values()) {
572      sa.snapshot(rb, all);
573    }
574    registry.snapshot(rb, all);
575  }

584  @Override
585  public synchronized boolean shutdown() {
586    LOG.debug("refCount="+ refCount);
587    if (refCount <= 0) {
588      LOG.debug("Redundant shutdown", new Throwable());
589      return true; // already shutdown
590    }
591    if (--refCount > 0) return false;
592    if (monitoring) {
593      try { stop(); }
594      catch (Exception e) {
595        LOG.warn("Error stopping the metrics system", e);
596      }
597    }
598    allSources.clear();
599    allSinks.clear();
600    callbacks.clear();
601    namedCallbacks.clear();
602    if (mbeanName != null) {
603      MBeans.unregister(mbeanName);
604      mbeanName = null;
605    }
606    LOG.info(prefix +" metrics system shutdown complete.");
607    return true;
608  }

619  private InitMode initMode() {
620    LOG.debug("from system property: "+ System.getProperty(MS_INIT_MODE_KEY));
621    LOG.debug("from environment variable: "+ System.getenv(MS_INIT_MODE_KEY));
622    String m = System.getProperty(MS_INIT_MODE_KEY);
623    String m2 = m == null ? System.getenv(MS_INIT_MODE_KEY) : m;
624    return InitMode.valueOf(
625        StringUtils.toUpperCase((m2 == null ? InitMode.NORMAL.name() : m2)));
626  }
