228    public static void chmod(String path, int mode) throws IOException {
229      if (!Shell.WINDOWS) {
230        chmodImpl(path, mode);
231      } else {
232        try {
233          chmodImpl(path, mode);
234        } catch (NativeIOException nioe) {
235          if (nioe.getErrorCode() == 3) {
236            throw new NativeIOException("No such file or directory",
237                Errno.ENOENT);
238          } else {
239            LOG.warn(String.format("NativeIO.chmod error (%d): %s",
240                nioe.getErrorCode(), nioe.getMessage()));
241            throw new NativeIOException("Unknown error", Errno.UNKNOWN);
242          }
243        }
244      }

300    /**
301     * Locks the provided direct ByteBuffer into memory, preventing it from
302     * swapping out. After a buffer is locked, future accesses will not incur
303     * a page fault.
304     * 
305     * See the mlock(2) man page for more information.
306     * 
307     * @throws NativeIOException
308     */
309    static void mlock(ByteBuffer buffer, long len)
310        throws IOException {
311      assertCodeLoaded();
312      if (!buffer.isDirect()) {
313        throw new IOException("Cannot mlock a non-direct ByteBuffer");
314      }
315      mlock_native(buffer, len);
316    }

318    /**
319     * Unmaps the block from memory. See munmap(2).
320     *
321     * There isn't any portable way to unmap a memory region in Java.
322     * So we use the sun.nio method here.
323     * Note that unmapping a memory region could cause crashes if code
324     * continues to reference the unmapped code.  However, if we don't
325     * manually unmap the memory, we are dependent on the finalizer to
326     * do it, and we have no idea when the finalizer will run.
327     *
328     * @param buffer    The buffer to unmap.
329     */
330    public static void munmap(MappedByteBuffer buffer) {
331      if (buffer instanceof sun.nio.ch.DirectBuffer) {
332        sun.misc.Cleaner cleaner =
333            ((sun.nio.ch.DirectBuffer)buffer).cleaner();
334        cleaner.clean();
335      }
336    }

404    /**
405     * Returns the file stat for a file descriptor.
406     *
407     * @param fd file descriptor.
408     * @return the file descriptor file stat.
409     * @throws IOException thrown if there was an IO error while obtaining the file stat.
410     */
411    public static Stat getFstat(FileDescriptor fd) throws IOException {
412      Stat stat = null;
413      if (!Shell.WINDOWS) {
414        stat = fstat(fd); 
415        stat.owner = getName(IdCache.USER, stat.ownerId);
416        stat.group = getName(IdCache.GROUP, stat.groupId);
417      } else {
418        try {
419          stat = fstat(fd);
420        } catch (NativeIOException nioe) {
421          if (nioe.getErrorCode() == 6) {
422            throw new NativeIOException("The handle is invalid.",
423                Errno.EBADF);
424          } else {
425            LOG.warn(String.format("NativeIO.getFstat error (%d): %s",
426                nioe.getErrorCode(), nioe.getMessage()));
427            throw new NativeIOException("Unknown error", Errno.UNKNOWN);
428          }
429        }
430      }
431      return stat;
432    }

434    private static String getName(IdCache domain, int id) throws IOException {
435      Map<Integer, CachedName> idNameCache = (domain == IdCache.USER)
436        ? USER_ID_NAME_CACHE : GROUP_ID_NAME_CACHE;
437      String name;
438      CachedName cachedName = idNameCache.get(id);
439      long now = System.currentTimeMillis();
440      if (cachedName != null && (cachedName.timestamp + cacheTimeout) > now) {
441        name = cachedName.name;
442      } else {
443        name = (domain == IdCache.USER) ? getUserName(id) : getGroupName(id);
444        if (LOG.isDebugEnabled()) {
445          String type = (domain == IdCache.USER) ? "UserName" : "GroupName";
446          LOG.debug("Got " + type + " " + name + " for ID " + id +
447            " from the native implementation");
448        }
449        cachedName = new CachedName(name, now);
450        idNameCache.put(id, cachedName);
451      }
452      return name;
453    }

544     * {@link java.io.FileOutputStream}.  The file is opened for truncate or

559      return new FileOutputStream(createFileWithMode0(path.getAbsolutePath(),
560          desiredAccess, shareMode, creationDisposition, mode));

682  /**
683   * @return the operating system's page size.
684   */
685  static long getOperatingSystemPageSize() {
686    try {
687      Field f = Unsafe.class.getDeclaredField("theUnsafe");
688      f.setAccessible(true);
689      Unsafe unsafe = (Unsafe)f.get(null);
690      return unsafe.pageSize();
691    } catch (Throwable e) {
692      LOG.warn("Unable to get operating system page size.  Guessing 4096.", e);
693      return 4096;
694    }
695  }

697  private static class CachedUid {
698    final long timestamp;
699    final String username;
700    public CachedUid(String username, long timestamp) {
701      this.timestamp = timestamp;
702      this.username = username;
703    }
704  }
705  private static final Map<Long, CachedUid> uidCache =
706      new ConcurrentHashMap<Long, CachedUid>();
707  private static long cacheTimeout;
708  private static boolean initialized = false;

710  /**
711   * The Windows logon name has two part, NetBIOS domain name and
712   * user account name, of the format DOMAIN\UserName. This method
713   * will remove the domain part of the full logon name.
714   *
715   * @param Fthe full principal name containing the domain
716   * @return name with domain removed
717   */
718  private static String stripDomain(String name) {
719    int i = name.indexOf('\\');
720    if (i != -1)
721      name = name.substring(i + 1);
722    return name;
723  }

725  public static String getOwner(FileDescriptor fd) throws IOException {
726    ensureInitialized();
727    if (Shell.WINDOWS) {
728      String owner = Windows.getOwner(fd);
729      owner = stripDomain(owner);
730      return owner;
731    } else {
732      long uid = POSIX.getUIDforFDOwnerforOwner(fd);
733      CachedUid cUid = uidCache.get(uid);
734      long now = System.currentTimeMillis();
735      if (cUid != null && (cUid.timestamp + cacheTimeout) > now) {
736        return cUid.username;
737      }
738      String user = POSIX.getUserName(uid);
739      LOG.info("Got UserName " + user + " for UID " + uid
740          + " from the native implementation");
741      cUid = new CachedUid(user, now);
742      uidCache.put(uid, cUid);
743      return user;
744    }
745  }

747  /**
748   * Create a FileInputStream that shares delete permission on the
749   * file opened, i.e. other process can delete the file the
750   * FileInputStream is reading. Only Windows implementation uses
751   * the native interface.
752   */
753  public static FileInputStream getShareDeleteFileInputStream(File f)
754      throws IOException {
755    if (!Shell.WINDOWS) {
756      // On Linux the default FileInputStream shares delete permission
757      // on the file opened.
758      //
759      return new FileInputStream(f);
760    } else {
761      // Use Windows native interface to create a FileInputStream that
762      // shares delete permission on the file opened.
763      //
764      FileDescriptor fd = Windows.createFile(
765          f.getAbsolutePath(),
766          Windows.GENERIC_READ,
767          Windows.FILE_SHARE_READ |
768              Windows.FILE_SHARE_WRITE |
769              Windows.FILE_SHARE_DELETE,
770          Windows.OPEN_EXISTING);
771      return new FileInputStream(fd);
772    }
773  }

775  /**
776   * Create a FileInputStream that shares delete permission on the
777   * file opened at a given offset, i.e. other process can delete
778   * the file the FileInputStream is reading. Only Windows implementation
779   * uses the native interface.
780   */
781  public static FileInputStream getShareDeleteFileInputStream(File f, long seekOffset)
782      throws IOException {
783    if (!Shell.WINDOWS) {
784      RandomAccessFile rf = new RandomAccessFile(f, "r");
785      if (seekOffset > 0) {
786        rf.seek(seekOffset);
787      }
788      return new FileInputStream(rf.getFD());
789    } else {
790      // Use Windows native interface to create a FileInputStream that
791      // shares delete permission on the file opened, and set it to the
792      // given offset.
793      //
794      FileDescriptor fd = NativeIO.Windows.createFile(
795          f.getAbsolutePath(),
796          NativeIO.Windows.GENERIC_READ,
797          NativeIO.Windows.FILE_SHARE_READ |
798              NativeIO.Windows.FILE_SHARE_WRITE |
799              NativeIO.Windows.FILE_SHARE_DELETE,
800          NativeIO.Windows.OPEN_EXISTING);
801      if (seekOffset > 0)
802        NativeIO.Windows.setFilePointer(fd, seekOffset, NativeIO.Windows.FILE_BEGIN);
803      return new FileInputStream(fd);
804    }
805  }

807  /**
808   * Create the specified File for write access, ensuring that it does not exist.
809   * @param f the file that we want to create
810   * @param permissions we want to have on the file (if security is enabled)
811   *
812   * @throws AlreadyExistsException if the file already exists
813   * @throws IOException if any other error occurred
814   */
815  public static FileOutputStream getCreateForWriteFileOutputStream(File f, int permissions)
816      throws IOException {
817    if (!Shell.WINDOWS) {
818      // Use the native wrapper around open(2)
819      try {
820        FileDescriptor fd = NativeIO.POSIX.open(f.getAbsolutePath(),
821            NativeIO.POSIX.O_WRONLY | NativeIO.POSIX.O_CREAT
822                | NativeIO.POSIX.O_EXCL, permissions);
823        return new FileOutputStream(fd);
824      } catch (NativeIOException nioe) {
825        if (nioe.getErrno() == Errno.EEXIST) {
826          throw new AlreadyExistsException(nioe);
827        }
828        throw nioe;
829      }
830    } else {
831      // Use the Windows native APIs to create equivalent FileOutputStream
832      try {
833        FileDescriptor fd = NativeIO.Windows.createFile(f.getCanonicalPath(),
834            NativeIO.Windows.GENERIC_WRITE,
835            NativeIO.Windows.FILE_SHARE_DELETE
836                | NativeIO.Windows.FILE_SHARE_READ
837                | NativeIO.Windows.FILE_SHARE_WRITE,
838            NativeIO.Windows.CREATE_NEW);
839        NativeIO.POSIX.chmod(f.getCanonicalPath(), permissions);
840        return new FileOutputStream(fd);
841      } catch (NativeIOException nioe) {
842        if (nioe.getErrorCode() == 80) {
843          // ERROR_FILE_EXISTS
844          // 80 (0x50)
845          // The file exists
846          throw new AlreadyExistsException(nioe);
847        }
848        throw nioe;
849      }
850    }
851  }

906  /**
907   * Unbuffered file copy from src to dst without tainting OS buffer cache
908   *
909   * In POSIX platform:
910   * It uses FileChannel#transferTo() which internally attempts
911   * unbuffered IO on OS with native sendfile64() support and falls back to
912   * buffered IO otherwise.
913   *
914   * It minimizes the number of FileChannel#transferTo call by passing the the
915   * src file size directly instead of a smaller size as the 3rd parameter.
916   * This saves the number of sendfile64() system call when native sendfile64()
917   * is supported. In the two fall back cases where sendfile is not supported,
918   * FileChannle#transferTo already has its own batching of size 8 MB and 8 KB,
919   * respectively.
920   *
921   * In Windows Platform:
922   * It uses its own native wrapper of CopyFileEx with COPY_FILE_NO_BUFFERING
923   * flag, which is supported on Windows Server 2008 and above.
924   *
925   * Ideally, we should use FileChannel#transferTo() across both POSIX and Windows
926   * platform. Unfortunately, the wrapper(Java_sun_nio_ch_FileChannelImpl_transferTo0)
927   * used by FileChannel#transferTo for unbuffered IO is not implemented on Windows.
928   * Based on OpenJDK 6/7/8 source code, Java_sun_nio_ch_FileChannelImpl_transferTo0
929   * on Windows simply returns IOS_UNSUPPORTED.
930   *
931   * Note: This simple native wrapper does minimal parameter checking before copy and
932   * consistency check (e.g., size) after copy.
933   * It is recommended to use wrapper function like
934   * the Storage#nativeCopyFileUnbuffered() function in hadoop-hdfs with pre/post copy
935   * checks.
936   *
937   * @param src                  The source path
938   * @param dst                  The destination path
939   * @throws IOException
940   */
941  public static void copyFileUnbuffered(File src, File dst) throws IOException {
942    if (nativeLoaded && Shell.WINDOWS) {
943      copyFileUnbuffered0(src.getAbsolutePath(), dst.getAbsolutePath());
944    } else {
945      FileInputStream fis = null;
946      FileOutputStream fos = null;
947      FileChannel input = null;
948      FileChannel output = null;
949      try {
950        fis = new FileInputStream(src);
951        fos = new FileOutputStream(dst);
952        input = fis.getChannel();
953        output = fos.getChannel();
954        long remaining = input.size();
955        long position = 0;
956        long transferred = 0;
957        while (remaining > 0) {
958          transferred = input.transferTo(position, remaining, output);
959          remaining -= transferred;
960          position += transferred;
961        }
962      } finally {
963        IOUtils.cleanup(LOG, output);
964        IOUtils.cleanup(LOG, fos);
965        IOUtils.cleanup(LOG, input);
966        IOUtils.cleanup(LOG, fis);
967      }
968    }
969  }
970
971  private static native void copyFileUnbuffered0(String src, String dst)
972      throws NativeIOException;
