038/**
039 * This supports input and output streams for a socket channels. 
040 * These streams can have a timeout.
041 */
042abstract class SocketIOWithTimeout {
043  // This is intentionally package private.
044
045  static final Log LOG = LogFactory.getLog(SocketIOWithTimeout.class);    
046  
047  private SelectableChannel channel;
048  private long timeout;
049  private boolean closed = false;
050  
051  private static SelectorPool selector = new SelectorPool();
052  
053  /* A timeout value of 0 implies wait for ever. 
054   * We should have a value of timeout that implies zero wait.. i.e. 
055   * read or write returns immediately.
056   * 
057   * This will set channel to non-blocking.
058   */
059  SocketIOWithTimeout(SelectableChannel channel, long timeout) 
060                                                 throws IOException {
061    checkChannelValidity(channel);
062    
063    this.channel = channel;
064    this.timeout = timeout;
065    // Set non-blocking
066    channel.configureBlocking(false);
067  }
068  
069  void close() {
070    closed = true;
071  }
072
073  boolean isOpen() {
074    return !closed && channel.isOpen();
075  }
076
077  SelectableChannel getChannel() {
078    return channel;
079  }
080  
081  /** 
082   * Utility function to check if channel is ok.
083   * Mainly to throw IOException instead of runtime exception
084   * in case of mismatch. This mismatch can occur for many runtime
085   * reasons.
086   */
087  static void checkChannelValidity(Object channel) throws IOException {
088    if (channel == null) {
089      /* Most common reason is that original socket does not have a channel.
090       * So making this an IOException rather than a RuntimeException.
091       */
092      throw new IOException("Channel is null. Check " +
093                            "how the channel or socket is created.");
094    }
095    
096    if (!(channel instanceof SelectableChannel)) {
097      throw new IOException("Channel should be a SelectableChannel");
098    }    
099  }
100  
101  /**
102   * Performs actual IO operations. This is not expected to block.
103   *  
104   * @param buf
105   * @return number of bytes (or some equivalent). 0 implies underlying
106   *         channel is drained completely. We will wait if more IO is 
107   *         required.
108   * @throws IOException
109   */
110  abstract int performIO(ByteBuffer buf) throws IOException;  
111  
112  /**
113   * Performs one IO and returns number of bytes read or written.
114   * It waits up to the specified timeout. If the channel is 
115   * not read before the timeout, SocketTimeoutException is thrown.
116   * 
117   * @param buf buffer for IO
118   * @param ops Selection Ops used for waiting. Suggested values: 
119   *        SelectionKey.OP_READ while reading and SelectionKey.OP_WRITE while
120   *        writing. 
121   *        
122   * @return number of bytes read or written. negative implies end of stream.
123   * @throws IOException
124   */
125  int doIO(ByteBuffer buf, int ops) throws IOException {
126    
127    /* For now only one thread is allowed. If user want to read or write
128     * from multiple threads, multiple streams could be created. In that
129     * case multiple threads work as well as underlying channel supports it.
130     */
131    if (!buf.hasRemaining()) {
132      throw new IllegalArgumentException("Buffer has no data left.");
133      //or should we just return 0?
134    }
135
136    while (buf.hasRemaining()) {
137      if (closed) {
138        return -1;
139      }
140
141      try {
142        int n = performIO(buf);
143        if (n != 0) {
144          // successful io or an error.
145          return n;
146        }
147      } catch (IOException e) {
148        if (!channel.isOpen()) {
149          closed = true;
150        }
151        throw e;
152      }
153
154      //now wait for socket to be ready.
155      int count = 0;
156      try {
157        count = selector.select(channel, ops, timeout);  
158      } catch (IOException e) { //unexpected IOException.
159        closed = true;
160        throw e;
161      } 
162
163      if (count == 0) {
164        throw new SocketTimeoutException(timeoutExceptionString(channel,
165                                                                timeout, ops));
166      }
167      // otherwise the socket should be ready for io.
168    }
169    
170    return 0; // does not reach here.
171  }
172  
173  /**
174   * The contract is similar to {@link SocketChannel#connect(SocketAddress)} 
175   * with a timeout.
176   * 
177   * @see SocketChannel#connect(SocketAddress)
178   * 
179   * @param channel - this should be a {@link SelectableChannel}
180   * @param endpoint
181   * @throws IOException
182   */
183  static void connect(SocketChannel channel, 
184                      SocketAddress endpoint, int timeout) throws IOException {
185    
186    boolean blockingOn = channel.isBlocking();
187    if (blockingOn) {
188      channel.configureBlocking(false);
189    }
190    
191    try { 
192      if (channel.connect(endpoint)) {
193        return;
194      }
195
196      long timeoutLeft = timeout;
197      long endTime = (timeout > 0) ? (Time.now() + timeout): 0;
198      
199      while (true) {
200        // we might have to call finishConnect() more than once
201        // for some channels (with user level protocols)
202        
203        int ret = selector.select((SelectableChannel)channel, 
204                                  SelectionKey.OP_CONNECT, timeoutLeft);
205        
206        if (ret > 0 && channel.finishConnect()) {
207          return;
208        }
209        
210        if (ret == 0 ||
211            (timeout > 0 &&  
212              (timeoutLeft = (endTime - Time.now())) <= 0)) {
213          throw new SocketTimeoutException(
214                    timeoutExceptionString(channel, timeout, 
215                                           SelectionKey.OP_CONNECT));
216        }
217      }
218    } catch (IOException e) {
219      // javadoc for SocketChannel.connect() says channel should be closed.
220      try {
221        channel.close();
222      } catch (IOException ignored) {}
223      throw e;
224    } finally {
225      if (blockingOn && channel.isOpen()) {
226        channel.configureBlocking(true);
227      }
228    }
229  }
230
231  /**
232   * This is similar to {@link #doIO(ByteBuffer, int)} except that it
233   * does not perform any I/O. It just waits for the channel to be ready
234   * for I/O as specified in ops.
235   * 
236   * @param ops Selection Ops used for waiting
237   * 
238   * @throws SocketTimeoutException 
239   *         if select on the channel times out.
240   * @throws IOException
241   *         if any other I/O error occurs. 
242   */
243  void waitForIO(int ops) throws IOException {
244    
245    if (selector.select(channel, ops, timeout) == 0) {
246      throw new SocketTimeoutException(timeoutExceptionString(channel, timeout,
247                                                              ops)); 
248    }
249  }
250
251  public void setTimeout(long timeoutMs) {
252    this.timeout = timeoutMs;
253  }
254    
255  private static String timeoutExceptionString(SelectableChannel channel,
256                                               long timeout, int ops) {
257    
258    String waitingFor;
259    switch(ops) {
260    
261    case SelectionKey.OP_READ :
262      waitingFor = "read"; break;
263      
264    case SelectionKey.OP_WRITE :
265      waitingFor = "write"; break;      
266      
267    case SelectionKey.OP_CONNECT :
268      waitingFor = "connect"; break;
269      
270    default :
271      waitingFor = "" + ops;  
272    }
273    
274    return timeout + " millis timeout while " +
275           "waiting for channel to be ready for " + 
276           waitingFor + ". ch : " + channel;    
277  }

279  /**
280   * This maintains a pool of selectors. These selectors are closed
281   * once they are idle (unused) for a few seconds.
282   */
283  private static class SelectorPool {
284    
285    private static class SelectorInfo {
286      Selector              selector;
287      long                  lastActivityTime;
288      LinkedList<SelectorInfo> queue; 
289      
290      void close() {
291        if (selector != null) {
292          try {
293            selector.close();
294          } catch (IOException e) {
295            LOG.warn("Unexpected exception while closing selector : ", e);
296          }
297        }
298      }    
299    }
300    
301    private static class ProviderInfo {
302      SelectorProvider provider;
303      LinkedList<SelectorInfo> queue; // lifo
304      ProviderInfo next;
305    }
306    
307    private static final long IDLE_TIMEOUT = 10 * 1000; // 10 seconds.
308    
309    private ProviderInfo providerList = null;
310    
311    /**
312     * Waits on the channel with the given timeout using one of the 
313     * cached selectors. It also removes any cached selectors that are
314     * idle for a few seconds.
315     * 
316     * @param channel
317     * @param ops
318     * @param timeout
319     * @return
320     * @throws IOException
321     */
322    int select(SelectableChannel channel, int ops, long timeout) 
323                                                   throws IOException {
324     
325      SelectorInfo info = get(channel);
326      
327      SelectionKey key = null;
328      int ret = 0;
329      
330      try {
331        while (true) {
332          long start = (timeout == 0) ? 0 : Time.now();
333
334          key = channel.register(info.selector, ops);
335          ret = info.selector.select(timeout);
336          
337          if (ret != 0) {
338            return ret;
339          }
340          
341          if (Thread.currentThread().isInterrupted()) {
342            throw new InterruptedIOException("Interrupted while waiting for "
343                + "IO on channel " + channel + ". " + timeout
344                + " millis timeout left.");
345          }
346
347          /* Sometimes select() returns 0 much before timeout for 
348           * unknown reasons. So select again if required.
349           */
350          if (timeout > 0) {
351            timeout -= Time.now() - start;
352            if (timeout <= 0) {
353              return 0;
354            }
355          }
356          
357        }
358      } finally {
359        if (key != null) {
360          key.cancel();
361        }
362        
363        //clear the canceled key.
364        try {
365          info.selector.selectNow();
366        } catch (IOException e) {
367          LOG.info("Unexpected Exception while clearing selector : ", e);
368          // don't put the selector back.
369          info.close();
370          return ret; 
371        }
372        
373        release(info);
374      }
375    }
376    
377    /**
378     * Takes one selector from end of LRU list of free selectors.
379     * If there are no selectors awailable, it creates a new selector.
380     * Also invokes trimIdleSelectors(). 
381     * 
382     * @param channel
383     * @return 
384     * @throws IOException
385     */
386    private synchronized SelectorInfo get(SelectableChannel channel) 
387                                                         throws IOException {
388      SelectorInfo selInfo = null;
389      
390      SelectorProvider provider = channel.provider();
391      
392      // pick the list : rarely there is more than one provider in use.
393      ProviderInfo pList = providerList;
394      while (pList != null && pList.provider != provider) {
395        pList = pList.next;
396      }      
397      if (pList == null) {
398        //LOG.info("Creating new ProviderInfo : " + provider.toString());
399        pList = new ProviderInfo();
400        pList.provider = provider;
401        pList.queue = new LinkedList<SelectorInfo>();
402        pList.next = providerList;
403        providerList = pList;
404      }
405      
406      LinkedList<SelectorInfo> queue = pList.queue;
407      
408      if (queue.isEmpty()) {
409        Selector selector = provider.openSelector();
410        selInfo = new SelectorInfo();
411        selInfo.selector = selector;
412        selInfo.queue = queue;
413      } else {
414        selInfo = queue.removeLast();
415      }
416      
417      trimIdleSelectors(Time.now());
418      return selInfo;
419    }
420    
421    /**
422     * puts selector back at the end of LRU list of free selectos.
423     * Also invokes trimIdleSelectors().
424     * 
425     * @param info
426     */
427    private synchronized void release(SelectorInfo info) {
428      long now = Time.now();
429      trimIdleSelectors(now);
430      info.lastActivityTime = now;
431      info.queue.addLast(info);
432    }
433    
434    /**
435     * Closes selectors that are idle for IDLE_TIMEOUT (10 sec). It does not
436     * traverse the whole list, just over the one that have crossed 
437     * the timeout.
438     */
439    private void trimIdleSelectors(long now) {
440      long cutoff = now - IDLE_TIMEOUT;
441      
442      for(ProviderInfo pList=providerList; pList != null; pList=pList.next) {
443        if (pList.queue.isEmpty()) {
444          continue;
445        }
446        for(Iterator<SelectorInfo> it = pList.queue.iterator(); it.hasNext();) {
447          SelectorInfo info = it.next();
448          if (info.lastActivityTime > cutoff) {
449            break;
450          }
451          it.remove();
452          info.close();
453        }
454      }
455    }
456  }
