080  /**
081   * This function is designed to validate that the path chosen for a UNIX
082   * domain socket is secure.  A socket path is secure if it doesn't allow
083   * unprivileged users to perform a man-in-the-middle attack against it.
084   * For example, one way to perform a man-in-the-middle attack would be for
085   * a malicious user to move the server socket out of the way and create his
086   * own socket in the same place.  Not good.
087   * 
088   * Note that we only check the path once.  It's possible that the
089   * permissions on the path could change, perhaps to something more relaxed,
090   * immediately after the path passes our validation test-- hence creating a
091   * security hole.  However, the purpose of this check is to spot common
092   * misconfigurations.  System administrators do not commonly change
093   * permissions on these paths while the server is running.
094   *
095   * @param path             the path to validate
096   * @param skipComponents   the number of starting path components to skip 
097   *                         validation for (used only for testing)
098   */
099  @VisibleForTesting
100  native static void validateSocketPathSecurity0(String path,
101      int skipComponents) throws IOException;

332  /**
333   * Close the Socket.
334   */
335  @Override
336  public void close() throws IOException {
337    // Set the closed bit on this DomainSocket
338    int count;
339    try {
340      count = refCount.setClosed();
341    } catch (ClosedChannelException e) {
342      // Someone else already closed the DomainSocket.
343      return;
344    }
345    // Wait for all references to go away
346    boolean didShutdown = false;
347    boolean interrupted = false;
348    while (count > 0) {
349      if (!didShutdown) {
350        try {
351          // Calling shutdown on the socket will interrupt blocking system
352          // calls like accept, write, and read that are going on in a
353          // different thread.
354          shutdown0(fd);
355        } catch (IOException e) {
356          LOG.error("shutdown error: ", e);
357        }
358        didShutdown = true;
359      }
360      try {
361        Thread.sleep(10);
362      } catch (InterruptedException e) {
363        interrupted = true;
364      }
365      count = refCount.getReferenceCount();
366    }
367
368    // At this point, nobody has a reference to the file descriptor, 
369    // and nobody will be able to get one in the future either.
370    // We now call close(2) on the file descriptor.
371    // After this point, the file descriptor number will be reused by 
372    // something else.  Although this DomainSocket object continues to hold 
373    // the old file descriptor number (it's a final field), we never use it 
374    // again because this DomainSocket is closed.
375    close0(fd);
376    if (interrupted) {
377      Thread.currentThread().interrupt();
378    }
379  }

427  /**
428   * Receive some FileDescriptor objects from the process on the other side of
429   * this socket.
430   *
431   * @param descriptors       (output parameter) Array of FileDescriptors.
432   *                          We will fill as many slots as possible with file
433   *                          descriptors passed from the remote process.  The
434   *                          other slots will contain NULL.
435   * @param jbuf              (output parameter) Buffer to read into.
436   *                          The UNIX domain sockets API requires you to read
437   *                          at least one byte from the remote process, even
438   *                          if all you care about is the file descriptors
439   *                          you will receive.
440   * @param offset            Offset into the byte buffer to load data
441   * @param length            Length of the byte buffer to use for data
442   *
443   * @return                  The number of bytes read.  This will be -1 if we
444   *                          reached EOF (similar to SocketInputStream);
445   *                          otherwise, it will be positive.
446   * @throws                  IOException if there was an I/O error.
447   */
448  public int receiveFileDescriptors(FileDescriptor[] descriptors,
449      byte jbuf[], int offset, int length) throws IOException {
450    refCount.reference();
451    boolean exc = true;
452    try {
453      int nBytes = receiveFileDescriptors0(fd, descriptors, jbuf, offset, length);
454      exc = false;
455      return nBytes;
456    } finally {
457      unreference(exc);
458    }

461  /**
462   * Receive some FileDescriptor objects from the process on the other side of
463   * this socket, and wrap them in FileInputStream objects.
464   *
465   * See {@link DomainSocket#recvFileInputStreams(ByteBuffer)}
466   */
467  public int recvFileInputStreams(FileInputStream[] streams, byte buf[],
468        int offset, int length) throws IOException {
469    FileDescriptor descriptors[] = new FileDescriptor[streams.length];
470    boolean success = false;
471    for (int i = 0; i < streams.length; i++) {
472      streams[i] = null;
473    }
474    refCount.reference();
475    try {
476      int ret = receiveFileDescriptors0(fd, descriptors, buf, offset, length);
477      for (int i = 0, j = 0; i < descriptors.length; i++) {
478        if (descriptors[i] != null) {
479          streams[j++] = new FileInputStream(descriptors[i]);
480          descriptors[i] = null;
481        }
482      }
483      success = true;
484      return ret;
485    } finally {
486      if (!success) {
487        for (int i = 0; i < descriptors.length; i++) {
488          if (descriptors[i] != null) {
489            try {
490              closeFileDescriptor0(descriptors[i]);
491            } catch (Throwable t) {
492              LOG.warn(t);
493            }
494          } else if (streams[i] != null) {
495            try {
496              streams[i].close();
497            } catch (Throwable t) {
498              LOG.warn(t);
499            } finally {
500              streams[i] = null; }
501          }
502        }
503      }
504      unreference(!success);
505    }
506  }

609  @InterfaceAudience.LimitedPrivate("HDFS")
610  public class DomainChannel implements ReadableByteChannel {
611    @Override
612    public boolean isOpen() {
613      return DomainSocket.this.isOpen();
614    }
615
616    @Override
617    public void close() throws IOException {
618      DomainSocket.this.close();
619    }
620
621    @Override
622    public int read(ByteBuffer dst) throws IOException {
623      refCount.reference();
624      boolean exc = true;
625      try {
626        int nread = 0;
627        if (dst.isDirect()) {
628          nread = DomainSocket.readByteBufferDirect0(DomainSocket.this.fd,
629              dst, dst.position(), dst.remaining());
630        } else if (dst.hasArray()) {
631          nread = DomainSocket.readArray0(DomainSocket.this.fd,
632              dst.array(), dst.position() + dst.arrayOffset(),
633              dst.remaining());
634        } else {
635          throw new AssertionError("we don't support " +
636              "using ByteBuffers that aren't either direct or backed by " +
637              "arrays");
638        }
639        if (nread > 0) {
640          dst.position(dst.position() + nread);
641        }
642        exc = false;
643        return nread;
644      } finally {
645        unreference(exc);
646      }
647    }
648  }
