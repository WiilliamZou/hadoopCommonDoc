032  public static final int INVALID_CALL_ID = -2;

038  public static final int INVALID_RETRY_COUNT = -1;
039  
040  /**
041   * The first four bytes of Hadoop RPC connections
042   */

044      (Charsets.UTF_8));
045  
046  // 1 : Introduce ping and server does not throw away RPCs
047  // 3 : Introduce the protocol into the RPC connection header
048  // 4 : Introduced SASL security layer
049  // 5 : Introduced use of {@link ArrayPrimitiveWritable$Internal}
050  //     in ObjectWritable to efficiently transmit arrays of primitives
051  // 6 : Made RPC Request header explicit
052  // 7 : Changed Ipc Connection Header to use Protocol buffers
053  // 8 : SASL server always sends a final response
054  // 9 : Changes to protocol for HADOOP-8990
055  public static final byte CURRENT_VERSION = 9;
056}
