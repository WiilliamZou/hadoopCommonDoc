152  private static class ClientExecutorServiceFactory {
153    private int executorRefCount = 0;
154    private ExecutorService clientExecutor = null;
155    
156    /**
157     * Get Executor on which IPC calls' parameters are sent.
158     * If the internal reference counter is zero, this method
159     * creates the instance of Executor. If not, this method
160     * just returns the reference of clientExecutor.
161     * 
162     * @return An ExecutorService instance
163     */
164    synchronized ExecutorService refAndGetInstance() {
165      if (executorRefCount == 0) {
166        clientExecutor = Executors.newCachedThreadPool(
167            new ThreadFactoryBuilder()
168            .setDaemon(true)
169            .setNameFormat("IPC Parameter Sending Thread #%d")
170            .build());
171      }
172      executorRefCount++;
173      
174      return clientExecutor;
175    }
176    
177    /**
178     * Cleanup Executor on which IPC calls' parameters are sent.
179     * If reference counter is zero, this method discards the
180     * instance of the Executor. If not, this method
181     * just decrements the internal reference counter.
182     * 
183     * @return An ExecutorService instance if it exists.
184     *   Null is returned if not.
185     */
186    synchronized ExecutorService unrefAndCleanup() {
187      executorRefCount--;
188      assert(executorRefCount >= 0);
189      
190      if (executorRefCount == 0) {
191        clientExecutor.shutdown();
192        try {
193          if (!clientExecutor.awaitTermination(1, TimeUnit.MINUTES)) {
194            clientExecutor.shutdownNow();
195          }
196        } catch (InterruptedException e) {
197          LOG.error("Interrupted while waiting for clientExecutor" +
198              "to stop", e);
199          clientExecutor.shutdownNow();
200        }
201        clientExecutor = null;
202      }
203      
204      return clientExecutor;
205    }
206  };

281  /** Check the rpc response header. */
282  void checkResponse(RpcResponseHeaderProto header) throws IOException {
283    if (header == null) {
284      throw new EOFException("Response is null.");
285    }
286    if (header.hasClientId()) {
287      // check client IDs
288      final byte[] id = header.getClientId().toByteArray();
289      if (!Arrays.equals(id, RpcConstants.DUMMY_CLIENT_ID)) {
290        if (!Arrays.equals(id, clientId)) {
291          throw new IOException("Client IDs not matched: local ID="
292              + StringUtils.byteToHexString(clientId) + ", ID in response="
293              + StringUtils.byteToHexString(header.getClientId().toByteArray()));
294        }
295      }
296    }
297  }

303  /** 
304   * Class that represents an RPC call
305   */
306  static class Call {
307    final int id;               // call id
308    final int retry;           // retry count
309    final Writable rpcRequest;  // the serialized rpc request
310    Writable rpcResponse;       // null if rpc has error
311    IOException error;          // exception, null if success
312    final RPC.RpcKind rpcKind;      // Rpc EngineKind
313    boolean done;               // true when call is done
314
315    private Call(RPC.RpcKind rpcKind, Writable param) {
316      this.rpcKind = rpcKind;
317      this.rpcRequest = param;
318
319      final Integer id = callId.get();
320      if (id == null) {
321        this.id = nextCallId();
322      } else {
323        callId.set(null);
324        this.id = id;
325      }
326      
327      final Integer rc = retryCount.get();
328      if (rc == null) {
329        this.retry = 0;
330      } else {
331        this.retry = rc;
332      }
333    }
334
335    /** Indicate when the call is complete and the
336     * value or error are available.  Notifies by default.  */
337    protected synchronized void callComplete() {
338      this.done = true;
339      notify();                                 // notify caller
340    }
341
342    /** Set the exception when there is an error.
343     * Notify the caller the call is done.
344     * 
345     * @param error exception thrown by the call; either local or remote
346     */
347    public synchronized void setException(IOException error) {
348      this.error = error;
349      callComplete();
350    }
351    
352    /** Set the return value when there is no error. 
353     * Notify the caller the call is done.
354     * 
355     * @param rpcResponse return value of the rpc call.
356     */
357    public synchronized void setRpcResponse(Writable rpcResponse) {
358      this.rpcResponse = rpcResponse;
359      callComplete();
360    }
361    
362    public synchronized Writable getRpcResponse() {
363      return rpcResponse;
364    }

400    public Connection(ConnectionId remoteId, int serviceClass) throws IOException {
401      this.remoteId = remoteId;
402      this.server = remoteId.getAddress();
403      if (server.isUnresolved()) {
404        throw NetUtils.wrapException(server.getHostName(),
405            server.getPort(),
406            null,
407            0,
408            new UnknownHostException());
409      }
410      this.rpcTimeout = remoteId.getRpcTimeout();
411      this.maxIdleTime = remoteId.getMaxIdleTime();
412      this.connectionRetryPolicy = remoteId.connectionRetryPolicy;
413      this.maxRetriesOnSasl = remoteId.getMaxRetriesOnSasl();
414      this.maxRetriesOnSocketTimeouts = remoteId.getMaxRetriesOnSocketTimeouts();
415      this.tcpNoDelay = remoteId.getTcpNoDelay();
416      this.doPing = remoteId.getDoPing();
417      if (doPing) {
418        // construct a RPC header with the callId as the ping callId
419        pingRequest = new ByteArrayOutputStream();
420        RpcRequestHeaderProto pingHeader = ProtoUtil
421            .makeRpcRequestHeader(RpcKind.RPC_PROTOCOL_BUFFER,
422                OperationProto.RPC_FINAL_PACKET, PING_CALL_ID,
423                RpcConstants.INVALID_RETRY_COUNT, clientId);
424        pingHeader.writeDelimitedTo(pingRequest);
425      }
426      this.pingInterval = remoteId.getPingInterval();
427      this.serviceClass = serviceClass;
428      if (LOG.isDebugEnabled()) {
429        LOG.debug("The ping interval is " + this.pingInterval + " ms.");
430      }
431
432      UserGroupInformation ticket = remoteId.getTicket();
433      // try SASL if security is enabled or if the ugi contains tokens.
434      // this causes a SIMPLE client with tokens to attempt SASL
435      boolean trySasl = UserGroupInformation.isSecurityEnabled() ||
436                        (ticket != null && !ticket.getTokens().isEmpty());
437      this.authProtocol = trySasl ? AuthProtocol.SASL : AuthProtocol.NONE;
438      
439      this.setName("IPC Client (" + socketFactory.hashCode() +") connection to " +
440          server.toString() +
441          " from " + ((ticket==null)?"an unknown user":ticket.getUserName()));
442      this.setDaemon(true);
443    }

465    /** This class sends a ping to the remote side when timeout on
466     * reading. If no failure is detected, it retries until at least
467     * a byte is read.
468     */
469    private class PingInputStream extends FilterInputStream {
470      /* constructor */
471      protected PingInputStream(InputStream in) {
472        super(in);
473      }
474
475      /* Process timeout exception
476       * if the connection is not going to be closed or 
477       * is not configured to have a RPC timeout, send a ping.
478       * (if rpcTimeout is not set to be 0, then RPC should timeout.
479       * otherwise, throw the timeout exception.
480       */
481      private void handleTimeout(SocketTimeoutException e) throws IOException {
482        if (shouldCloseConnection.get() || !running.get() || rpcTimeout > 0) {
483          throw e;
484        } else {
485          sendPing();
486        }
487      }
488      
489      /** Read a byte from the stream.
490       * Send a ping if timeout on read. Retries if no failure is detected
491       * until a byte is read.
492       * @throws IOException for any IO problem other than socket timeout
493       */
494      @Override
495      public int read() throws IOException {
496        do {
497          try {
498            return super.read();
499          } catch (SocketTimeoutException e) {
500            handleTimeout(e);
501          }
502        } while (true);
503      }
504
505      /** Read bytes into a buffer starting from offset <code>off</code>
506       * Send a ping if timeout on read. Retries if no failure is detected
507       * until a byte is read.
508       * 
509       * @return the total number of bytes read; -1 if the connection is closed.
510       */
511      @Override
512      public int read(byte[] buf, int off, int len) throws IOException {
513        do {
514          try {
515            return super.read(buf, off, len);
516          } catch (SocketTimeoutException e) {
517            handleTimeout(e);
518          }
519        } while (true);
520      }
521    }

533    private synchronized boolean shouldAuthenticateOverKrb() throws IOException {
534      UserGroupInformation loginUser = UserGroupInformation.getLoginUser();
535      UserGroupInformation currentUser = UserGroupInformation.getCurrentUser();
536      UserGroupInformation realUser = currentUser.getRealUser();
537      if (authMethod == AuthMethod.KERBEROS && loginUser != null &&
538      // Make sure user logged in using Kerberos either keytab or TGT
539          loginUser.hasKerberosCredentials() &&
540          // relogin only in case it is the login user (e.g. JT)
541          // or superuser (like oozie).
542          (loginUser.equals(currentUser) || loginUser.equals(realUser))) {
543        return true;
544      }
545      return false;
546    }

558    /**
559     * Update the server address if the address corresponding to the host
560     * name has changed.
561     *
562     * @return true if an addr change was detected.
563     * @throws IOException when the hostname cannot be resolved.
564     */
565    private synchronized boolean updateAddress() throws IOException {
566      // Do a fresh lookup with the old host name.
567      InetSocketAddress currentAddr = NetUtils.createSocketAddrForHost(
568                               server.getHostName(), server.getPort());
569
570      if (!server.equals(currentAddr)) {
571        LOG.warn("Address change detected. Old: " + server.toString() +
572                                 " New: " + currentAddr.toString());
573        server = currentAddr;
574        return true;
575      }
576      return false;
577    }

579    private synchronized void setupConnection() throws IOException {
580      short ioFailures = 0;
581      short timeoutFailures = 0;
582      while (true) {
583        try {
584          this.socket = socketFactory.createSocket();
585          this.socket.setTcpNoDelay(tcpNoDelay);
586          this.socket.setKeepAlive(true);
587          
588          /*
589           * Bind the socket to the host specified in the principal name of the
590           * client, to ensure Server matching address of the client connection
591           * to host name in principal passed.
592           */
593          UserGroupInformation ticket = remoteId.getTicket();
594          if (ticket != null && ticket.hasKerberosCredentials()) {
595            KerberosInfo krbInfo = 
596              remoteId.getProtocol().getAnnotation(KerberosInfo.class);
597            if (krbInfo != null && krbInfo.clientPrincipal() != null) {
598              String host = 
599                SecurityUtil.getHostFromPrincipal(remoteId.getTicket().getUserName());
600              
601              // If host name is a valid local address then bind socket to it
602              InetAddress localAddr = NetUtils.getLocalInetAddress(host);
603              if (localAddr != null) {
604                this.socket.bind(new InetSocketAddress(localAddr, 0));
605              }
606            }
607          }
608          
609          NetUtils.connect(this.socket, server, connectionTimeout);
610          if (rpcTimeout > 0) {
611            pingInterval = rpcTimeout;  // rpcTimeout overwrites pingInterval
612          }
613          this.socket.setSoTimeout(pingInterval);
614          return;
615        } catch (ConnectTimeoutException toe) {
616          /* Check for an address change and update the local reference.
617           * Reset the failure counter if the address was changed
618           */
619          if (updateAddress()) {
620            timeoutFailures = ioFailures = 0;
621          }
622          handleConnectionTimeout(timeoutFailures++,
623              maxRetriesOnSocketTimeouts, toe);
624        } catch (IOException ie) {
625          if (updateAddress()) {
626            timeoutFailures = ioFailures = 0;
627          }
628          handleConnectionFailure(ioFailures++, ie);
629        }
630      }
631    }

633    /**
634     * If multiple clients with the same principal try to connect to the same
635     * server at the same time, the server assumes a replay attack is in
636     * progress. This is a feature of kerberos. In order to work around this,
637     * what is done is that the client backs off randomly and tries to initiate
638     * the connection again. The other problem is to do with ticket expiry. To
639     * handle that, a relogin is attempted.
640     */
641    private synchronized void handleSaslConnectionFailure(
642        final int currRetries, final int maxRetries, final Exception ex,
643        final Random rand, final UserGroupInformation ugi) throws IOException,
644        InterruptedException {
645      ugi.doAs(new PrivilegedExceptionAction<Object>() {
646        @Override
647        public Object run() throws IOException, InterruptedException {
648          final short MAX_BACKOFF = 5000;
649          closeConnection();
650          disposeSasl();
651          if (shouldAuthenticateOverKrb()) {
652            if (currRetries < maxRetries) {
653              if(LOG.isDebugEnabled()) {
654                LOG.debug("Exception encountered while connecting to "
655                    + "the server : " + ex);
656              }
657              // try re-login
658              if (UserGroupInformation.isLoginKeytabBased()) {
659                UserGroupInformation.getLoginUser().reloginFromKeytab();
660              } else if (UserGroupInformation.isLoginTicketBased()) {
661                UserGroupInformation.getLoginUser().reloginFromTicketCache();
662              }
663              // have granularity of milliseconds
664              //we are sleeping with the Connection lock held but since this
665              //connection instance is being used for connecting to the server
666              //in question, it is okay
667              Thread.sleep((rand.nextInt(MAX_BACKOFF) + 1));
668              return null;
669            } else {
670              String msg = "Couldn't setup connection for "
671                  + UserGroupInformation.getLoginUser().getUserName() + " to "
672                  + remoteId;
673              LOG.warn(msg, ex);
674              throw (IOException) new IOException(msg).initCause(ex);
675            }
676          } else {
677            LOG.warn("Exception encountered while connecting to "
678                + "the server : " + ex);
679          }
680          if (ex instanceof RemoteException)
681            throw (RemoteException) ex;
682          throw new IOException(ex);
683        }
684      });
685    }

688    /** Connect to the server and set up the I/O streams. It then sends
689     * a header to the server and starts
690     * the connection thread that waits for responses.
691     */
692    private synchronized void setupIOstreams(
693        AtomicBoolean fallbackToSimpleAuth) {
694      if (socket != null || shouldCloseConnection.get()) {
695        return;
696      } 
697      try {
698        if (LOG.isDebugEnabled()) {
699          LOG.debug("Connecting to "+server);
700        }
701        if (Trace.isTracing()) {
702          Trace.addTimelineAnnotation("IPC client connecting to " + server);
703        }
704        short numRetries = 0;
705        Random rand = null;
706        while (true) {
707          setupConnection();
708          InputStream inStream = NetUtils.getInputStream(socket);
709          OutputStream outStream = NetUtils.getOutputStream(socket);
710          writeConnectionHeader(outStream);
711          if (authProtocol == AuthProtocol.SASL) {
712            final InputStream in2 = inStream;
713            final OutputStream out2 = outStream;
714            UserGroupInformation ticket = remoteId.getTicket();
715            if (ticket.getRealUser() != null) {
716              ticket = ticket.getRealUser();
717            }
718            try {
719              authMethod = ticket
720                  .doAs(new PrivilegedExceptionAction<AuthMethod>() {
721                    @Override
722                    public AuthMethod run()
723                        throws IOException, InterruptedException {
724                      return setupSaslConnection(in2, out2);
725                    }
726                  });
727            } catch (Exception ex) {
728              authMethod = saslRpcClient.getAuthMethod();
729              if (rand == null) {
730                rand = new Random();
731              }
732              handleSaslConnectionFailure(numRetries++, maxRetriesOnSasl, ex,
733                  rand, ticket);
734              continue;
735            }
736            if (authMethod != AuthMethod.SIMPLE) {
737              // Sasl connect is successful. Let's set up Sasl i/o streams.
738              inStream = saslRpcClient.getInputStream(inStream);
739              outStream = saslRpcClient.getOutputStream(outStream);
740              // for testing
741              remoteId.saslQop =
742                  (String)saslRpcClient.getNegotiatedProperty(Sasl.QOP);
743              LOG.debug("Negotiated QOP is :" + remoteId.saslQop);
744              if (fallbackToSimpleAuth != null) {
745                fallbackToSimpleAuth.set(false);
746              }
747            } else if (UserGroupInformation.isSecurityEnabled()) {
748              if (!fallbackAllowed) {
749                throw new IOException("Server asks us to fall back to SIMPLE " +
750                    "auth, but this client is configured to only allow secure " +
751                    "connections.");
752              }
753              if (fallbackToSimpleAuth != null) {
754                fallbackToSimpleAuth.set(true);
755              }
756            }
757          }
758        
759          if (doPing) {
760            inStream = new PingInputStream(inStream);
761          }
762          this.in = new DataInputStream(new BufferedInputStream(inStream));
763
764          // SASL may have already buffered the stream
765          if (!(outStream instanceof BufferedOutputStream)) {
766            outStream = new BufferedOutputStream(outStream);
767          }
768          this.out = new DataOutputStream(outStream);
769          
770          writeConnectionContext(remoteId, authMethod);
771
772          // update last activity time
773          touch();
774
775          if (Trace.isTracing()) {
776            Trace.addTimelineAnnotation("IPC client connected to " + server);
777          }
778
779          // start the receiver thread after the socket connection has been set
780          // up
781          start();
782          return;
783        }
784      } catch (Throwable t) {
785        if (t instanceof IOException) {
786          markClosed((IOException)t);
787        } else {
788          markClosed(new IOException("Couldn't set up IO streams", t));
789        }
790        close();
791      }
792    }

794    private void closeConnection() {
795      if (socket == null) {
796        return;
797      }
798      // close the current connection
799      try {
800        socket.close();
801      } catch (IOException e) {
802        LOG.warn("Not able to close a socket", e);
803      }
804      // set socket to null so that the next call to setupIOstreams
805      // can start the process of connect all over again.
806      socket = null;
807    }

809    /* Handle connection failures due to timeout on connect
810     *
811     * If the current number of retries is equal to the max number of retries,
812     * stop retrying and throw the exception; Otherwise backoff 1 second and
813     * try connecting again.
814     *
815     * This Method is only called from inside setupIOstreams(), which is
816     * synchronized. Hence the sleep is synchronized; the locks will be retained.
817     *
818     * @param curRetries current number of retries
819     * @param maxRetries max number of retries allowed
820     * @param ioe failure reason
821     * @throws IOException if max number of retries is reached
822     */
823    private void handleConnectionTimeout(
824        int curRetries, int maxRetries, IOException ioe) throws IOException {
825
826      closeConnection();
827
828      // throw the exception if the maximum number of retries is reached
829      if (curRetries >= maxRetries) {
830        throw ioe;
831      }
832      LOG.info("Retrying connect to server: " + server + ". Already tried "
833          + curRetries + " time(s); maxRetries=" + maxRetries);
834    }
835
836    private void handleConnectionFailure(int curRetries, IOException ioe
837        ) throws IOException {
838      closeConnection();
839
840      final RetryAction action;
841      try {
842        action = connectionRetryPolicy.shouldRetry(ioe, curRetries, 0, true);
843      } catch(Exception e) {
844        throw e instanceof IOException? (IOException)e: new IOException(e);
845      }
846      if (action.action == RetryAction.RetryDecision.FAIL) {
847        if (action.reason != null) {
848          LOG.warn("Failed to connect to server: " + server + ": "
849              + action.reason, ioe);
850        }
851        throw ioe;
852      }
853
854      // Throw the exception if the thread is interrupted
855      if (Thread.currentThread().isInterrupted()) {
856        LOG.warn("Interrupted while trying for connection");
857        throw ioe;
858      }
859
860      try {
861        Thread.sleep(action.delayMillis);
862      } catch (InterruptedException e) {
863        throw (IOException)new InterruptedIOException("Interrupted: action="
864            + action + ", retry policy=" + connectionRetryPolicy).initCause(e);
865      }
866      LOG.info("Retrying connect to server: " + server + ". Already tried "
867          + curRetries + " time(s); retry policy is " + connectionRetryPolicy);
868    }

870    /**
871     * Write the connection header - this is sent when connection is established
872     * +----------------------------------+
873     * |  "hrpc" 4 bytes                  |      
874     * +----------------------------------+
875     * |  Version (1 byte)                |
876     * +----------------------------------+
877     * |  Service Class (1 byte)          |
878     * +----------------------------------+
879     * |  AuthProtocol (1 byte)           |      
880     * +----------------------------------+
881     */
882    private void writeConnectionHeader(OutputStream outStream)
883        throws IOException {
884      DataOutputStream out = new DataOutputStream(new BufferedOutputStream(outStream));
885      // Write out the header, version and authentication method
886      out.write(RpcConstants.HEADER.array());
887      out.write(RpcConstants.CURRENT_VERSION);
888      out.write(serviceClass);
889      out.write(authProtocol.callId);
890      out.flush();
891    }

893    /* Write the connection context header for each connection
894     * Out is not synchronized because only the first thread does this.
895     */
896    private void writeConnectionContext(ConnectionId remoteId,
897                                        AuthMethod authMethod)
898                                            throws IOException {
899      // Write out the ConnectionHeader
900      IpcConnectionContextProto message = ProtoUtil.makeIpcConnectionContext(
901          RPC.getProtocolName(remoteId.getProtocol()),
902          remoteId.getTicket(),
903          authMethod);
904      RpcRequestHeaderProto connectionContextHeader = ProtoUtil
905          .makeRpcRequestHeader(RpcKind.RPC_PROTOCOL_BUFFER,
906              OperationProto.RPC_FINAL_PACKET, CONNECTION_CONTEXT_CALL_ID,
907              RpcConstants.INVALID_RETRY_COUNT, clientId);
908      RpcRequestMessageWrapper request =
909          new RpcRequestMessageWrapper(connectionContextHeader, message);
910      
911      // Write out the packet length
912      out.writeInt(request.getLength());
913      request.write(out);
914    }

916    /* wait till someone signals us to start reading RPC response or
917     * it is idle too long, it is marked as to be closed, 
918     * or the client is marked as not running.
919     * 
920     * Return true if it is time to read a response; false otherwise.
921     */
922    private synchronized boolean waitForWork() {
923      if (calls.isEmpty() && !shouldCloseConnection.get()  && running.get())  {
924        long timeout = maxIdleTime-
925              (Time.now()-lastActivity.get());
926        if (timeout>0) {
927          try {
928            wait(timeout);
929          } catch (InterruptedException e) {}
930        }
931      }
932      
933      if (!calls.isEmpty() && !shouldCloseConnection.get() && running.get()) {
934        return true;
935      } else if (shouldCloseConnection.get()) {
936        return false;
937      } else if (calls.isEmpty()) { // idle connection closed or stopped
938        markClosed(null);
939        return false;
940      } else { // get stopped but there are still pending requests 
941        markClosed((IOException)new IOException().initCause(
942            new InterruptedException()));
943        return false;
944      }
945    }

951    /* Send a ping to the server if the time elapsed 
952     * since last I/O activity is equal to or greater than the ping interval
953     */
954    private synchronized void sendPing() throws IOException {
955      long curTime = Time.now();
956      if ( curTime - lastActivity.get() >= pingInterval) {
957        lastActivity.set(curTime);
958        synchronized (out) {
959          out.writeInt(pingRequest.size());
960          pingRequest.writeTo(out);
961          out.flush();
962        }
963      }
964    }

966    @Override
967    public void run() {
968      if (LOG.isDebugEnabled())
969        LOG.debug(getName() + ": starting, having connections " 
970            + connections.size());
971
972      try {
973        while (waitForWork()) {//wait here for work - read or close connection
974          receiveRpcResponse();
975        }
976      } catch (Throwable t) {
977        // This truly is unexpected, since we catch IOException in receiveResponse
978        // -- this is only to be really sure that we don't leave a client hanging
979        // forever.
980        LOG.warn("Unexpected error reading responses on connection " + this, t);
981        markClosed(new IOException("Error reading responses", t));
982      }
983      
984      close();
985      
986      if (LOG.isDebugEnabled())
987        LOG.debug(getName() + ": stopped, remaining connections "
988            + connections.size());
989    }

991    /** Initiates a rpc call by sending the rpc request to the remote server.
992     * Note: this is not called from the Connection thread, but by other
993     * threads.
994     * @param call - the rpc request
995     */
996    public void sendRpcRequest(final Call call)
997        throws InterruptedException, IOException {
998      if (shouldCloseConnection.get()) {
999        return;
1000      }
1001
1002      // Serialize the call to be sent. This is done from the actual
1003      // caller thread, rather than the sendParamsExecutor thread,
1004      
1005      // so that if the serialization throws an error, it is reported
1006      // properly. This also parallelizes the serialization.
1007      //
1008      // Format of a call on the wire:
1009      // 0) Length of rest below (1 + 2)
1010      // 1) RpcRequestHeader  - is serialized Delimited hence contains length
1011      // 2) RpcRequest
1012      //
1013      // Items '1' and '2' are prepared here. 
1014      final DataOutputBuffer d = new DataOutputBuffer();
1015      RpcRequestHeaderProto header = ProtoUtil.makeRpcRequestHeader(
1016          call.rpcKind, OperationProto.RPC_FINAL_PACKET, call.id, call.retry,
1017          clientId);
1018      header.writeDelimitedTo(d);
1019      call.rpcRequest.write(d);
1020
1021      synchronized (sendRpcRequestLock) {
1022        Future<?> senderFuture = sendParamsExecutor.submit(new Runnable() {
1023          @Override
1024          public void run() {
1025            try {
1026              synchronized (Connection.this.out) {
1027                if (shouldCloseConnection.get()) {
1028                  return;
1029                }
1030                
1031                if (LOG.isDebugEnabled())
1032                  LOG.debug(getName() + " sending #" + call.id);
1033         
1034                byte[] data = d.getData();
1035                int totalLength = d.getLength();
1036                out.writeInt(totalLength); // Total Length
1037                out.write(data, 0, totalLength);// RpcRequestHeader + RpcRequest
1038                out.flush();
1039              }
1040            } catch (IOException e) {
1041              // exception at this point would leave the connection in an
1042              // unrecoverable state (eg half a call left on the wire).
1043              // So, close the connection, killing any outstanding calls
1044              markClosed(e);
1045            } finally {
1046              //the buffer is just an in-memory buffer, but it is still polite to
1047              // close early
1048              IOUtils.closeStream(d);
1049            }
1050          }
1051        });
1052      
1053        try {
1054          senderFuture.get();
1055        } catch (ExecutionException e) {
1056          Throwable cause = e.getCause();
1057          
1058          // cause should only be a RuntimeException as the Runnable above
1059          // catches IOException
1060          if (cause instanceof RuntimeException) {
1061            throw (RuntimeException) cause;
1062          } else {
1063            throw new RuntimeException("unexpected checked exception", cause);
1064          }
1065        }
1066      }
1067    }

1069    /* Receive a response.
1070     * Because only one receiver, so no synchronization on in.
1071     */
1072    private void receiveRpcResponse() {
1073      if (shouldCloseConnection.get()) {
1074        return;
1075      }
1076      touch();
1077      
1078      try {
1079        int totalLen = in.readInt();
1080        RpcResponseHeaderProto header = 
1081            RpcResponseHeaderProto.parseDelimitedFrom(in);
1082        checkResponse(header);
1083
1084        int headerLen = header.getSerializedSize();
1085        headerLen += CodedOutputStream.computeRawVarint32Size(headerLen);
1086
1087        int callId = header.getCallId();
1088        if (LOG.isDebugEnabled())
1089          LOG.debug(getName() + " got value #" + callId);
1090
1091        Call call = calls.get(callId);
1092        RpcStatusProto status = header.getStatus();
1093        if (status == RpcStatusProto.SUCCESS) {
1094          Writable value = ReflectionUtils.newInstance(valueClass, conf);
1095          value.readFields(in);                 // read value
1096          calls.remove(callId);
1097          call.setRpcResponse(value);
1098          
1099          // verify that length was correct
1100          // only for ProtobufEngine where len can be verified easily
1101          if (call.getRpcResponse() instanceof ProtobufRpcEngine.RpcWrapper) {
1102            ProtobufRpcEngine.RpcWrapper resWrapper = 
1103                (ProtobufRpcEngine.RpcWrapper) call.getRpcResponse();
1104            if (totalLen != headerLen + resWrapper.getLength()) { 
1105              throw new RpcClientException(
1106                  "RPC response length mismatch on rpc success");
1107            }
1108          }
1109        } else { // Rpc Request failed
1110          // Verify that length was correct
1111          if (totalLen != headerLen) {
1112            throw new RpcClientException(
1113                "RPC response length mismatch on rpc error");
1114          }
1115          
1116          final String exceptionClassName = header.hasExceptionClassName() ?
1117                header.getExceptionClassName() : 
1118                  "ServerDidNotSetExceptionClassName";
1119          final String errorMsg = header.hasErrorMsg() ? 
1120                header.getErrorMsg() : "ServerDidNotSetErrorMsg" ;
1121          final RpcErrorCodeProto erCode = 
1122                    (header.hasErrorDetail() ? header.getErrorDetail() : null);
1123          if (erCode == null) {
1124             LOG.warn("Detailed error code not set by server on rpc error");
1125          }
1126          RemoteException re = 
1127              ( (erCode == null) ? 
1128                  new RemoteException(exceptionClassName, errorMsg) :
1129              new RemoteException(exceptionClassName, errorMsg, erCode));
1130          if (status == RpcStatusProto.ERROR) {
1131            calls.remove(callId);
1132            call.setException(re);
1133          } else if (status == RpcStatusProto.FATAL) {
1134            // Close the connection
1135            markClosed(re);
1136          }
1137        }
1138      } catch (IOException e) {
1139        markClosed(e);
1140      }
1141    }
1142    
1143    private synchronized void markClosed(IOException e) {
1144      if (shouldCloseConnection.compareAndSet(false, true)) {
1145        closeException = e;
1146        notifyAll();
1147      }
1148    }

1150    /** Close the connection. */
1151    private synchronized void close() {
1152      if (!shouldCloseConnection.get()) {
1153        LOG.error("The connection is not in the closed state");
1154        return;
1155      }
1156
1157      // release the resources
1158      // first thing to do;take the connection out of the connection list
1159      synchronized (connections) {
1160        if (connections.get(remoteId) == this) {
1161          connections.remove(remoteId);
1162        }
1163      }
1164
1165      // close the streams and therefore the socket
1166      IOUtils.closeStream(out);
1167      IOUtils.closeStream(in);
1168      disposeSasl();
1169
1170      // clean up all calls
1171      if (closeException == null) {
1172        if (!calls.isEmpty()) {
1173          LOG.warn(
1174              "A connection is closed for no cause and calls are not empty");
1175
1176          // clean up calls anyway
1177          closeException = new IOException("Unexpected closed connection");
1178          cleanupCalls();
1179        }
1180      } else {
1181        // log the info
1182        if (LOG.isDebugEnabled()) {
1183          LOG.debug("closing ipc connection to " + server + ": " +
1184              closeException.getMessage(),closeException);
1185        }
1186
1187        // cleanup calls
1188        cleanupCalls();
1189      }
1190      closeConnection();
1191      if (LOG.isDebugEnabled())
1192        LOG.debug(getName() + ": closed");
1193    }
1194    
1195    /* Cleanup all calls and mark them as done */
1196    private void cleanupCalls() {
1197      Iterator<Entry<Integer, Call>> itor = calls.entrySet().iterator() ;
1198      while (itor.hasNext()) {
1199        Call c = itor.next().getValue(); 
1200        itor.remove();
1201        c.setException(closeException); // local exception
1202      }
1203    }

1238  /** Stop all threads related to this client.  No further calls may be made
1239   * using this client. */
1240  public void stop() {
1241    if (LOG.isDebugEnabled()) {
1242      LOG.debug("Stopping client");
1243    }
1244
1245    if (!running.compareAndSet(true, false)) {
1246      return;
1247    }
1248    
1249    // wake up all connections
1250    synchronized (connections) {
1251      for (Connection conn : connections.values()) {
1252        conn.interrupt();
1253      }
1254    }
1255    
1256    // wait until all connections are closed
1257    while (!connections.isEmpty()) {
1258      try {
1259        Thread.sleep(100);
1260      } catch (InterruptedException e) {
1261      }
1262    }
1263    
1264    clientExcecutorFactory.unrefAndCleanup();
1265  }

1428  /**
1429   * Make a call, passing <code>rpcRequest</code>, to the IPC server defined by
1430   * <code>remoteId</code>, returning the rpc response.
1431   *
1432   * @param rpcKind
1433   * @param rpcRequest -  contains serialized method and method parameters
1434   * @param remoteId - the target rpc server
1435   * @param serviceClass - service class for RPC
1436   * @param fallbackToSimpleAuth - set to true or false during this method to
1437   *   indicate if a secure client falls back to simple auth
1438   * @returns the rpc response
1439   * Throws exceptions if there are network problems or if the remote code
1440   * threw an exception.
1441   */
1442  public Writable call(RPC.RpcKind rpcKind, Writable rpcRequest,
1443      ConnectionId remoteId, int serviceClass,
1444      AtomicBoolean fallbackToSimpleAuth) throws IOException {
1445    final Call call = createCall(rpcKind, rpcRequest);
1446    Connection connection = getConnection(remoteId, call, serviceClass,
1447      fallbackToSimpleAuth);
1448    try {
1449      connection.sendRpcRequest(call);                 // send the rpc request
1450    } catch (RejectedExecutionException e) {
1451      throw new IOException("connection has been closed", e);
1452    } catch (InterruptedException e) {
1453      Thread.currentThread().interrupt();
1454      LOG.warn("interrupted waiting to send rpc request to server", e);
1455      throw new IOException(e);
1456    }
1457
1458    boolean interrupted = false;
1459    synchronized (call) {
1460      while (!call.done) {
1461        try {
1462          call.wait();                           // wait for the result
1463        } catch (InterruptedException ie) {
1464          // save the fact that we were interrupted
1465          interrupted = true;
1466        }
1467      }
1468
1469      if (interrupted) {
1470        // set the interrupt flag now that we are done waiting
1471        Thread.currentThread().interrupt();
1472      }
1473
1474      if (call.error != null) {
1475        if (call.error instanceof RemoteException) {
1476          call.error.fillInStackTrace();
1477          throw call.error;
1478        } else { // local exception
1479          InetSocketAddress address = connection.getRemoteAddress();
1480          throw NetUtils.wrapException(address.getHostName(),
1481                  address.getPort(),
1482                  NetUtils.getHostname(),
1483                  0,
1484                  call.error);
1485        }
1486      } else {
1487        return call.getRpcResponse();
1488      }
1489    }
1490  }

1501  /** Get a connection from the pool, or create a new one and add it to the
1502   * pool.  Connections to a given ConnectionId are reused. */
1503  private Connection getConnection(ConnectionId remoteId,
1504      Call call, int serviceClass, AtomicBoolean fallbackToSimpleAuth)
1505      throws IOException {
1506    if (!running.get()) {
1507      // the client is stopped
1508      throw new IOException("The client is stopped");
1509    }
1510    Connection connection;
1511    /* we could avoid this allocation for each RPC by having a  
1512     * connectionsId object and with set() method. We need to manage the
1513     * refs for keys in HashMap properly. For now its ok.
1514     */
1515    do {
1516      synchronized (connections) {
1517        connection = connections.get(remoteId);
1518        if (connection == null) {
1519          connection = new Connection(remoteId, serviceClass);
1520          connections.put(remoteId, connection);
1521        }
1522      }
1523    } while (!connection.addCall(call));
1524    
1525    //we don't invoke the method below inside "synchronized (connections)"
1526    //block above. The reason for that is if the server happens to be slow,
1527    //it will take longer to establish a connection and that will slow the
1528    //entire system down.
1529    connection.setupIOstreams(fallbackToSimpleAuth);
1530    return connection;
1531  }

1712  /**
1713   * Returns the next valid sequential call ID by incrementing an atomic counter
1714   * and masking off the sign bit.  Valid call IDs are non-negative integers in
1715   * the range [ 0, 2^31 - 1 ].  Negative numbers are reserved for special
1716   * purposes.  The values can overflow back to 0 and be reused.  Note that prior
1717   * versions of the client did not mask off the sign bit, so a server may still
1718   * see a negative call ID if it receives connections from an old client.
1719   * 
1720   * @return next call ID
1721   */
1722  public static int nextCallId() {
1723    return callIdCounter.getAndIncrement() & 0x7FFFFFFF;
