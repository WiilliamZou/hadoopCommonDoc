119  private static class Invoker implements RpcInvocationHandler {
120    private final Map<String, Message> returnTypes = 
121        new ConcurrentHashMap<String, Message>();
122    private boolean isClosed = false;
123    private final Client.ConnectionId remoteId;
124    private final Client client;
125    private final long clientProtocolVersion;
126    private final String protocolName;
127    private AtomicBoolean fallbackToSimpleAuth;
128
129    private Invoker(Class<?> protocol, InetSocketAddress addr,
130        UserGroupInformation ticket, Configuration conf, SocketFactory factory,
131        int rpcTimeout, RetryPolicy connectionRetryPolicy,
132        AtomicBoolean fallbackToSimpleAuth) throws IOException {
133      this(protocol, Client.ConnectionId.getConnectionId(
134          addr, protocol, ticket, rpcTimeout, connectionRetryPolicy, conf),
135          conf, factory);
136      this.fallbackToSimpleAuth = fallbackToSimpleAuth;
137    }
138    
139    /**
140     * This constructor takes a connectionId, instead of creating a new one.
141     */
142    private Invoker(Class<?> protocol, Client.ConnectionId connId,
143        Configuration conf, SocketFactory factory) {
144      this.remoteId = connId;
145      this.client = CLIENTS.getClient(conf, factory, RpcResponseWrapper.class);
146      this.protocolName = RPC.getProtocolName(protocol);
147      this.clientProtocolVersion = RPC
148          .getProtocolVersion(protocol);
149    }
150
151    private RequestHeaderProto constructRpcRequestHeader(Method method) {
152      RequestHeaderProto.Builder builder = RequestHeaderProto
153          .newBuilder();
154      builder.setMethodName(method.getName());
155     
156
157      // For protobuf, {@code protocol} used when creating client side proxy is
158      // the interface extending BlockingInterface, which has the annotations 
159      // such as ProtocolName etc.
160      //
161      // Using Method.getDeclaringClass(), as in WritableEngine to get at
162      // the protocol interface will return BlockingInterface, from where 
163      // the annotation ProtocolName and Version cannot be
164      // obtained.
165      //
166      // Hence we simply use the protocol class used to create the proxy.
167      // For PB this may limit the use of mixins on client side.
168      builder.setDeclaringClassProtocolName(protocolName);
169      builder.setClientProtocolVersion(clientProtocolVersion);
170      return builder.build();
171    }
172
173    /**
174     * This is the client side invoker of RPC method. It only throws
175     * ServiceException, since the invocation proxy expects only
176     * ServiceException to be thrown by the method in case protobuf service.
177     * 
178     * ServiceException has the following causes:
179     * <ol>
180     * <li>Exceptions encountered on the client side in this method are 
181     * set as cause in ServiceException as is.</li>
182     * <li>Exceptions from the server are wrapped in RemoteException and are
183     * set as cause in ServiceException</li>
184     * </ol>
185     * 
186     * Note that the client calling protobuf RPC methods, must handle
187     * ServiceException by getting the cause from the ServiceException. If the
188     * cause is RemoteException, then unwrap it to get the exception thrown by
189     * the server.
190     */
191    @Override
192    public Object invoke(Object proxy, Method method, Object[] args)
193        throws ServiceException {
194      long startTime = 0;
195      if (LOG.isDebugEnabled()) {
196        startTime = Time.now();
197      }
198      
199      if (args.length != 2) { // RpcController + Message
200        throw new ServiceException("Too many parameters for request. Method: ["
201            + method.getName() + "]" + ", Expected: 2, Actual: "
202            + args.length);
203      }
204      if (args[1] == null) {
205        throw new ServiceException("null param while calling Method: ["
206            + method.getName() + "]");
207      }
208
209      TraceScope traceScope = null;
210      // if Tracing is on then start a new span for this rpc.
211      // guard it in the if statement to make sure there isn't
212      // any extra string manipulation.
213      if (Trace.isTracing()) {
214        traceScope = Trace.startSpan(RpcClientUtil.methodToTraceString(method));
215      }
216
217      RequestHeaderProto rpcRequestHeader = constructRpcRequestHeader(method);
218      
219      if (LOG.isTraceEnabled()) {
220        LOG.trace(Thread.currentThread().getId() + ": Call -> " +
221            remoteId + ": " + method.getName() +
222            " {" + TextFormat.shortDebugString((Message) args[1]) + "}");
223      }
224
225
226      Message theRequest = (Message) args[1];
227      final RpcResponseWrapper val;
228      try {
229        val = (RpcResponseWrapper) client.call(RPC.RpcKind.RPC_PROTOCOL_BUFFER,
230            new RpcRequestWrapper(rpcRequestHeader, theRequest), remoteId,
231            fallbackToSimpleAuth);
232
233      } catch (Throwable e) {
234        if (LOG.isTraceEnabled()) {
235          LOG.trace(Thread.currentThread().getId() + ": Exception <- " +
236              remoteId + ": " + method.getName() +
237                " {" + e + "}");
238        }
239        if (Trace.isTracing()) {
240          traceScope.getSpan().addTimelineAnnotation(
241              "Call got exception: " + e.getMessage());
242        }
243        throw new ServiceException(e);
244      } finally {
245        if (traceScope != null) traceScope.close();
246      }
247
248      if (LOG.isDebugEnabled()) {
249        long callTime = Time.now() - startTime;
250        LOG.debug("Call: " + method.getName() + " took " + callTime + "ms");
251      }
252      
253      Message prototype = null;
254      try {
255        prototype = getReturnProtoType(method);
256      } catch (Exception e) {
257        throw new ServiceException(e);
258      }
259      Message returnMessage;
260      try {
261        returnMessage = prototype.newBuilderForType()
262            .mergeFrom(val.theResponseRead).build();
263
264        if (LOG.isTraceEnabled()) {
265          LOG.trace(Thread.currentThread().getId() + ": Response <- " +
266              remoteId + ": " + method.getName() +
267                " {" + TextFormat.shortDebugString(returnMessage) + "}");
268        }
269
270      } catch (Throwable e) {
271        throw new ServiceException(e);
272      }
273      return returnMessage;
274    }
275
276    @Override
277    public void close() throws IOException {
278      if (!isClosed) {
279        isClosed = true;
280        CLIENTS.stopClient(client);
281      }
282    }
283
284    private Message getReturnProtoType(Method method) throws Exception {
285      if (returnTypes.containsKey(method.getName())) {
286        return returnTypes.get(method.getName());
287      }
288      
289      Class<?> returnType = method.getReturnType();
290      Method newInstMethod = returnType.getMethod("getDefaultInstance");
291      newInstMethod.setAccessible(true);
292      Message prototype = (Message) newInstMethod.invoke(null, (Object[]) null);
293      returnTypes.put(method.getName(), prototype);
294      return prototype;
295    }

299      return remoteId;
300    }

303  interface RpcWrapper extends Writable {
304    int getLength();
305  }
306  /**
307   * Wrapper for Protocol Buffer Requests
308   * 
309   * Note while this wrapper is writable, the request on the wire is in
310   * Protobuf. Several methods on {@link org.apache.hadoop.ipc.Server and RPC} 
311   * use type Writable as a wrapper to work across multiple RpcEngine kinds.
312   */
313  private static abstract class RpcMessageWithHeader<T extends GeneratedMessage>
314    implements RpcWrapper {
315    T requestHeader;
316    Message theRequest; // for clientSide, the request is here
317    byte[] theRequestRead; // for server side, the request is here
318
319    public RpcMessageWithHeader() {
320    }
321
322    public RpcMessageWithHeader(T requestHeader, Message theRequest) {
323      this.requestHeader = requestHeader;
324      this.theRequest = theRequest;
325    }
326
327    @Override
328    public void write(DataOutput out) throws IOException {
329      OutputStream os = DataOutputOutputStream.constructOutputStream(out);
330      
331      ((Message)requestHeader).writeDelimitedTo(os);
332      theRequest.writeDelimitedTo(os);
333    }
334
335    @Override
336    public void readFields(DataInput in) throws IOException {
337      requestHeader = parseHeaderFrom(readVarintBytes(in));
338      theRequestRead = readMessageRequest(in);
339    }
340
341    abstract T parseHeaderFrom(byte[] bytes) throws IOException;
342
343    byte[] readMessageRequest(DataInput in) throws IOException {
344      return readVarintBytes(in);
345    }
346
347    private static byte[] readVarintBytes(DataInput in) throws IOException {
348      final int length = ProtoUtil.readRawVarint32(in);
349      final byte[] bytes = new byte[length];
350      in.readFully(bytes);
351      return bytes;
352    }
353
354    public T getMessageHeader() {
355      return requestHeader;
356    }
357
358    public byte[] getMessageBytes() {
359      return theRequestRead;
360    }
361    
362    @Override
363    public int getLength() {
364      int headerLen = requestHeader.getSerializedSize();
365      int reqLen;
366      if (theRequest != null) {
367        reqLen = theRequest.getSerializedSize();
368      } else if (theRequestRead != null ) {
369        reqLen = theRequestRead.length;
370      } else {
371        throw new IllegalArgumentException(
372            "getLength on uninitialized RpcWrapper");      
373      }
374      return CodedOutputStream.computeRawVarint32Size(headerLen) +  headerLen
375          + CodedOutputStream.computeRawVarint32Size(reqLen) + reqLen;
376    }
377  }

542    /**
543     * Protobuf invoker for {@link RpcInvoker}
544     */
545    static class ProtoBufRpcInvoker implements RpcInvoker {
546      private static ProtoClassProtoImpl getProtocolImpl(RPC.Server server,
547          String protoName, long clientVersion) throws RpcServerException {
548        ProtoNameVer pv = new ProtoNameVer(protoName, clientVersion);
549        ProtoClassProtoImpl impl = 
550            server.getProtocolImplMap(RPC.RpcKind.RPC_PROTOCOL_BUFFER).get(pv);
551        if (impl == null) { // no match for Protocol AND Version
552          VerProtocolImpl highest = 
553              server.getHighestSupportedProtocol(RPC.RpcKind.RPC_PROTOCOL_BUFFER, 
554                  protoName);
555          if (highest == null) {
556            throw new RpcNoSuchProtocolException(
557                "Unknown protocol: " + protoName);
558          }
559          // protocol supported but not the version that client wants
560          throw new RPC.VersionMismatch(protoName, clientVersion,
561              highest.version);
562        }
563        return impl;
564      }
565
566      @Override 
567      /**
568       * This is a server side method, which is invoked over RPC. On success
569       * the return response has protobuf response payload. On failure, the
570       * exception name and the stack trace are return in the resposne.
571       * See {@link HadoopRpcResponseProto}
572       * 
573       * In this method there three types of exceptions possible and they are
574       * returned in response as follows.
575       * <ol>
576       * <li> Exceptions encountered in this method that are returned 
577       * as {@link RpcServerException} </li>
578       * <li> Exceptions thrown by the service is wrapped in ServiceException. 
579       * In that this method returns in response the exception thrown by the 
580       * service.</li>
581       * <li> Other exceptions thrown by the service. They are returned as
582       * it is.</li>
583       * </ol>
584       */
585      public Writable call(RPC.Server server, String protocol,
586          Writable writableRequest, long receiveTime) throws Exception {
587        RpcRequestWrapper request = (RpcRequestWrapper) writableRequest;
588        RequestHeaderProto rpcRequest = request.requestHeader;
589        String methodName = rpcRequest.getMethodName();
590        String protoName = rpcRequest.getDeclaringClassProtocolName();
591        long clientVersion = rpcRequest.getClientProtocolVersion();
592        if (server.verbose)
593          LOG.info("Call: protocol=" + protocol + ", method=" + methodName);
594        
595        ProtoClassProtoImpl protocolImpl = getProtocolImpl(server, protoName,
596            clientVersion);
597        BlockingService service = (BlockingService) protocolImpl.protocolImpl;
598        MethodDescriptor methodDescriptor = service.getDescriptorForType()
599            .findMethodByName(methodName);
600        if (methodDescriptor == null) {
601          String msg = "Unknown method " + methodName + " called on " + protocol
602              + " protocol.";
603          LOG.warn(msg);
604          throw new RpcNoSuchMethodException(msg);
605        }
606        Message prototype = service.getRequestPrototype(methodDescriptor);
607        Message param = prototype.newBuilderForType()
608            .mergeFrom(request.theRequestRead).build();
609        
610        Message result;
611        long startTime = Time.now();
612        int qTime = (int) (startTime - receiveTime);
613        Exception exception = null;
614        try {
615          server.rpcDetailedMetrics.init(protocolImpl.protocolClass);
616          result = service.callBlockingMethod(methodDescriptor, null, param);
617        } catch (ServiceException e) {
618          exception = (Exception) e.getCause();
619          throw (Exception) e.getCause();
620        } catch (Exception e) {
621          exception = e;
622          throw e;
623        } finally {
624          int processingTime = (int) (Time.now() - startTime);
625          if (LOG.isDebugEnabled()) {
626            String msg = "Served: " + methodName + " queueTime= " + qTime +
627                " procesingTime= " + processingTime;
628            if (exception != null) {
629              msg += " exception= " + exception.getClass().getSimpleName();
630            }
631            LOG.debug(msg);
632          }
633          String detailedMetricsName = (exception == null) ?
634              methodName :
635              exception.getClass().getSimpleName();
636          server.rpcMetrics.addRpcQueueTime(qTime);
637          server.rpcMetrics.addRpcProcessingTime(processingTime);
638          server.rpcDetailedMetrics.addProcessingTime(detailedMetricsName,
639              processingTime);
640        }
641        return new RpcResponseWrapper(result);
642      }
643    }
