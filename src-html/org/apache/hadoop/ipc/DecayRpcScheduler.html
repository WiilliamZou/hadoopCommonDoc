042/**
043 * The decay RPC scheduler counts incoming requests in a map, then
044 * decays the counts at a fixed time interval. The scheduler is optimized
045 * for large periods (on the order of seconds), as it offloads work to the
046 * decay sweep.
047 */
048public class DecayRpcScheduler implements RpcScheduler, DecayRpcSchedulerMXBean {
049  /**
050   * Period controls how many milliseconds between each decay sweep.
051   */
052  public static final String IPC_CALLQUEUE_DECAYSCHEDULER_PERIOD_KEY =
053    "faircallqueue.decay-scheduler.period-ms";
054  public static final long   IPC_CALLQUEUE_DECAYSCHEDULER_PERIOD_DEFAULT =
055    5000L;
056
057  /**
058   * Decay factor controls how much each count is suppressed by on each sweep.
059   * Valid numbers are > 0 and < 1. Decay factor works in tandem with period
060   * to control how long the scheduler remembers an identity.
061   */
062  public static final String IPC_CALLQUEUE_DECAYSCHEDULER_FACTOR_KEY =
063    "faircallqueue.decay-scheduler.decay-factor";
064  public static final double IPC_CALLQUEUE_DECAYSCHEDULER_FACTOR_DEFAULT =
065    0.5;
066
067  /**
068   * Thresholds are specified as integer percentages, and specify which usage
069   * range each queue will be allocated to. For instance, specifying the list
070   *  10, 40, 80
071   * implies 4 queues, with
072   * - q3 from 80% up
073   * - q2 from 40 up to 80
074   * - q1 from 10 up to 40
075   * - q0 otherwise.
076   */
077  public static final String IPC_CALLQUEUE_DECAYSCHEDULER_THRESHOLDS_KEY =
078    "faircallqueue.decay-scheduler.thresholds";
079
080  // Specifies the identity to use when the IdentityProvider cannot handle
081  // a schedulable.
082  public static final String DECAYSCHEDULER_UNKNOWN_IDENTITY =
083    "IdentityProvider.Unknown";
084
085  public static final Log LOG = LogFactory.getLog(DecayRpcScheduler.class);
086
087  // Track the number of calls for each schedulable identity
088  private final ConcurrentHashMap<Object, AtomicLong> callCounts =
089    new ConcurrentHashMap<Object, AtomicLong>();
090
091  // Should be the sum of all AtomicLongs in callCounts
092  private final AtomicLong totalCalls = new AtomicLong();
093
094  // Pre-computed scheduling decisions during the decay sweep are
095  // atomically swapped in as a read-only map
096  private final AtomicReference<Map<Object, Integer>> scheduleCacheRef =
097    new AtomicReference<Map<Object, Integer>>();
098
099  // Tune the behavior of the scheduler
100  private final long decayPeriodMillis; // How long between each tick
101  private final double decayFactor; // nextCount = currentCount / decayFactor
102  private final int numQueues; // affects scheduling decisions, from 0 to numQueues - 1
103  private final double[] thresholds;
104  private final IdentityProvider identityProvider;
105
106  /**
107   * This TimerTask will call decayCurrentCounts until
108   * the scheduler has been garbage collected.
109   */
110  public static class DecayTask extends TimerTask {
111    private WeakReference<DecayRpcScheduler> schedulerRef;
112    private Timer timer;
113
114    public DecayTask(DecayRpcScheduler scheduler, Timer timer) {
115      this.schedulerRef = new WeakReference<DecayRpcScheduler>(scheduler);
116      this.timer = timer;
117    }
118
119    @Override
120    public void run() {
121      DecayRpcScheduler sched = schedulerRef.get();
122      if (sched != null) {
123        sched.decayCurrentCounts();
124      } else {
125        // Our scheduler was garbage collected since it is no longer in use,
126        // so we should terminate the timer as well
127        timer.cancel();
128        timer.purge();
129      }
130    }
131  }
132
133  /**
134   * Create a decay scheduler.
135   * @param numQueues number of queues to schedule for
136   * @param ns config prefix, so that we can configure multiple schedulers
137   *           in a single instance.
138   * @param conf configuration to use.
139   */
140  public DecayRpcScheduler(int numQueues, String ns, Configuration conf) {
141    if (numQueues < 1) {
142      throw new IllegalArgumentException("number of queues must be > 0");
143    }
144
145    this.numQueues = numQueues;
146    this.decayFactor = parseDecayFactor(ns, conf);
147    this.decayPeriodMillis = parseDecayPeriodMillis(ns, conf);
148    this.identityProvider = this.parseIdentityProvider(ns, conf);
149    this.thresholds = parseThresholds(ns, conf, numQueues);
150
151    // Setup delay timer
152    Timer timer = new Timer();
153    DecayTask task = new DecayTask(this, timer);
154    timer.scheduleAtFixedRate(task, 0, this.decayPeriodMillis);
155
156    MetricsProxy prox = MetricsProxy.getInstance(ns);
157    prox.setDelegate(this);
158  }
159
160  // Load configs
161  private IdentityProvider parseIdentityProvider(String ns, Configuration conf) {
162    List<IdentityProvider> providers = conf.getInstances(
163      ns + "." + CommonConfigurationKeys.IPC_CALLQUEUE_IDENTITY_PROVIDER_KEY,
164      IdentityProvider.class);
165
166    if (providers.size() < 1) {
167      LOG.info("IdentityProvider not specified, " +
168        "defaulting to UserIdentityProvider");
169      return new UserIdentityProvider();
170    }
171
172    return providers.get(0); // use the first
173  }
174
175  private static double parseDecayFactor(String ns, Configuration conf) {
176    double factor = conf.getDouble(ns + "." +
177        IPC_CALLQUEUE_DECAYSCHEDULER_FACTOR_KEY,
178      IPC_CALLQUEUE_DECAYSCHEDULER_FACTOR_DEFAULT
179    );
180
181    if (factor <= 0 || factor >= 1) {
182      throw new IllegalArgumentException("Decay Factor " +
183        "must be between 0 and 1");
184    }
185
186    return factor;
187  }
188
189  private static long parseDecayPeriodMillis(String ns, Configuration conf) {
190    long period = conf.getLong(ns + "." +
191        IPC_CALLQUEUE_DECAYSCHEDULER_PERIOD_KEY,
192      IPC_CALLQUEUE_DECAYSCHEDULER_PERIOD_DEFAULT
193    );
194
195    if (period <= 0) {
196      throw new IllegalArgumentException("Period millis must be >= 0");
197    }
198
199    return period;
200  }
201
202  private static double[] parseThresholds(String ns, Configuration conf,
203      int numQueues) {
204    int[] percentages = conf.getInts(ns + "." +
205      IPC_CALLQUEUE_DECAYSCHEDULER_THRESHOLDS_KEY);
206
207    if (percentages.length == 0) {
208      return getDefaultThresholds(numQueues);
209    } else if (percentages.length != numQueues-1) {
210      throw new IllegalArgumentException("Number of thresholds should be " +
211        (numQueues-1) + ". Was: " + percentages.length);
212    }
213
214    // Convert integer percentages to decimals
215    double[] decimals = new double[percentages.length];
216    for (int i = 0; i < percentages.length; i++) {
217      decimals[i] = percentages[i] / 100.0;
218    }
219
220    return decimals;
221  }
222
223  /**
224   * Generate default thresholds if user did not specify. Strategy is
225   * to halve each time, since queue usage tends to be exponential.
226   * So if numQueues is 4, we would generate: double[]{0.125, 0.25, 0.5}
227   * which specifies the boundaries between each queue's usage.
228   * @param numQueues number of queues to compute for
229   * @return array of boundaries of length numQueues - 1
230   */
231  private static double[] getDefaultThresholds(int numQueues) {
232    double[] ret = new double[numQueues - 1];
233    double div = Math.pow(2, numQueues - 1);
234
235    for (int i = 0; i < ret.length; i++) {
236      ret[i] = Math.pow(2, i)/div;
237    }
238    return ret;
239  }

241  /**
242   * Decay the stored counts for each user and clean as necessary.
243   * This method should be called periodically in order to keep
244   * counts current.
245   */
246  private void decayCurrentCounts() {
247    long total = 0;
248    Iterator<Map.Entry<Object, AtomicLong>> it =
249      callCounts.entrySet().iterator();
250
251    while (it.hasNext()) {
252      Map.Entry<Object, AtomicLong> entry = it.next();
253      AtomicLong count = entry.getValue();
254
255      // Compute the next value by reducing it by the decayFactor
256      long currentValue = count.get();
257      long nextValue = (long)(currentValue * decayFactor);
258      total += nextValue;
259      count.set(nextValue);
260
261      if (nextValue == 0) {
262        // We will clean up unused keys here. An interesting optimization might
263        // be to have an upper bound on keyspace in callCounts and only
264        // clean once we pass it.
265        it.remove();
266      }
267    }
268
269    // Update the total so that we remain in sync
270    totalCalls.set(total);
271
272    // Now refresh the cache of scheduling decisions
273    recomputeScheduleCache();
274  }
275
276  /**
277   * Update the scheduleCache to match current conditions in callCounts.
278   */
279  private void recomputeScheduleCache() {
280    Map<Object, Integer> nextCache = new HashMap<Object, Integer>();
281
282    for (Map.Entry<Object, AtomicLong> entry : callCounts.entrySet()) {
283      Object id = entry.getKey();
284      AtomicLong value = entry.getValue();
285
286      long snapshot = value.get();
287      int computedLevel = computePriorityLevel(snapshot);
288
289      nextCache.put(id, computedLevel);
290    }
291
292    // Swap in to activate
293    scheduleCacheRef.set(Collections.unmodifiableMap(nextCache));
294  }

296  /**
297   * Get the number of occurrences and increment atomically.
298   * @param identity the identity of the user to increment
299   * @return the value before incrementation
300   */
301  private long getAndIncrement(Object identity) throws InterruptedException {
302    // We will increment the count, or create it if no such count exists
303    AtomicLong count = this.callCounts.get(identity);
304    if (count == null) {
305      // Create the count since no such count exists.
306      count = new AtomicLong(0);
307
308      // Put it in, or get the AtomicInteger that was put in by another thread
309      AtomicLong otherCount = callCounts.putIfAbsent(identity, count);
310      if (otherCount != null) {
311        count = otherCount;
312      }
313    }
314
315    // Update the total
316    totalCalls.getAndIncrement();
317
318    // At this point value is guaranteed to be not null. It may however have
319    // been clobbered from callCounts. Nonetheless, we return what
320    // we have.
321    return count.getAndIncrement();
322  }

324  /**
325   * Given the number of occurrences, compute a scheduling decision.
326   * @param occurrences how many occurrences
327   * @return scheduling decision from 0 to numQueues - 1
328   */
329  private int computePriorityLevel(long occurrences) {
330    long totalCallSnapshot = totalCalls.get();
331
332    double proportion = 0;
333    if (totalCallSnapshot > 0) {
334      proportion = (double) occurrences / totalCallSnapshot;
335    }
336
337    // Start with low priority queues, since they will be most common
338    for(int i = (numQueues - 1); i > 0; i--) {
339      if (proportion >= this.thresholds[i - 1]) {
340        return i; // We've found our queue number
341      }
342    }
343
344    // If we get this far, we're at queue 0
345    return 0;
346  }
347
348  /**
349   * Returns the priority level for a given identity by first trying the cache,
350   * then computing it.
351   * @param identity an object responding to toString and hashCode
352   * @return integer scheduling decision from 0 to numQueues - 1
353   */
354  private int cachedOrComputedPriorityLevel(Object identity) {
355    try {
356      long occurrences = this.getAndIncrement(identity);
357
358      // Try the cache
359      Map<Object, Integer> scheduleCache = scheduleCacheRef.get();
360      if (scheduleCache != null) {
361        Integer priority = scheduleCache.get(identity);
362        if (priority != null) {
363          return priority;
364        }
365      }
366
367      // Cache was no good, compute it
368      return computePriorityLevel(occurrences);
369    } catch (InterruptedException ie) {
370      LOG.warn("Caught InterruptedException, returning low priority queue");
371      return numQueues - 1;
372    }
373  }
374
375  /**
376   * Compute the appropriate priority for a schedulable based on past requests.
377   * @param obj the schedulable obj to query and remember
378   * @return the queue index which we recommend scheduling in
379   */
380  @Override
381  public int getPriorityLevel(Schedulable obj) {
382    // First get the identity
383    String identity = this.identityProvider.makeIdentity(obj);
384    if (identity == null) {
385      // Identity provider did not handle this
386      identity = DECAYSCHEDULER_UNKNOWN_IDENTITY;
387    }
388
389    return cachedOrComputedPriorityLevel(identity);
390  }

421  /**
422   * MetricsProxy is a singleton because we may init multiple schedulers and we
423   * want to clean up resources when a new scheduler replaces the old one.
424   */
425  private static final class MetricsProxy implements DecayRpcSchedulerMXBean {
426    // One singleton per namespace
427    private static final HashMap<String, MetricsProxy> INSTANCES =
428      new HashMap<String, MetricsProxy>();
429
430    // Weakref for delegate, so we don't retain it forever if it can be GC'd
431    private WeakReference<DecayRpcScheduler> delegate;
432
433    private MetricsProxy(String namespace) {
434      MBeans.register(namespace, "DecayRpcScheduler", this);
435    }
436
437    public static synchronized MetricsProxy getInstance(String namespace) {
438      MetricsProxy mp = INSTANCES.get(namespace);
439      if (mp == null) {
440        // We must create one
441        mp = new MetricsProxy(namespace);
442        INSTANCES.put(namespace, mp);
443      }
444      return mp;
445    }
446
447    public void setDelegate(DecayRpcScheduler obj) {
448      this.delegate = new WeakReference<DecayRpcScheduler>(obj);
449    }
450
451    @Override
452    public String getSchedulingDecisionSummary() {
453      DecayRpcScheduler scheduler = delegate.get();
454      if (scheduler == null) {
455        return "No Active Scheduler";
456      } else {
457        return scheduler.getSchedulingDecisionSummary();
458      }
459    }
460
461    @Override
462    public String getCallVolumeSummary() {
463      DecayRpcScheduler scheduler = delegate.get();
464      if (scheduler == null) {
465        return "No Active Scheduler";
466      } else {
467        return scheduler.getCallVolumeSummary();
468      }
469    }
470
471    @Override
472    public int getUniqueIdentityCount() {
473      DecayRpcScheduler scheduler = delegate.get();
474      if (scheduler == null) {
475        return -1;
476      } else {
477        return scheduler.getUniqueIdentityCount();
478      }
479    }
480
481    @Override
482    public long getTotalCallVolume() {
483      DecayRpcScheduler scheduler = delegate.get();
484      if (scheduler == null) {
485        return -1;
486      } else {
487        return scheduler.getTotalCallVolume();
488      }
489    }
490  }
491
492  public int getUniqueIdentityCount() {
493    return callCounts.size();
494  }
495
496  public long getTotalCallVolume() {
497    return totalCalls.get();
498  }
499
500  public String getSchedulingDecisionSummary() {
501    Map<Object, Integer> decisions = scheduleCacheRef.get();
502    if (decisions == null) {
503      return "{}";
504    } else {
505      try {
506        ObjectMapper om = new ObjectMapper();
507        return om.writeValueAsString(decisions);
508      } catch (Exception e) {
509        return "Error: " + e.getMessage();
510      }
511    }
512  }
513
514  public String getCallVolumeSummary() {
515    try {
516      ObjectMapper om = new ObjectMapper();
517      return om.writeValueAsString(callCounts);
518    } catch (Exception e) {
519      return "Error: " + e.getMessage();
520    }
521  }
