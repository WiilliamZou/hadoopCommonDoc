087  /** A method invocation, including the method name and its parameters.*/
088  private static class Invocation implements Writable, Configurable {
089    private String methodName;
090    private Class<?>[] parameterClasses;
091    private Object[] parameters;
092    private Configuration conf;
093    private long clientVersion;
094    private int clientMethodsHash;
095    private String declaringClassProtocolName;
096    
097    //This could be different from static writableRpcVersion when received
098    //at server, if client is using a different version.
099    private long rpcVersion;
100
101    @SuppressWarnings("unused") // called when deserializing an invocation
102    public Invocation() {}
103
104    public Invocation(Method method, Object[] parameters) {
105      this.methodName = method.getName();
106      this.parameterClasses = method.getParameterTypes();
107      this.parameters = parameters;
108      rpcVersion = writableRpcVersion;
109      if (method.getDeclaringClass().equals(VersionedProtocol.class)) {
110        //VersionedProtocol is exempted from version check.
111        clientVersion = 0;
112        clientMethodsHash = 0;
113      } else {
114        this.clientVersion = RPC.getProtocolVersion(method.getDeclaringClass());
115        this.clientMethodsHash = ProtocolSignature.getFingerprint(method
116            .getDeclaringClass().getMethods());
117      }
118      this.declaringClassProtocolName = 
119          RPC.getProtocolName(method.getDeclaringClass());
120    }
121
122    /** The name of the method invoked. */
123    public String getMethodName() { return methodName; }
124
125    /** The parameter classes. */
126    public Class<?>[] getParameterClasses() { return parameterClasses; }
127
128    /** The parameter instances. */
129    public Object[] getParameters() { return parameters; }
130    
131    private long getProtocolVersion() {
132      return clientVersion;
133    }
134
135    @SuppressWarnings("unused")
136    private int getClientMethodsHash() {
137      return clientMethodsHash;
138    }
139    
140    /**
141     * Returns the rpc version used by the client.
142     * @return rpcVersion
143     */
144    public long getRpcVersion() {
145      return rpcVersion;
146    }
147
148    @Override
149    @SuppressWarnings("deprecation")
150    public void readFields(DataInput in) throws IOException {
151      rpcVersion = in.readLong();
152      declaringClassProtocolName = UTF8.readString(in);
153      methodName = UTF8.readString(in);
154      clientVersion = in.readLong();
155      clientMethodsHash = in.readInt();
156      parameters = new Object[in.readInt()];
157      parameterClasses = new Class[parameters.length];
158      ObjectWritable objectWritable = new ObjectWritable();
159      for (int i = 0; i < parameters.length; i++) {
160        parameters[i] = 
161            ObjectWritable.readObject(in, objectWritable, this.conf);
162        parameterClasses[i] = objectWritable.getDeclaredClass();
163      }
164    }
165
166    @Override
167    @SuppressWarnings("deprecation")
168    public void write(DataOutput out) throws IOException {
169      out.writeLong(rpcVersion);
170      UTF8.writeString(out, declaringClassProtocolName);
171      UTF8.writeString(out, methodName);
172      out.writeLong(clientVersion);
173      out.writeInt(clientMethodsHash);
174      out.writeInt(parameterClasses.length);
175      for (int i = 0; i < parameterClasses.length; i++) {
176        ObjectWritable.writeObject(out, parameters[i], parameterClasses[i],
177                                   conf, true);
178      }
179    }
180
181    @Override
182    public String toString() {
183      StringBuilder buffer = new StringBuilder();
184      buffer.append(methodName);
185      buffer.append("(");
186      for (int i = 0; i < parameters.length; i++) {
187        if (i != 0)
188          buffer.append(", ");
189        buffer.append(parameters[i]);
190      }
191      buffer.append(")");
192      buffer.append(", rpc version="+rpcVersion);
193      buffer.append(", client version="+clientVersion);
194      buffer.append(", methodsFingerPrint="+clientMethodsHash);
195      return buffer.toString();
196    }
197
198    @Override
199    public void setConf(Configuration conf) {
200      this.conf = conf;
201    }
202
203    @Override
204    public Configuration getConf() {
205      return this.conf;
206    }
207
208  }

212  private static class Invoker implements RpcInvocationHandler {
213    private Client.ConnectionId remoteId;
214    private Client client;
215    private boolean isClosed = false;
216    private final AtomicBoolean fallbackToSimpleAuth;
217
218    public Invoker(Class<?> protocol,
219                   InetSocketAddress address, UserGroupInformation ticket,
220                   Configuration conf, SocketFactory factory,
221                   int rpcTimeout, AtomicBoolean fallbackToSimpleAuth)
222        throws IOException {
223      this.remoteId = Client.ConnectionId.getConnectionId(address, protocol,
224          ticket, rpcTimeout, conf);
225      this.client = CLIENTS.getClient(conf, factory);
226      this.fallbackToSimpleAuth = fallbackToSimpleAuth;
227    }
228
229    @Override
230    public Object invoke(Object proxy, Method method, Object[] args)
231      throws Throwable {
232      long startTime = 0;
233      if (LOG.isDebugEnabled()) {
234        startTime = Time.now();
235      }
236      TraceScope traceScope = null;
237      if (Trace.isTracing()) {
238        traceScope = Trace.startSpan(RpcClientUtil.methodToTraceString(method));
239      }
240      ObjectWritable value;
241      try {
242        value = (ObjectWritable)
243          client.call(RPC.RpcKind.RPC_WRITABLE, new Invocation(method, args),
244            remoteId, fallbackToSimpleAuth);
245      } finally {
246        if (traceScope != null) traceScope.close();
247      }
248      if (LOG.isDebugEnabled()) {
249        long callTime = Time.now() - startTime;
250        LOG.debug("Call: " + method.getName() + " " + callTime);
251      }
252      return value.get();
253    }
254    
255    /* close the IPC client that's responsible for this invoker's RPCs */ 
256    @Override
257    synchronized public void close() {
258      if (!isClosed) {
259        isClosed = true;
260        CLIENTS.stopClient(client);
261      }
262    }
263
264    @Override
265    public ConnectionId getConnectionId() {
266      return remoteId;
267    }
268  }

278   * talking to a server at the named address. 
279   * @param <T>*/
280  @Override
281  public <T> ProtocolProxy<T> getProxy(Class<T> protocol, long clientVersion,
282                         InetSocketAddress addr, UserGroupInformation ticket,
283                         Configuration conf, SocketFactory factory,
284                         int rpcTimeout, RetryPolicy connectionRetryPolicy)
285    throws IOException {
286    return getProxy(protocol, clientVersion, addr, ticket, conf, factory,
287      rpcTimeout, connectionRetryPolicy, null);
288  }
289
290  /** Construct a client-side proxy object that implements the named protocol,
291   * talking to a server at the named address. 
292   * @param <T>*/
293  @Override
294  @SuppressWarnings("unchecked")
295  public <T> ProtocolProxy<T> getProxy(Class<T> protocol, long clientVersion,
296                         InetSocketAddress addr, UserGroupInformation ticket,
297                         Configuration conf, SocketFactory factory,
298                         int rpcTimeout, RetryPolicy connectionRetryPolicy,
299                         AtomicBoolean fallbackToSimpleAuth)
300    throws IOException {    
301
302    if (connectionRetryPolicy != null) {
303      throw new UnsupportedOperationException(
304          "Not supported: connectionRetryPolicy=" + connectionRetryPolicy);
305    }
306
307    T proxy = (T) Proxy.newProxyInstance(protocol.getClassLoader(),
308        new Class[] { protocol }, new Invoker(protocol, addr, ticket, conf,
309            factory, rpcTimeout, fallbackToSimpleAuth));
310    return new ProtocolProxy<T>(protocol, proxy, true);
311  }

361    /** 
362     * Construct an RPC server.
363     * @param protocolImpl the instance whose methods will be called
364     * @param conf the configuration to use
365     * @param bindAddress the address to bind on to listen for connection
366     * @param port the port to listen for connections on
367     * @param numHandlers the number of method handler threads to run
368     * @param verbose whether each call should be logged
369     * 
370     * @deprecated use Server#Server(Class, Object, 
371     *      Configuration, String, int, int, int, int, boolean, SecretManager)
372     */
373    @Deprecated
374    public Server(Object protocolImpl, Configuration conf, String bindAddress,
375        int port, int numHandlers, int numReaders, int queueSizePerHandler,
376        boolean verbose, SecretManager<? extends TokenIdentifier> secretManager) 
377            throws IOException {
378       this(null, protocolImpl,  conf,  bindAddress,   port,
379                   numHandlers,  numReaders,  queueSizePerHandler,  verbose, 
380                   secretManager, null);
381   
382    }
383    
384    /** 
385     * Construct an RPC server.
386     * @param protocolClass - the protocol being registered
387     *     can be null for compatibility with old usage (see below for details)
388     * @param protocolImpl the protocol impl that will be called
389     * @param conf the configuration to use
390     * @param bindAddress the address to bind on to listen for connection
391     * @param port the port to listen for connections on
392     * @param numHandlers the number of method handler threads to run
393     * @param verbose whether each call should be logged
394     */
395    public Server(Class<?> protocolClass, Object protocolImpl,
396        Configuration conf, String bindAddress,  int port,
397        int numHandlers, int numReaders, int queueSizePerHandler, 
398        boolean verbose, SecretManager<? extends TokenIdentifier> secretManager,
399        String portRangeConfig) 
400        throws IOException {
401      super(bindAddress, port, null, numHandlers, numReaders,
402          queueSizePerHandler, conf,
403          classNameBase(protocolImpl.getClass().getName()), secretManager,
404          portRangeConfig);
405
406      this.verbose = verbose;
407      
408      
409      Class<?>[] protocols;
410      if (protocolClass == null) { // derive protocol from impl
411        /*
412         * In order to remain compatible with the old usage where a single
413         * target protocolImpl is suppled for all protocol interfaces, and
414         * the protocolImpl is derived from the protocolClass(es) 
415         * we register all interfaces extended by the protocolImpl
416         */
417        protocols = RPC.getProtocolInterfaces(protocolImpl.getClass());
418
419      } else {
420        if (!protocolClass.isAssignableFrom(protocolImpl.getClass())) {
421          throw new IOException("protocolClass "+ protocolClass +
422              " is not implemented by protocolImpl which is of class " +
423              protocolImpl.getClass());
424        }
425        // register protocol class and its super interfaces
426        registerProtocolAndImpl(RPC.RpcKind.RPC_WRITABLE, protocolClass, protocolImpl);
427        protocols = RPC.getProtocolInterfaces(protocolClass);
428      }
429      for (Class<?> p : protocols) {
430        if (!p.equals(VersionedProtocol.class)) {
431          registerProtocolAndImpl(RPC.RpcKind.RPC_WRITABLE, p, protocolImpl);
432        }
433      }
434
435    }

437    private static void log(String value) {
438      if (value!= null && value.length() > 55)
439        value = value.substring(0, 55)+"...";
440      LOG.info(value);
441    }

443    static class WritableRpcInvoker implements RpcInvoker {
444
445     @Override
446      public Writable call(org.apache.hadoop.ipc.RPC.Server server,
447          String protocolName, Writable rpcRequest, long receivedTime)
448          throws IOException, RPC.VersionMismatch {
449
450        Invocation call = (Invocation)rpcRequest;
451        if (server.verbose) log("Call: " + call);
452
453        // Verify writable rpc version
454        if (call.getRpcVersion() != writableRpcVersion) {
455          // Client is using a different version of WritableRpc
456          throw new RpcServerException(
457              "WritableRpc version mismatch, client side version="
458                  + call.getRpcVersion() + ", server side version="
459                  + writableRpcVersion);
460        }
461
462        long clientVersion = call.getProtocolVersion();
463        final String protoName;
464        ProtoClassProtoImpl protocolImpl;
465        if (call.declaringClassProtocolName.equals(VersionedProtocol.class.getName())) {
466          // VersionProtocol methods are often used by client to figure out
467          // which version of protocol to use.
468          //
469          // Versioned protocol methods should go the protocolName protocol
470          // rather than the declaring class of the method since the
471          // the declaring class is VersionedProtocol which is not 
472          // registered directly.
473          // Send the call to the highest  protocol version
474          VerProtocolImpl highest = server.getHighestSupportedProtocol(
475              RPC.RpcKind.RPC_WRITABLE, protocolName);
476          if (highest == null) {
477            throw new RpcServerException("Unknown protocol: " + protocolName);
478          }
479          protocolImpl = highest.protocolTarget;
480        } else {
481          protoName = call.declaringClassProtocolName;
482
483          // Find the right impl for the protocol based on client version.
484          ProtoNameVer pv = 
485              new ProtoNameVer(call.declaringClassProtocolName, clientVersion);
486          protocolImpl = 
487              server.getProtocolImplMap(RPC.RpcKind.RPC_WRITABLE).get(pv);
488          if (protocolImpl == null) { // no match for Protocol AND Version
489             VerProtocolImpl highest = 
490                 server.getHighestSupportedProtocol(RPC.RpcKind.RPC_WRITABLE, 
491                     protoName);
492            if (highest == null) {
493              throw new RpcServerException("Unknown protocol: " + protoName);
494            } else { // protocol supported but not the version that client wants
495              throw new RPC.VersionMismatch(protoName, clientVersion,
496                highest.version);
497            }
498          }
499        }
500          
501
502          // Invoke the protocol method
503       long startTime = Time.now();
504       int qTime = (int) (startTime-receivedTime);
505       Exception exception = null;
506       try {
507          Method method =
508              protocolImpl.protocolClass.getMethod(call.getMethodName(),
509              call.getParameterClasses());
510          method.setAccessible(true);
511          server.rpcDetailedMetrics.init(protocolImpl.protocolClass);
512          Object value = 
513              method.invoke(protocolImpl.protocolImpl, call.getParameters());
514          if (server.verbose) log("Return: "+value);
515          return new ObjectWritable(method.getReturnType(), value);
516
517        } catch (InvocationTargetException e) {
518          Throwable target = e.getTargetException();
519          if (target instanceof IOException) {
520            exception = (IOException)target;
521            throw (IOException)target;
522          } else {
523            IOException ioe = new IOException(target.toString());
524            ioe.setStackTrace(target.getStackTrace());
525            exception = ioe;
526            throw ioe;
527          }
528        } catch (Throwable e) {
529          if (!(e instanceof IOException)) {
530            LOG.error("Unexpected throwable object ", e);
531          }
532          IOException ioe = new IOException(e.toString());
533          ioe.setStackTrace(e.getStackTrace());
534          exception = ioe;
535          throw ioe;
536        } finally {
537         int processingTime = (int) (Time.now() - startTime);
538         if (LOG.isDebugEnabled()) {
539           String msg = "Served: " + call.getMethodName() +
540               " queueTime= " + qTime +
541               " procesingTime= " + processingTime;
542           if (exception != null) {
543             msg += " exception= " + exception.getClass().getSimpleName();
544           }
545           LOG.debug(msg);
546         }
547         String detailedMetricsName = (exception == null) ?
548             call.getMethodName() :
549             exception.getClass().getSimpleName();
550         server.rpcMetrics.addRpcQueueTime(qTime);
551         server.rpcMetrics.addRpcProcessingTime(processingTime);
552         server.rpcDetailedMetrics.addProcessingTime(detailedMetricsName,
553             processingTime);
554       }
555      }
556    }
