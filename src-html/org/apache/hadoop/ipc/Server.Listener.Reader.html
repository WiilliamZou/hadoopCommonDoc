150  static class ExceptionsHandler {
151    private volatile Set<String> terseExceptions = new HashSet<String>();
152
153    /**
154     * Add exception class so server won't log its stack trace.
155     * Modifying the terseException through this method is thread safe.
156     *
157     * @param exceptionClass exception classes 
158     */
159    void addTerseExceptions(Class<?>... exceptionClass) {
160
161      // Make a copy of terseException for performing modification
162      final HashSet<String> newSet = new HashSet<String>(terseExceptions);
163
164      // Add all class names into the HashSet
165      for (Class<?> name : exceptionClass) {
166        newSet.add(name.toString());
167      }
168      // Replace terseException set
169      terseExceptions = Collections.unmodifiableSet(newSet);
170    }
171
172    boolean isTerse(Class<?> t) {
173      return terseExceptions.contains(t.toString());
174    }

400  public static void bind(ServerSocket socket, InetSocketAddress address, 
401      int backlog, Configuration conf, String rangeConf) throws IOException {
402    try {
403      IntegerRanges range = null;
404      if (rangeConf != null) {
405        range = conf.getRange(rangeConf, "");
406      }
407      if (range == null || range.isEmpty() || (address.getPort() != 0)) {
408        socket.bind(address, backlog);
409      } else {
410        for (Integer port : range) {
411          if (socket.isBound()) break;
412          try {
413            InetSocketAddress temp = new InetSocketAddress(address.getAddress(),
414                port);
415            socket.bind(temp, backlog);
416          } catch(BindException e) {
417            //Ignored
418          }
419        }
420        if (!socket.isBound()) {
421          throw new BindException("Could not find a free port in "+range);
422        }
423      }
424    } catch (SocketException e) {
425      throw NetUtils.wrapException(null,
426          0,
427          address.getHostName(),
428          address.getPort(), e);
429    }
430  }

555  /** Listens on the socket. Creates jobs for the handler threads*/
556  private class Listener extends Thread {
557    
558    private ServerSocketChannel acceptChannel = null; //the accept channel
559    private Selector selector = null; //the selector that we use for the server
560    private Reader[] readers = null;
561    private int currentReader = 0;
562    private InetSocketAddress address; //the address we bind at
563    private int backlogLength = conf.getInt(
564        CommonConfigurationKeysPublic.IPC_SERVER_LISTEN_QUEUE_SIZE_KEY,
565        CommonConfigurationKeysPublic.IPC_SERVER_LISTEN_QUEUE_SIZE_DEFAULT);
566    
567    public Listener() throws IOException {
568      address = new InetSocketAddress(bindAddress, port);
569      // Create a new server socket and set to non blocking mode
570      acceptChannel = ServerSocketChannel.open();
571      acceptChannel.configureBlocking(false);
572
573      // Bind the server socket to the local host and port
574      bind(acceptChannel.socket(), address, backlogLength, conf, portRangeConfig);
575      port = acceptChannel.socket().getLocalPort(); //Could be an ephemeral port
576      // create a selector;
577      selector= Selector.open();
578      readers = new Reader[readThreads];
579      for (int i = 0; i < readThreads; i++) {
580        Reader reader = new Reader(
581            "Socket Reader #" + (i + 1) + " for port " + port);
582        readers[i] = reader;
583        reader.start();
584      }
585
586      // Register accepts on the server socket with the selector.
587      acceptChannel.register(selector, SelectionKey.OP_ACCEPT);
588      this.setName("IPC Server listener on " + port);
589      this.setDaemon(true);
590    }
591    
592    private class Reader extends Thread {
593      final private BlockingQueue<Connection> pendingConnections;
594      private final Selector readSelector;
595
596      Reader(String name) throws IOException {
597        super(name);
598
599        this.pendingConnections =
600            new LinkedBlockingQueue<Connection>(readerPendingConnectionQueue);
601        this.readSelector = Selector.open();
602      }
603      
604      @Override
605      public void run() {
606        LOG.info("Starting " + Thread.currentThread().getName());
607        try {
608          doRunLoop();
609        } finally {
610          try {
611            readSelector.close();
612          } catch (IOException ioe) {
613            LOG.error("Error closing read selector in " + Thread.currentThread().getName(), ioe);
614          }
615        }
616      }
617
618      private synchronized void doRunLoop() {
619        while (running) {
620          SelectionKey key = null;
621          try {
622            // consume as many connections as currently queued to avoid
623            // unbridled acceptance of connections that starves the select
624            int size = pendingConnections.size();
625            for (int i=size; i>0; i--) {
626              Connection conn = pendingConnections.take();
627              conn.channel.register(readSelector, SelectionKey.OP_READ, conn);
628            }
629            readSelector.select();
630
631            Iterator<SelectionKey> iter = readSelector.selectedKeys().iterator();
632            while (iter.hasNext()) {
633              key = iter.next();
634              iter.remove();
635              if (key.isValid()) {
636                if (key.isReadable()) {
637                  doRead(key);
638                }
639              }
640              key = null;
641            }
642          } catch (InterruptedException e) {
643            if (running) {                      // unexpected -- log it
644              LOG.info(Thread.currentThread().getName() + " unexpectedly interrupted", e);
645            }
646          } catch (IOException ex) {
647            LOG.error("Error in Reader", ex);
648          }
649        }
650      }
651
652      /**
653       * Updating the readSelector while it's being used is not thread-safe,
654       * so the connection must be queued.  The reader will drain the queue
655       * and update its readSelector before performing the next select
656       */
657      public void addConnection(Connection conn) throws InterruptedException {
658        pendingConnections.put(conn);
659        readSelector.wakeup();
660      }
661
662      void shutdown() {
663        assert !running;
664        readSelector.wakeup();
665        try {
666          super.interrupt();
667          super.join();

698          // log the event and sleep for a minute and give 
699          // some thread(s) a chance to finish

701          closeCurrentConnection(key, e);

703          try { Thread.sleep(60000); } catch (Exception ie) {}
704        } catch (Exception e) {

713          selector.close();
714        } catch (IOException e) { }
715

717        acceptChannel= null;
718

762    void doRead(SelectionKey key) throws InterruptedException {
763      int count = 0;
764      Connection c = (Connection)key.attachment();
765      if (c == null) {
766        return;  
767      }
768      c.setLastContact(Time.now());
769      
770      try {
771        count = c.readAndProcess();
772      } catch (InterruptedException ieo) {
773        LOG.info(Thread.currentThread().getName() + ": readAndProcess caught InterruptedException", ieo);
774        throw ieo;
775      } catch (Exception e) {
776        // a WrappedRpcServerException is an exception that has been sent
777        // to the client, so the stacktrace is unnecessary; any other
778        // exceptions are unexpected internal server errors and thus the
779        // stacktrace should be logged
780        LOG.info(Thread.currentThread().getName() + ": readAndProcess from client " +
781            c.getHostAddress() + " threw exception [" + e + "]",
782            (e instanceof WrappedRpcServerException) ? null : e);
783        count = -1; //so that the (count < 0) block is executed
784      }
785      if (count < 0) {
786        closeConnection(c);
787        c = null;
788      }
789      else {
790        c.setLastContact(Time.now());
791      }
792    }

837      SERVER.set(Server.this);
838      try {

841        LOG.info("Stopping " + Thread.currentThread().getName());

850    private void doRunLoop() {
851      long lastPurgeTime = 0;   // last check for old calls.
852
853      while (running) {
854        try {
855          waitPending();     // If a channel is being registered, wait.
856          writeSelector.select(PURGE_INTERVAL);
857          Iterator<SelectionKey> iter = writeSelector.selectedKeys().iterator();
858          while (iter.hasNext()) {
859            SelectionKey key = iter.next();
860            iter.remove();
861            try {
862              if (key.isValid() && key.isWritable()) {
863                  doAsyncWrite(key);
864              }
865            } catch (IOException e) {
866              LOG.info(Thread.currentThread().getName() + ": doAsyncWrite threw exception " + e);
867            }
868          }
869          long now = Time.now();
870          if (now < lastPurgeTime + PURGE_INTERVAL) {
871            continue;
872          }
873          lastPurgeTime = now;
874          //
875          // If there were some calls that have not been sent out for a
876          // long time, discard them.
877          //
878          if(LOG.isDebugEnabled()) {
879            LOG.debug("Checking for old call responses.");
880          }
881          ArrayList<Call> calls;
882          
883          // get the list of channels from list of keys.
884          synchronized (writeSelector.keys()) {
885            calls = new ArrayList<Call>(writeSelector.keys().size());
886            iter = writeSelector.keys().iterator();
887            while (iter.hasNext()) {
888              SelectionKey key = iter.next();
889              Call call = (Call)key.attachment();
890              if (call != null && key.channel() == call.connection.channel) { 
891                calls.add(call);
892              }
893            }
894          }
895          
896          for(Call call : calls) {
897            doPurge(call, now);
898          }
899        } catch (OutOfMemoryError e) {
900          //
901          // we can run out of memory if we have too many threads
902          // log the event and sleep for a minute and give
903          // some thread(s) a chance to finish
904          //
905          LOG.warn("Out of Memory in server select", e);
906          try { Thread.sleep(60000); } catch (Exception ie) {}
907        } catch (Exception e) {
908          LOG.warn("Exception in Responder", e);
909        }
910      }
911    }
912
913    private void doAsyncWrite(SelectionKey key) throws IOException {
914      Call call = (Call)key.attachment();
915      if (call == null) {
916        return;
917      }
918      if (key.channel() != call.connection.channel) {
919        throw new IOException("doAsyncWrite: bad channel");
920      }
921
922      synchronized(call.connection.responseQueue) {
923        if (processResponse(call.connection.responseQueue, false)) {
924          try {
925            key.interestOps(0);
926          } catch (CancelledKeyException e) {
927            /* The Listener/reader might have closed the socket.
928             * We don't explicitly cancel the key, so not sure if this will
929             * ever fire.
930             * This warning could be removed.
931             */
932            LOG.warn("Exception while changing ops : " + e);
933          }
934        }
935      }
936    }

938    //
939    // Remove calls that have been pending in the responseQueue 
940    // for a long time.
941    //
942    private void doPurge(Call call, long now) {
943      LinkedList<Call> responseQueue = call.connection.responseQueue;
944      synchronized (responseQueue) {
945        Iterator<Call> iter = responseQueue.listIterator(0);
946        while (iter.hasNext()) {
947          call = iter.next();
948          if (now > call.timestamp + PURGE_INTERVAL) {
949            closeConnection(call.connection);
950            break;
951          }
952        }
953      }
954    }

956    // Processes one response. Returns true if there are no more pending
957    // data for this channel.
958    //
959    private boolean processResponse(LinkedList<Call> responseQueue,
960                                    boolean inHandler) throws IOException {
961      boolean error = true;
962      boolean done = false;       // there is more data for this channel.
963      int numElements = 0;
964      Call call = null;
965      try {
966        synchronized (responseQueue) {
967          //
968          // If there are no items for this channel, then we are done
969          //
970          numElements = responseQueue.size();
971          if (numElements == 0) {
972            error = false;
973            return true;              // no more data for this channel.
974          }
975          //
976          // Extract the first call
977          //
978          call = responseQueue.removeFirst();
979          SocketChannel channel = call.connection.channel;
980          if (LOG.isDebugEnabled()) {
981            LOG.debug(Thread.currentThread().getName() + ": responding to " + call);
982          }
983          //
984          // Send as much data as we can in the non-blocking fashion
985          //
986          int numBytes = channelWrite(channel, call.rpcResponse);
987          if (numBytes < 0) {
988            return true;
989          }
990          if (!call.rpcResponse.hasRemaining()) {
991            //Clear out the response buffer so it can be collected
992            call.rpcResponse = null;
993            call.connection.decRpcCount();
994            if (numElements == 1) {    // last call fully processes.
995              done = true;             // no more data for this channel.
996            } else {
997              done = false;            // more calls pending to be sent.
998            }
999            if (LOG.isDebugEnabled()) {
1000              LOG.debug(Thread.currentThread().getName() + ": responding to " + call
1001                  + " Wrote " + numBytes + " bytes.");
1002            }
1003          } else {
1004            //
1005            // If we were unable to write the entire response out, then 
1006            // insert in Selector queue. 
1007            //
1008            call.connection.responseQueue.addFirst(call);
1009            
1010            if (inHandler) {
1011              // set the serve time when the response has to be sent later
1012              call.timestamp = Time.now();
1013              
1014              incPending();
1015              try {
1016                // Wakeup the thread blocked on select, only then can the call 
1017                // to channel.register() complete.
1018                writeSelector.wakeup();
1019                channel.register(writeSelector, SelectionKey.OP_WRITE, call);
1020              } catch (ClosedChannelException e) {
1021                //Its ok. channel might be closed else where.
1022                done = true;
1023              } finally {
1024                decPending();
1025              }
1026            }
1027            if (LOG.isDebugEnabled()) {
1028              LOG.debug(Thread.currentThread().getName() + ": responding to " + call
1029                  + " Wrote partial " + numBytes + " bytes.");
1030            }
1031          }
1032          error = false;              // everything went off well
1033        }
1034      } finally {
1035        if (error && call != null) {
1036          LOG.warn(Thread.currentThread().getName()+", call " + call + ": output error");
1037          done = true;               // error. no more data for this channel.
1038          closeConnection(call.connection);
1039        }
1040      }
1041      return done;
1042    }

1044    //
1045    // Enqueue a response from the application.
1046    //
1047    void doRespond(Call call) throws IOException {
1048      synchronized (call.connection.responseQueue) {
1049        call.connection.responseQueue.addLast(call);
1050        if (call.connection.responseQueue.size() == 1) {
1051          processResponse(call.connection.responseQueue, true);
1052        }
1053      }
1054    }

1056    private synchronized void incPending() {   // call waiting to be enqueued.
1057      pending++;
1058    }
1059
1060    private synchronized void decPending() { // call done enqueueing.
1061      pending--;
1062      notify();
1063    }

1066      while (pending > 0) {
1067        wait();
1068      }

1287    private void saslProcess(RpcSaslProto saslMessage)
1288        throws WrappedRpcServerException, IOException, InterruptedException {
1289      if (saslContextEstablished) {
1290        throw new WrappedRpcServerException(
1291            RpcErrorCodeProto.FATAL_INVALID_RPC_HEADER,
1292            new SaslException("Negotiation is already complete"));
1293      }
1294      RpcSaslProto saslResponse = null;
1295      try {
1296        try {
1297          saslResponse = processSaslMessage(saslMessage);
1298        } catch (IOException e) {
1299          rpcMetrics.incrAuthenticationFailures();
1300          // attempting user could be null
1301          AUDITLOG.warn(AUTH_FAILED_FOR + this.toString() + ":"
1302              + attemptingUser + " (" + e.getLocalizedMessage() + ")");
1303          throw (IOException) getCauseForInvalidToken(e);
1304        }
1305        
1306        if (saslServer != null && saslServer.isComplete()) {
1307          if (LOG.isDebugEnabled()) {
1308            LOG.debug("SASL server context established. Negotiated QoP is "
1309                + saslServer.getNegotiatedProperty(Sasl.QOP));
1310          }
1311          user = getAuthorizedUgi(saslServer.getAuthorizationID());
1312          if (LOG.isDebugEnabled()) {
1313            LOG.debug("SASL server successfully authenticated client: " + user);
1314          }
1315          rpcMetrics.incrAuthenticationSuccesses();
1316          AUDITLOG.info(AUTH_SUCCESSFUL_FOR + user);
1317          saslContextEstablished = true;
1318        }
1319      } catch (WrappedRpcServerException wrse) { // don't re-wrap
1320        throw wrse;
1321      } catch (IOException ioe) {
1322        throw new WrappedRpcServerException(
1323            RpcErrorCodeProto.FATAL_UNAUTHORIZED, ioe);
1324      }
1325      // send back response if any, may throw IOException
1326      if (saslResponse != null) {
1327        doSaslReply(saslResponse);
1328      }
1329      // do NOT enable wrapping until the last auth response is sent
1330      if (saslContextEstablished) {
1331        String qop = (String) saslServer.getNegotiatedProperty(Sasl.QOP);
1332        // SASL wrapping is only used if the connection has a QOP, and
1333        // the value is not auth.  ex. auth-int & auth-priv
1334        useWrap = (qop != null && !"auth".equalsIgnoreCase(qop));        
1335      }
1336    }

1338    private RpcSaslProto processSaslMessage(RpcSaslProto saslMessage)
1339        throws IOException, InterruptedException {
1340      final RpcSaslProto saslResponse;
1341      final SaslState state = saslMessage.getState(); // required      
1342      switch (state) {
1343        case NEGOTIATE: {
1344          if (sentNegotiate) {
1345            throw new AccessControlException(
1346                "Client already attempted negotiation");
1347          }
1348          saslResponse = buildSaslNegotiateResponse();
1349          // simple-only server negotiate response is success which client
1350          // interprets as switch to simple
1351          if (saslResponse.getState() == SaslState.SUCCESS) {
1352            switchToSimple();
1353          }
1354          break;
1355        }
1356        case INITIATE: {
1357          if (saslMessage.getAuthsCount() != 1) {
1358            throw new SaslException("Client mechanism is malformed");
1359          }
1360          // verify the client requested an advertised authType
1361          SaslAuth clientSaslAuth = saslMessage.getAuths(0);
1362          if (!negotiateResponse.getAuthsList().contains(clientSaslAuth)) {
1363            if (sentNegotiate) {
1364              throw new AccessControlException(
1365                  clientSaslAuth.getMethod() + " authentication is not enabled."
1366                      + "  Available:" + enabledAuthMethods);
1367            }
1368            saslResponse = buildSaslNegotiateResponse();
1369            break;
1370          }
1371          authMethod = AuthMethod.valueOf(clientSaslAuth.getMethod());
1372          // abort SASL for SIMPLE auth, server has already ensured that
1373          // SIMPLE is a legit option above.  we will send no response
1374          if (authMethod == AuthMethod.SIMPLE) {
1375            switchToSimple();
1376            saslResponse = null;
1377            break;
1378          }
1379          // sasl server for tokens may already be instantiated
1380          if (saslServer == null || authMethod != AuthMethod.TOKEN) {
1381            saslServer = createSaslServer(authMethod);
1382          }
1383          saslResponse = processSaslToken(saslMessage);
1384          break;
1385        }
1386        case RESPONSE: {
1387          saslResponse = processSaslToken(saslMessage);
1388          break;
1389        }
1390        default:
1391          throw new SaslException("Client sent unsupported state " + state);
1392      }
1393      return saslResponse;
1394    }

1396    private RpcSaslProto processSaslToken(RpcSaslProto saslMessage)
1397        throws SaslException {
1398      if (!saslMessage.hasToken()) {
1399        throw new SaslException("Client did not send a token");
1400      }
1401      byte[] saslToken = saslMessage.getToken().toByteArray();
1402      if (LOG.isDebugEnabled()) {

1404            + " for processing by saslServer.evaluateResponse()");

1406      saslToken = saslServer.evaluateResponse(saslToken);
1407      return buildSaslResponse(
1408          saslServer.isComplete() ? SaslState.SUCCESS : SaslState.CHALLENGE,
1409          saslToken);
1410    }

1473    public int readAndProcess()
1474        throws WrappedRpcServerException, IOException, InterruptedException {
1475      while (true) {
1476        /* Read at most one RPC. If the header is not read completely yet
1477         * then iterate until we read first RPC or until there is no data left.
1478         */    
1479        int count = -1;
1480        if (dataLengthBuffer.remaining() > 0) {
1481          count = channelRead(channel, dataLengthBuffer);       
1482          if (count < 0 || dataLengthBuffer.remaining() > 0) 
1483            return count;
1484        }
1485        
1486        if (!connectionHeaderRead) {
1487          //Every connection is expected to send the header.
1488          if (connectionHeaderBuf == null) {
1489            connectionHeaderBuf = ByteBuffer.allocate(3);
1490          }
1491          count = channelRead(channel, connectionHeaderBuf);
1492          if (count < 0 || connectionHeaderBuf.remaining() > 0) {
1493            return count;
1494          }
1495          int version = connectionHeaderBuf.get(0);
1496          // TODO we should add handler for service class later
1497          this.setServiceClass(connectionHeaderBuf.get(1));
1498          dataLengthBuffer.flip();
1499          
1500          // Check if it looks like the user is hitting an IPC port
1501          // with an HTTP GET - this is a common error, so we can
1502          // send back a simple string indicating as much.
1503          if (HTTP_GET_BYTES.equals(dataLengthBuffer)) {
1504            setupHttpRequestOnIpcPortResponse();
1505            return -1;
1506          }
1507          
1508          if (!RpcConstants.HEADER.equals(dataLengthBuffer)
1509              || version != CURRENT_VERSION) {
1510            //Warning is ok since this is not supposed to happen.
1511            LOG.warn("Incorrect header or version mismatch from " + 
1512                     hostAddress + ":" + remotePort +
1513                     " got version " + version + 
1514                     " expected version " + CURRENT_VERSION);
1515            setupBadVersionResponse(version);
1516            return -1;
1517          }
1518          
1519          // this may switch us into SIMPLE
1520          authProtocol = initializeAuthContext(connectionHeaderBuf.get(2));          
1521          
1522          dataLengthBuffer.clear();
1523          connectionHeaderBuf = null;
1524          connectionHeaderRead = true;
1525          continue;
1526        }
1527        
1528        if (data == null) {
1529          dataLengthBuffer.flip();
1530          dataLength = dataLengthBuffer.getInt();
1531          checkDataLength(dataLength);
1532          data = ByteBuffer.allocate(dataLength);
1533        }
1534        
1535        count = channelRead(channel, data);
1536        
1537        if (data.remaining() == 0) {
1538          dataLengthBuffer.clear();
1539          data.flip();
1540          boolean isHeaderRead = connectionContextRead;
1541          processOneRpc(data.array());
1542          data = null;
1543          if (!isHeaderRead) {
1544            continue;
1545          }
1546        } 
1547        return count;
1548      }
1549    }

1551    private AuthProtocol initializeAuthContext(int authType)
1552        throws IOException {
1553      AuthProtocol authProtocol = AuthProtocol.valueOf(authType);
1554      if (authProtocol == null) {
1555        IOException ioe = new IpcException("Unknown auth protocol:" + authType);
1556        doSaslReply(ioe);
1557        throw ioe;        
1558      }
1559      boolean isSimpleEnabled = enabledAuthMethods.contains(AuthMethod.SIMPLE);
1560      switch (authProtocol) {
1561        case NONE: {
1562          // don't reply if client is simple and server is insecure
1563          if (!isSimpleEnabled) {
1564            IOException ioe = new AccessControlException(
1565                "SIMPLE authentication is not enabled."
1566                    + "  Available:" + enabledAuthMethods);
1567            doSaslReply(ioe);
1568            throw ioe;
1569          }
1570          break;
1571        }
1572        default: {
1573          break;
1574        }
1575      }
1576      return authProtocol;
1577    }

1579    private RpcSaslProto buildSaslNegotiateResponse()
1580        throws IOException, InterruptedException {
1581      RpcSaslProto negotiateMessage = negotiateResponse;
1582      // accelerate token negotiation by sending initial challenge
1583      // in the negotiation response
1584      if (enabledAuthMethods.contains(AuthMethod.TOKEN)) {
1585        saslServer = createSaslServer(AuthMethod.TOKEN);
1586        byte[] challenge = saslServer.evaluateResponse(new byte[0]);
1587        RpcSaslProto.Builder negotiateBuilder =
1588            RpcSaslProto.newBuilder(negotiateResponse);
1589        negotiateBuilder.getAuthsBuilder(0)  // TOKEN is always first
1590            .setChallenge(ByteString.copyFrom(challenge));
1591        negotiateMessage = negotiateBuilder.build();
1592      }
1593      sentNegotiate = true;
1594      return negotiateMessage;
1595    }

1605     * Try to set up the response to indicate that the client version
1606     * is incompatible with the server. This can contain special-case
1607     * code to speak enough of past IPC protocols to pass back
1608     * an exception to the caller.
1609     * @param clientVersion the version the caller is using 
1610     * @throws IOException
1611     */
1612    private void setupBadVersionResponse(int clientVersion) throws IOException {
1613      String errMsg = "Server IPC version " + CURRENT_VERSION +
1614      " cannot communicate with client version " + clientVersion;
1615      ByteArrayOutputStream buffer = new ByteArrayOutputStream();
1616      
1617      if (clientVersion >= 9) {
1618        // Versions >>9  understand the normal response
1619        Call fakeCall = new Call(-1, RpcConstants.INVALID_RETRY_COUNT, null,
1620            this);
1621        setupResponse(buffer, fakeCall, 
1622            RpcStatusProto.FATAL, RpcErrorCodeProto.FATAL_VERSION_MISMATCH,
1623            null, VersionMismatch.class.getName(), errMsg);
1624        responder.doRespond(fakeCall);
1625      } else if (clientVersion >= 3) {
1626        Call fakeCall = new Call(-1, RpcConstants.INVALID_RETRY_COUNT, null,
1627            this);
1628        // Versions 3 to 8 use older response
1629        setupResponseOldVersionFatal(buffer, fakeCall,
1630            null, VersionMismatch.class.getName(), errMsg);
1631
1632        responder.doRespond(fakeCall);
1633      } else if (clientVersion == 2) { // Hadoop 0.18.3
1634        Call fakeCall = new Call(0, RpcConstants.INVALID_RETRY_COUNT, null,
1635            this);
1636        DataOutputStream out = new DataOutputStream(buffer);
1637        out.writeInt(0); // call ID
1638        out.writeBoolean(true); // error
1639        WritableUtils.writeString(out, VersionMismatch.class.getName());
1640        WritableUtils.writeString(out, errMsg);
1641        fakeCall.setResponse(ByteBuffer.wrap(buffer.toByteArray()));
1642        
1643        responder.doRespond(fakeCall);
1644      }
1645    }

1654    /** Reads the connection context following the connection header
1655     * @param dis - DataInputStream from which to read the header 
1656     * @throws WrappedRpcServerException - if the header cannot be
1657     *         deserialized, or the user is not authorized
1658     */ 
1659    private void processConnectionContext(DataInputStream dis)
1660        throws WrappedRpcServerException {
1661      // allow only one connection context during a session
1662      if (connectionContextRead) {
1663        throw new WrappedRpcServerException(
1664            RpcErrorCodeProto.FATAL_INVALID_RPC_HEADER,
1665            "Connection context already processed");
1666      }
1667      connectionContext = decodeProtobufFromStream(
1668          IpcConnectionContextProto.newBuilder(), dis);
1669      protocolName = connectionContext.hasProtocol() ? connectionContext
1670          .getProtocol() : null;
1671
1672      UserGroupInformation protocolUser = ProtoUtil.getUgi(connectionContext);
1673      if (saslServer == null) {
1674        user = protocolUser;
1675      } else {
1676        // user is authenticated
1677        user.setAuthenticationMethod(authMethod);
1678        //Now we check if this is a proxy user case. If the protocol user is
1679        //different from the 'user', it is a proxy user scenario. However, 
1680        //this is not allowed if user authenticated with DIGEST.
1681        if ((protocolUser != null)
1682            && (!protocolUser.getUserName().equals(user.getUserName()))) {
1683          if (authMethod == AuthMethod.TOKEN) {
1684            // Not allowed to doAs if token authentication is used
1685            throw new WrappedRpcServerException(
1686                RpcErrorCodeProto.FATAL_UNAUTHORIZED,
1687                new AccessControlException("Authenticated user (" + user
1688                    + ") doesn't match what the client claims to be ("
1689                    + protocolUser + ")"));
1690          } else {
1691            // Effective user can be different from authenticated user
1692            // for simple auth or kerberos auth
1693            // The user is the real user. Now we create a proxy user
1694            UserGroupInformation realUser = user;
1695            user = UserGroupInformation.createProxyUser(protocolUser
1696                .getUserName(), realUser);
1697          }
1698        }
1699      }
1700      authorizeConnection();
1701      // don't set until after authz because connection isn't established
1702      connectionContextRead = true;
1703    }

1705    /**
1706     * Process a wrapped RPC Request - unwrap the SASL packet and process
1707     * each embedded RPC request 
1708     * @param buf - SASL wrapped request of one or more RPCs
1709     * @throws IOException - SASL packet cannot be unwrapped
1710     * @throws InterruptedException
1711     */    
1712    private void unwrapPacketAndProcessRpcs(byte[] inBuf)
1713        throws WrappedRpcServerException, IOException, InterruptedException {
1714      if (LOG.isDebugEnabled()) {
1715        LOG.debug("Have read input token of size " + inBuf.length
1716            + " for processing by saslServer.unwrap()");
1717      }
1718      inBuf = saslServer.unwrap(inBuf, 0, inBuf.length);
1719      ReadableByteChannel ch = Channels.newChannel(new ByteArrayInputStream(
1720          inBuf));
1721      // Read all RPCs contained in the inBuf, even partial ones
1722      while (true) {
1723        int count = -1;
1724        if (unwrappedDataLengthBuffer.remaining() > 0) {
1725          count = channelRead(ch, unwrappedDataLengthBuffer);
1726          if (count <= 0 || unwrappedDataLengthBuffer.remaining() > 0)
1727            return;
1728        }
1729
1730        if (unwrappedData == null) {
1731          unwrappedDataLengthBuffer.flip();
1732          int unwrappedDataLength = unwrappedDataLengthBuffer.getInt();
1733          unwrappedData = ByteBuffer.allocate(unwrappedDataLength);
1734        }
1735
1736        count = channelRead(ch, unwrappedData);
1737        if (count <= 0 || unwrappedData.remaining() > 0)
1738          return;
1739
1740        if (unwrappedData.remaining() == 0) {
1741          unwrappedDataLengthBuffer.clear();
1742          unwrappedData.flip();
1743          processOneRpc(unwrappedData.array());
1744          unwrappedData = null;
1745        }
1746      }
1747    }

1749    /**
1750     * Process an RPC Request - handle connection setup and decoding of
1751     * request into a Call
1752     * @param buf - contains the RPC request header and the rpc request
1753     * @throws IOException - internal error that should not be returned to
1754     *         client, typically failure to respond to client
1755     * @throws WrappedRpcServerException - an exception to be sent back to
1756     *         the client that does not require verbose logging by the
1757     *         Listener thread
1758     * @throws InterruptedException
1759     */    
1760    private void processOneRpc(byte[] buf)
1761        throws IOException, WrappedRpcServerException, InterruptedException {
1762      int callId = -1;
1763      int retry = RpcConstants.INVALID_RETRY_COUNT;
1764      try {
1765        final DataInputStream dis =
1766            new DataInputStream(new ByteArrayInputStream(buf));
1767        final RpcRequestHeaderProto header =
1768            decodeProtobufFromStream(RpcRequestHeaderProto.newBuilder(), dis);
1769        callId = header.getCallId();
1770        retry = header.getRetryCount();
1771        if (LOG.isDebugEnabled()) {
1772          LOG.debug(" got #" + callId);
1773        }
1774        checkRpcHeaders(header);
1775        
1776        if (callId < 0) { // callIds typically used during connection setup
1777          processRpcOutOfBandRequest(header, dis);
1778        } else if (!connectionContextRead) {
1779          throw new WrappedRpcServerException(
1780              RpcErrorCodeProto.FATAL_INVALID_RPC_HEADER,
1781              "Connection context not established");
1782        } else {
1783          processRpcRequest(header, dis);
1784        }
1785      } catch (WrappedRpcServerException wrse) { // inform client of error
1786        Throwable ioe = wrse.getCause();
1787        final Call call = new Call(callId, retry, null, this);
1788        setupResponse(authFailedResponse, call,
1789            RpcStatusProto.FATAL, wrse.getRpcErrorCodeProto(), null,
1790            ioe.getClass().getName(), ioe.getMessage());
1791        responder.doRespond(call);
1792        throw wrse;
1793      }
1794    }

1796    /**
1797     * Verify RPC header is valid
1798     * @param header - RPC request header
1799     * @throws WrappedRpcServerException - header contains invalid values 
1800     */
1801    private void checkRpcHeaders(RpcRequestHeaderProto header)
1802        throws WrappedRpcServerException {
1803      if (!header.hasRpcOp()) {
1804        String err = " IPC Server: No rpc op in rpcRequestHeader";
1805        throw new WrappedRpcServerException(
1806            RpcErrorCodeProto.FATAL_INVALID_RPC_HEADER, err);
1807      }
1808      if (header.getRpcOp() != 
1809          RpcRequestHeaderProto.OperationProto.RPC_FINAL_PACKET) {
1810        String err = "IPC Server does not implement rpc header operation" + 
1811                header.getRpcOp();
1812        throw new WrappedRpcServerException(
1813            RpcErrorCodeProto.FATAL_INVALID_RPC_HEADER, err);
1814      }
1815      // If we know the rpc kind, get its class so that we can deserialize
1816      // (Note it would make more sense to have the handler deserialize but 
1817      // we continue with this original design.
1818      if (!header.hasRpcKind()) {
1819        String err = " IPC Server: No rpc kind in rpcRequestHeader";
1820        throw new WrappedRpcServerException(
1821            RpcErrorCodeProto.FATAL_INVALID_RPC_HEADER, err);
1822      }
1823    }

1825    /**
1826     * Process an RPC Request - the connection headers and context must
1827     * have been already read
1828     * @param header - RPC request header
1829     * @param dis - stream to request payload
1830     * @throws WrappedRpcServerException - due to fatal rpc layer issues such
1831     *   as invalid header or deserialization error. In this case a RPC fatal
1832     *   status response will later be sent back to client.
1833     * @throws InterruptedException
1834     */
1835    private void processRpcRequest(RpcRequestHeaderProto header,
1836        DataInputStream dis) throws WrappedRpcServerException,
1837        InterruptedException {
1838      Class<? extends Writable> rpcRequestClass = 
1839          getRpcRequestWrapper(header.getRpcKind());
1840      if (rpcRequestClass == null) {
1841        LOG.warn("Unknown rpc kind "  + header.getRpcKind() + 
1842            " from client " + getHostAddress());
1843        final String err = "Unknown rpc kind in rpc header"  + 
1844            header.getRpcKind();
1845        throw new WrappedRpcServerException(
1846            RpcErrorCodeProto.FATAL_INVALID_RPC_HEADER, err);   
1847      }
1848      Writable rpcRequest;
1849      try { //Read the rpc request
1850        rpcRequest = ReflectionUtils.newInstance(rpcRequestClass, conf);
1851        rpcRequest.readFields(dis);
1852      } catch (Throwable t) { // includes runtime exception from newInstance
1853        LOG.warn("Unable to read call parameters for client " +
1854                 getHostAddress() + "on connection protocol " +
1855            this.protocolName + " for rpcKind " + header.getRpcKind(),  t);
1856        String err = "IPC server unable to read call parameters: "+ t.getMessage();
1857        throw new WrappedRpcServerException(
1858            RpcErrorCodeProto.FATAL_DESERIALIZING_REQUEST, err);
1859      }
1860        
1861      Span traceSpan = null;
1862      if (header.hasTraceInfo()) {
1863        // If the incoming RPC included tracing info, always continue the trace
1864        TraceInfo parentSpan = new TraceInfo(header.getTraceInfo().getTraceId(),
1865                                             header.getTraceInfo().getParentId());
1866        traceSpan = Trace.startSpan(rpcRequest.toString(), parentSpan).detach();
1867      }
1868
1869      Call call = new Call(header.getCallId(), header.getRetryCount(),
1870          rpcRequest, this, ProtoUtil.convert(header.getRpcKind()),
1871          header.getClientId().toByteArray(), traceSpan);
1872
1873      callQueue.put(call);              // queue the call; maybe blocked here
1874      incRpcCount();  // Increment the rpc count
1875    }

1878    /**
1879     * Establish RPC connection setup by negotiating SASL if required, then
1880     * reading and authorizing the connection header
1881     * @param header - RPC header
1882     * @param dis - stream to request payload
1883     * @throws WrappedRpcServerException - setup failed due to SASL
1884     *         negotiation failure, premature or invalid connection context,
1885     *         or other state errors 
1886     * @throws IOException - failed to send a response back to the client
1887     * @throws InterruptedException
1888     */
1889    private void processRpcOutOfBandRequest(RpcRequestHeaderProto header,
1890        DataInputStream dis) throws WrappedRpcServerException, IOException,
1891        InterruptedException {
1892      final int callId = header.getCallId();
1893      if (callId == CONNECTION_CONTEXT_CALL_ID) {
1894        // SASL must be established prior to connection context
1895        if (authProtocol == AuthProtocol.SASL && !saslContextEstablished) {
1896          throw new WrappedRpcServerException(
1897              RpcErrorCodeProto.FATAL_INVALID_RPC_HEADER,
1898              "Connection header sent during SASL negotiation");
1899        }
1900        // read and authorize the user
1901        processConnectionContext(dis);
1902      } else if (callId == AuthProtocol.SASL.callId) {
1903        // if client was switched to simple, ignore first SASL message
1904        if (authProtocol != AuthProtocol.SASL) {
1905          throw new WrappedRpcServerException(
1906              RpcErrorCodeProto.FATAL_INVALID_RPC_HEADER,
1907              "SASL protocol not requested by client");
1908        }
1909        saslReadAndProcess(dis);
1910      } else if (callId == PING_CALL_ID) {
1911        LOG.debug("Received ping message");
1912      } else {
1913        throw new WrappedRpcServerException(
1914            RpcErrorCodeProto.FATAL_INVALID_RPC_HEADER,
1915            "Unknown out of band call #" + callId);
1916      }
1917    }

1951     * @param dis - DataInputStream to read the protobuf

1964            RpcErrorCodeProto.FATAL_DESERIALIZING_REQUEST,
1965            "Error decoding " + protoClass.getSimpleName() + ": "+ ioe);

2008    @Override
2009    public void run() {
2010      LOG.debug(Thread.currentThread().getName() + ": starting");
2011      SERVER.set(Server.this);
2012      ByteArrayOutputStream buf = 
2013        new ByteArrayOutputStream(INITIAL_RESP_BUF_SIZE);
2014      while (running) {
2015        TraceScope traceScope = null;
2016        try {
2017          final Call call = callQueue.take(); // pop the queue; maybe blocked here
2018          if (LOG.isDebugEnabled()) {
2019            LOG.debug(Thread.currentThread().getName() + ": " + call + " for RpcKind " + call.rpcKind);
2020          }
2021          if (!call.connection.channel.isOpen()) {
2022            LOG.info(Thread.currentThread().getName() + ": skipped " + call);
2023            continue;
2024          }
2025          String errorClass = null;
2026          String error = null;
2027          RpcStatusProto returnStatus = RpcStatusProto.SUCCESS;
2028          RpcErrorCodeProto detailedErr = null;
2029          Writable value = null;
2030
2031          CurCall.set(call);
2032          if (call.traceSpan != null) {
2033            traceScope = Trace.continueSpan(call.traceSpan);
2034          }
2035
2036          try {
2037            // Make the call as the user via Subject.doAs, thus associating
2038            // the call with the Subject
2039            if (call.connection.user == null) {
2040              value = call(call.rpcKind, call.connection.protocolName, call.rpcRequest, 
2041                           call.timestamp);
2042            } else {
2043              value = 
2044                call.connection.user.doAs
2045                  (new PrivilegedExceptionAction<Writable>() {
2046                     @Override
2047                     public Writable run() throws Exception {
2048                       // make the call
2049                       return call(call.rpcKind, call.connection.protocolName, 
2050                                   call.rpcRequest, call.timestamp);
2051
2052                     }
2053                   }
2054                  );
2055            }
2056          } catch (Throwable e) {
2057            if (e instanceof UndeclaredThrowableException) {
2058              e = e.getCause();
2059            }
2060            String logMsg = Thread.currentThread().getName() + ", call " + call;
2061            if (exceptionsHandler.isTerse(e.getClass())) {
2062              // Don't log the whole stack trace. Way too noisy!
2063              LOG.info(logMsg + ": " + e);
2064            } else if (e instanceof RuntimeException || e instanceof Error) {
2065              // These exception types indicate something is probably wrong
2066              // on the server side, as opposed to just a normal exceptional
2067              // result.
2068              LOG.warn(logMsg, e);
2069            } else {
2070              LOG.info(logMsg, e);
2071            }
2072            if (e instanceof RpcServerException) {
2073              RpcServerException rse = ((RpcServerException)e); 
2074              returnStatus = rse.getRpcStatusProto();
2075              detailedErr = rse.getRpcErrorCodeProto();
2076            } else {
2077              returnStatus = RpcStatusProto.ERROR;
2078              detailedErr = RpcErrorCodeProto.ERROR_APPLICATION;
2079            }
2080            errorClass = e.getClass().getName();
2081            error = StringUtils.stringifyException(e);
2082            // Remove redundant error class name from the beginning of the stack trace
2083            String exceptionHdr = errorClass + ": ";
2084            if (error.startsWith(exceptionHdr)) {
2085              error = error.substring(exceptionHdr.length());
2086            }
2087          }
2088          CurCall.set(null);
2089          synchronized (call.connection.responseQueue) {
2090            // setupResponse() needs to be sync'ed together with 
2091            // responder.doResponse() since setupResponse may use
2092            // SASL to encrypt response data and SASL enforces
2093            // its own message ordering.
2094            setupResponse(buf, call, returnStatus, detailedErr, 
2095                value, errorClass, error);
2096            
2097            // Discard the large buf and reset it back to smaller size 
2098            // to free up heap
2099            if (buf.size() > maxRespSize) {
2100              LOG.warn("Large response size " + buf.size() + " for call "
2101                  + call.toString());
2102              buf = new ByteArrayOutputStream(INITIAL_RESP_BUF_SIZE);
2103            }
2104            responder.doRespond(call);
2105          }
2106        } catch (InterruptedException e) {
2107          if (running) {                          // unexpected -- log it
2108            LOG.info(Thread.currentThread().getName() + " unexpectedly interrupted", e);
2109            if (Trace.isTracing()) {
2110              traceScope.getSpan().addTimelineAnnotation("unexpectedly interrupted: " +
2111                  StringUtils.stringifyException(e));
2112            }
2113          }
2114        } catch (Exception e) {
2115          LOG.info(Thread.currentThread().getName() + " caught an exception", e);
2116          if (Trace.isTracing()) {
2117            traceScope.getSpan().addTimelineAnnotation("Exception: " +
2118                StringUtils.stringifyException(e));
2119          }
2120        } finally {
2121          if (traceScope != null) {
2122            traceScope.close();
2123          }
2124          IOUtils.cleanup(LOG, traceScope);
2125        }
2126      }
2127      LOG.debug(Thread.currentThread().getName() + ": exiting");
2128    }

2252          builder.setServerId(saslRpcServer.serverId);

2254      }
2255    }

2288  /**
2289   * Setup response for the IPC Call.
2290   * 
2291   * @param responseBuf buffer to serialize the response into
2292   * @param call {@link Call} to which we are setting up the response
2293   * @param status of the IPC call
2294   * @param rv return value for the IPC Call, if the call was successful
2295   * @param errorClass error class, if the the call failed
2296   * @param error error message, if the call failed
2297   * @throws IOException
2298   */
2299  private void setupResponse(ByteArrayOutputStream responseBuf,
2300                             Call call, RpcStatusProto status, RpcErrorCodeProto erCode,
2301                             Writable rv, String errorClass, String error) 
2302  throws IOException {
2303    responseBuf.reset();
2304    DataOutputStream out = new DataOutputStream(responseBuf);
2305    RpcResponseHeaderProto.Builder headerBuilder =  
2306        RpcResponseHeaderProto.newBuilder();
2307    headerBuilder.setClientId(ByteString.copyFrom(call.clientId));
2308    headerBuilder.setCallId(call.callId);
2309    headerBuilder.setRetryCount(call.retryCount);
2310    headerBuilder.setStatus(status);
2311    headerBuilder.setServerIpcVersionNum(CURRENT_VERSION);
2312
2313    if (status == RpcStatusProto.SUCCESS) {
2314      RpcResponseHeaderProto header = headerBuilder.build();
2315      final int headerLen = header.getSerializedSize();
2316      int fullLength  = CodedOutputStream.computeRawVarint32Size(headerLen) +
2317          headerLen;
2318      try {
2319        if (rv instanceof ProtobufRpcEngine.RpcWrapper) {
2320          ProtobufRpcEngine.RpcWrapper resWrapper = 
2321              (ProtobufRpcEngine.RpcWrapper) rv;
2322          fullLength += resWrapper.getLength();
2323          out.writeInt(fullLength);
2324          header.writeDelimitedTo(out);
2325          rv.write(out);
2326        } else { // Have to serialize to buffer to get len
2327          final DataOutputBuffer buf = new DataOutputBuffer();
2328          rv.write(buf);
2329          byte[] data = buf.getData();
2330          fullLength += buf.getLength();
2331          out.writeInt(fullLength);
2332          header.writeDelimitedTo(out);
2333          out.write(data, 0, buf.getLength());
2334        }
2335      } catch (Throwable t) {
2336        LOG.warn("Error serializing call response for call " + call, t);
2337        // Call back to same function - this is OK since the
2338        // buffer is reset at the top, and since status is changed
2339        // to ERROR it won't infinite loop.
2340        setupResponse(responseBuf, call, RpcStatusProto.ERROR,
2341            RpcErrorCodeProto.ERROR_SERIALIZING_RESPONSE,
2342            null, t.getClass().getName(),
2343            StringUtils.stringifyException(t));
2344        return;
2345      }
2346    } else { // Rpc Failure
2347      headerBuilder.setExceptionClassName(errorClass);
2348      headerBuilder.setErrorMsg(error);
2349      headerBuilder.setErrorDetail(erCode);
2350      RpcResponseHeaderProto header = headerBuilder.build();
2351      int headerLen = header.getSerializedSize();
2352      final int fullLength  = 
2353          CodedOutputStream.computeRawVarint32Size(headerLen) + headerLen;
2354      out.writeInt(fullLength);
2355      header.writeDelimitedTo(out);
2356    }
2357    if (call.connection.useWrap) {
2358      wrapWithSasl(responseBuf, call);
2359    }
2360    call.setResponse(ByteBuffer.wrap(responseBuf.toByteArray()));
2361  }

2395  private void wrapWithSasl(ByteArrayOutputStream response, Call call)
2396      throws IOException {
2397    if (call.connection.saslServer != null) {
2398      byte[] token = response.toByteArray();
2399      // synchronization may be needed since there can be multiple Handler
2400      // threads using saslServer to wrap responses.
2401      synchronized (call.connection.saslServer) {
2402        token = call.connection.saslServer.wrap(token, 0, token.length);
2403      }
2404      if (LOG.isDebugEnabled())
2405        LOG.debug("Adding saslServer wrapped token of size " + token.length
2406            + " as call response.");
2407      response.reset();
2408      // rebuild with sasl header and payload
2409      RpcResponseHeaderProto saslHeader = RpcResponseHeaderProto.newBuilder()
2410          .setCallId(AuthProtocol.SASL.callId)
2411          .setStatus(RpcStatusProto.SUCCESS)
2412          .build();
2413      RpcSaslProto saslMessage = RpcSaslProto.newBuilder()
2414          .setState(SaslState.WRAP)
2415          .setToken(ByteString.copyFrom(token, 0, token.length))
2416          .build();
2417      RpcResponseMessageWrapper saslResponse =
2418          new RpcResponseMessageWrapper(saslHeader, saslMessage);
2419
2420      DataOutputStream out = new DataOutputStream(response);
2421      out.writeInt(saslResponse.getLength());
2422      saslResponse.write(out);
2423    }
2424  }

2570  /**
2571   * This is a wrapper around {@link WritableByteChannel#write(ByteBuffer)}.
2572   * If the amount of data is large, it writes to channel in smaller chunks. 
2573   * This is to avoid jdk from creating many direct buffers as the size of 
2574   * buffer increases. This also minimizes extra copies in NIO layer
2575   * as a result of multiple write operations required to write a large 
2576   * buffer.  
2577   *
2578   * @see WritableByteChannel#write(ByteBuffer)
2579   */
2580  private int channelWrite(WritableByteChannel channel, 
2581                           ByteBuffer buffer) throws IOException {
2582    
2583    int count =  (buffer.remaining() <= NIO_BUFFER_LIMIT) ?
2584                 channel.write(buffer) : channelIO(null, channel, buffer);
2585    if (count > 0) {
2586      rpcMetrics.incrSentBytes(count);
2587    }
2588    return count;
2589  }

2593   * This is a wrapper around {@link ReadableByteChannel#read(ByteBuffer)}.
2594   * If the amount of data is large, it writes to channel in smaller chunks. 
2595   * This is to avoid jdk from creating many direct buffers as the size of 
2596   * ByteBuffer increases. There should not be any performance degredation.
2597   * 
2598   * @see ReadableByteChannel#read(ByteBuffer)
2599   */
2600  private int channelRead(ReadableByteChannel channel, 
2601                          ByteBuffer buffer) throws IOException {
2602    
2603    int count = (buffer.remaining() <= NIO_BUFFER_LIMIT) ?
2604                channel.read(buffer) : channelIO(channel, null, buffer);
2605    if (count > 0) {
2606      rpcMetrics.incrReceivedBytes(count);
2607    }
2608    return count;
2609  }

2611  /**
2612   * Helper for {@link #channelRead(ReadableByteChannel, ByteBuffer)}
2613   * and {@link #channelWrite(WritableByteChannel, ByteBuffer)}. Only
2614   * one of readCh or writeCh should be non-null.
2615   * 
2616   * @see #channelRead(ReadableByteChannel, ByteBuffer)
2617   * @see #channelWrite(WritableByteChannel, ByteBuffer)
2618   */
2619  private static int channelIO(ReadableByteChannel readCh, 
2620                               WritableByteChannel writeCh,
2621                               ByteBuffer buf) throws IOException {
2622    
2623    int originalLimit = buf.limit();
2624    int initialRemaining = buf.remaining();
2625    int ret = 0;
2626    
2627    while (buf.remaining() > 0) {
2628      try {
2629        int ioSize = Math.min(buf.remaining(), NIO_BUFFER_LIMIT);
2630        buf.limit(buf.position() + ioSize);
2631        
2632        ret = (readCh == null) ? writeCh.write(buf) : readCh.read(buf); 
2633        
2634        if (ret < ioSize) {
2635          break;
2636        }
2637
2638      } finally {
2639        buf.limit(originalLimit);        
2640      }
2641    }
2642
2643    int nBytes = initialRemaining - buf.remaining(); 
2644    return (nBytes > 0) ? nBytes : ret;
2645  }

2647  private class ConnectionManager {
2648    final private AtomicInteger count = new AtomicInteger();    
2649    final private Set<Connection> connections;
2650
2651    final private Timer idleScanTimer;
2652    final private int idleScanThreshold;
2653    final private int idleScanInterval;
2654    final private int maxIdleTime;
2655    final private int maxIdleToClose;

2658    ConnectionManager() {

2660          "IPC Server idle connection scanner for port " + getPort(), true);
2661      this.idleScanThreshold = conf.getInt(

2668          CommonConfigurationKeysPublic.IPC_CLIENT_CONNECTION_MAXIDLETIME_KEY,
2669          CommonConfigurationKeysPublic.IPC_CLIENT_CONNECTION_MAXIDLETIME_DEFAULT);
2670      this.maxIdleToClose = conf.getInt(
2671          CommonConfigurationKeysPublic.IPC_CLIENT_KILL_MAX_KEY,
2672          CommonConfigurationKeysPublic.IPC_CLIENT_KILL_MAX_DEFAULT);
2673      this.maxConnections = conf.getInt(
2674          CommonConfigurationKeysPublic.IPC_SERVER_MAX_CONNECTIONS_KEY,
2675          CommonConfigurationKeysPublic.IPC_SERVER_MAX_CONNECTIONS_DEFAULT);
2676      // create a set with concurrency -and- a thread-safe iterator, add 2
2677      // for listener and idle closer threads
2678      this.connections = Collections.newSetFromMap(
2679          new ConcurrentHashMap<Connection,Boolean>(
2680              maxQueueSize, 0.75f, readThreads+2));
2681    }

2683    private boolean add(Connection connection) {
2684      boolean added = connections.add(connection);
2685      if (added) {
2686        count.getAndIncrement();
2687      }
2688      return added;
2689    }

2691    private boolean remove(Connection connection) {
2692      boolean removed = connections.remove(connection);

2694        count.getAndDecrement();
2695      }
2696      return removed;

2699    int size() {
2700      return count.get();
2701    }
2702
2703    boolean isFull() {
2704      // The check is disabled when maxConnections <= 0.
2705      return ((maxConnections > 0) && (size() >= maxConnections));
2706    }
2707
2708    Connection[] toArray() {
2709      return connections.toArray(new Connection[0]);

2712    Connection register(SocketChannel channel) {
2713      if (isFull()) {
2714        return null;
2715      }
2716      Connection connection = new Connection(channel, Time.now());
2717      add(connection);
2718      if (LOG.isDebugEnabled()) {
2719        LOG.debug("Server connection from " + connection +
2720            "; # active connections: " + size() +
2721            "; # queued calls: " + callQueue.size());

2726    boolean close(Connection connection) {
2727      boolean exists = remove(connection);
2728      if (exists) {
2729        if (LOG.isDebugEnabled()) {
2730          LOG.debug(Thread.currentThread().getName() +
2731              ": disconnecting client " + connection +
2732              ". Number of active connections: "+ size());
2733        }
2734        // only close if actually removed to avoid double-closing due
2735        // to possible races
2736        connection.close();
2737      }
2738      return exists;
2739    }
2740    
2741    // synch'ed to avoid explicit invocation upon OOM from colliding with
2742    // timer task firing
2743    synchronized void closeIdle(boolean scanAll) {
2744      long minLastContact = Time.now() - maxIdleTime;
2745      // concurrent iterator might miss new connections added
2746      // during the iteration, but that's ok because they won't
2747      // be idle yet anyway and will be caught on next scan
2748      int closed = 0;
2749      for (Connection connection : connections) {
2750        // stop if connections dropped below threshold unless scanning all
2751        if (!scanAll && size() < idleScanThreshold) {
2752          break;
2753        }
2754        // stop if not scanning all and max connections are closed
2755        if (connection.isIdle() &&
2756            connection.getLastContact() < minLastContact &&
2757            close(connection) &&
2758            !scanAll && (++closed == maxIdleToClose)) {
2759          break;
2760        }
2761      }
2762    }
2763    
2764    void closeAll() {
2765      // use a copy of the connections to be absolutely sure the concurrent
2766      // iterator doesn't miss a connection
2767      for (Connection connection : toArray()) {
2768        close(connection);
2769      }
2770    }
2771    
2772    void startIdleScan() {
2773      scheduleIdleScanTask();
2774    }
2775    
2776    void stopIdleScan() {
2777      idleScanTimer.cancel();
2778    }
2779    
2780    private void scheduleIdleScanTask() {
2781      if (!running) {
2782        return;
2783      }
2784      TimerTask idleScanTask = new TimerTask(){
2785        @Override
2786        public void run() {
2787          if (!running) {
2788            return;
2789          }
2790          if (LOG.isDebugEnabled()) {
2791            LOG.debug(Thread.currentThread().getName()+": task running");
2792          }
2793          try {
2794            closeIdle(false);
2795          } finally {
2796            // explicitly reschedule so next execution occurs relative
2797            // to the end of this scan, not the beginning
2798            scheduleIdleScanTask();
2799          }
2800        }
2801      };
2802      idleScanTimer.schedule(idleScanTask, idleScanInterval);
2803    }
